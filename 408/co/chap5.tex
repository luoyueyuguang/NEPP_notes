\section{中央处理器（CPU）}

\subsection{CPU 的功能与基本结构}

\begin{mydef}{CPU 的组成}{cpu-structure}
CPU 由\textbf{运算器（ALU）}和\textbf{控制器（CU）}两大部分组成：
\begin{itemize}
    \item \textbf{运算器}：执行算术和逻辑运算。核心部件：ALU、累加器 ACC、通用寄存器组、状态寄存器 PSW（标志位：ZF/CF/OF/SF）。
    \item \textbf{控制器}：取指令、分析指令、执行指令，控制各部件协调工作。核心部件：PC（程序计数器）、IR（指令寄存器）、ID（指令译码器）、时序发生器、微操作信号发生器。
\end{itemize}
CPU 与主存通过 MAR（存储器地址寄存器）和 MDR（存储器数据寄存器）交互。
\end{mydef}

\begin{conclusion}{CPU 中的专用寄存器}{cpu-registers}
\begin{tablebox}{CPU 主要寄存器}
\centering
\begin{tabularx}{\linewidth}{|p{3.6cm}|p{2.0cm}|X|}
\hline
寄存器 & 缩写 & 功能 \\
\hline
程序计数器 & PC & 存放下一条指令的地址，自动 +1（或 + 指令长度）\\
指令寄存器 & IR & 存放当前执行的指令 \\
存储器地址寄存器 & MAR & 存放访存地址（连接地址总线）\\
存储器数据寄存器 & MDR & 存放与主存交换的数据（连接数据总线）\\
累加器 & ACC & 存放 ALU 运算结果 \\
程序状态字 & PSW & 存放标志位（ZF/CF/OF/SF/IF）\\
通用寄存器 & R0-Rn & 存放操作数和中间结果 \\
\hline
\end{tabularx}
\end{tablebox}
\textbf{408 常考：}哪些寄存器程序员可见（通用寄存器、PSW、PC 部分可见），哪些透明（MAR、MDR、IR）。
\end{conclusion}

\subsection{指令执行过程}

\begin{conclusion}{指令周期}{instruction-cycle}
一条指令的指令周期由取指和执行阶段构成，并可能包含间址阶段；若本条指令执行后检测到可响应的中断，还会进入中断周期：
\[
\begin{aligned}
\text{取指（Fetch）} &\to \text{必要时进行间址（Indirect）} \to \text{执行（Execute）}\\
&\to \text{若响应中断，则进入中断周期（Interrupt）}
\end{aligned}
\]
\begin{itemize}
    \item \textbf{取指周期（FE）}：PC$\to$MAR $\to$ 读主存 $\to$ MDR$\to$IR，并把 PC 更新为下一条指令地址。
    \item \textbf{间址周期（IND）}：若指令含间接寻址，取操作数地址。
    \item \textbf{执行周期（EX）}：根据操作码执行具体操作。
    \item \textbf{中断周期（INT）}：保存断点并取得中断服务程序入口；具体由硬件自动完成哪些现场保存操作取决于体系结构。
\end{itemize}

\begin{attention}
\textbf{408 常考数据通路分析题}：给定 CPU 数据通路图和控制信号，写出某条指令（如 \texttt{ADD R1, (R2)}）在\textbf{每个时钟周期}的数据流动路径和有效控制信号。
\end{attention}
\end{conclusion}

\subsection{数据通路}

\begin{mydef}{数据通路}{datapath}
数据通路是 CPU 中执行单元（ALU、寄存器堆、多路选择器）和它们之间的互连总线。分为：
\begin{itemize}
    \item \textbf{单总线结构}：所有部件共享一条内部总线，一个时钟周期只能传送一个数据。结构简单但速度慢。
    \item \textbf{多总线结构}：多条总线并行传送，速度快但控制复杂。
    \item \textbf{专用数据通路}：各部件之间有专用连接，并行度高但硬件代价大。
\end{itemize}
\end{mydef}

\subsection{控制器}

\begin{conclusion}{硬布线控制器 vs 微程序控制器}{hardwire-microprogram}
\begin{tablebox}{两种控制器对比}
\centering
\begin{tabular}{|l|c|c|}
\hline
 & 硬布线控制器 & 微程序控制器 \\
\hline
实现方式 & 组合逻辑电路 & 微程序（控制存储器 CM）\\
速度 & 快 & 较慢（需多次读 CM）\\
可修改性 & 困难（需重新设计电路）& 容易（修改微程序）\\
设计复杂度 & 复杂（需化简逻辑表达式）& 规整 \\
典型应用 & RISC & CISC \\
\hline
\end{tabular}
\end{tablebox}

\textbf{微程序控制器的核心概念：}
\begin{itemize}
    \item \textbf{微指令}：控制存储器中的一个字，控制一个时钟周期内的所有微操作。
    \item \textbf{微程序}：实现一条机器指令的微指令序列。
    \item \textbf{CM（控制存储器）}：存放全部微程序（通常在 CPU 内部 ROM）。
    \item \textbf{$\mu$PC（微程序计数器）}：顺序执行微程序时 +1；遇到转移微指令时加载新地址。
\end{itemize}
\end{conclusion}

\begin{conclusion}{微指令的编码方式}{microinstruction-encoding}
\begin{itemize}
    \item \textbf{直接编码}：微指令的每一位直接控制一个微操作。简单快速，但微指令字长过长。
    \item \textbf{字段直接编码}：互斥的微操作放在同一字段，通过译码器选择。折中方案，408 常考。
    \item \textbf{字段间接编码}：一个字段的含义由另一个字段的值决定。进一步缩短字长但译码更复杂。
\end{itemize}

\textbf{微指令格式：}
\begin{itemize}
    \item \textbf{水平型微指令}：一次能控制多个并行微操作，字长较长。
    \item \textbf{垂直型微指令}：类似机器指令格式，一次只控制一个微操作，需多步完成一条机器指令。
\end{itemize}
\end{conclusion}

\subsection{指令流水线}

\begin{mydef}{流水线}{pipeline}
将指令执行过程分解为多个子过程，每个子过程由独立的硬件部件完成。多条指令的不同阶段在时间上重叠执行，提高 CPU 吞吐率（非单条指令执行速度）。
\end{mydef}

\begin{conclusion}{经典五级流水线}{five-stage-pipeline}
\[
\begin{aligned}
&\text{IF（取指）}\to\text{ID（译码/读寄存器）}\to\text{EX（执行/计算地址）}\\
&\hspace{7em}\to\text{MEM（访存）}\to\text{WB（写回）}
\end{aligned}
\]

\textbf{流水线性能指标：}
\begin{itemize}
    \item \textbf{吞吐率（TP）}：单位时间完成的指令数。理想情况 $TP = \dfrac{1}{\max\{T_i\}}$。
    \item \textbf{加速比（S）}：$S = \dfrac{\text{非流水线执行时间}}{\text{流水线执行时间}}$。$k$ 段流水线执行 $n$ 条指令：$S = \dfrac{n \cdot k \cdot T}{k \cdot T + (n-1) \cdot T} = \dfrac{nk}{k+n-1}$（各段等长时间 $T$）。
    \item \textbf{效率（E）}：$E = \dfrac{\text{时空图中有效面积}}{\text{总面积}}$。
\end{itemize}
\end{conclusion}

\subsection{流水线冒险（Hazard）}

\begin{conclusion}{三种流水线冒险}{pipeline-hazards}
\begin{enumerate}
    \item \textbf{结构冒险（Structural Hazard）}：硬件资源冲突——多条指令同时访问同一功能部件（如单端口存储器）。\textbf{解决：}增加硬件资源（如指令 Cache 与数据 Cache 分离）、流水化功能部件。
    
    \item \textbf{数据冒险（Data Hazard）}：指令间存在数据依赖。类型：
    \begin{itemize}
        \item \textbf{RAW（写后读）}：I1 写入后 I2 才读，但 I2 在 I1 写入前就进入读阶段。
        \item \textbf{WAW（写后写）}：两条指令写同一位置，顺序错。
        \item \textbf{WAR（读后写）}：I2 在 I1 读之前就写入。
    \end{itemize}
    \textbf{解决：}
    \begin{itemize}
        \item \textbf{转发/旁路（Forwarding/Bypassing）}：将 ALU 结果直接从 EX/MEM 流水线寄存器「转发」回 EX 阶段输入，无需等 WB 写回。可解决大部分 RAW。
        \item \textbf{阻塞/气泡（Stall/Bubble）}：硬件插入空操作周期。LW 指令后紧跟使用该数据的指令时，即使转发也需阻塞 1 个周期（Load-use Hazard）。
        \item \textbf{编译器调度}：重排指令顺序以消除依赖。
    \end{itemize}
    
    \item \textbf{控制冒险（Control Hazard）}：分支/跳转指令导致 PC 值不确定。\textbf{解决：}
    \begin{itemize}
        \item \textbf{分支预测}：静态预测（总是预测不跳转）、动态预测（基于历史的分支预测缓冲器 BHT/BTB）。
        \item \textbf{延迟分支}：在分支指令后插入不受影响的指令（延迟槽）。
    \end{itemize}
\end{enumerate}
\end{conclusion}

\begin{attention}
\textbf{408 最常考的数据冒险场景：} 
\begin{enumerate}
    \item 相邻 RAW：\texttt{add R1,R2,R3; sub R4,R1,R5} — 可通过转发解决（无需阻塞）。
    \item Load-use：\texttt{lw R1,0(R2); add R3,R1,R4} — 必须阻塞 1 个周期（转发不够，数据在 MEM 阶段才就绪）。
\end{enumerate}
\end{attention}

\subsection{多核处理器基本概念}

\begin{mydef}{多核处理器}{multi-core}
在一个芯片上集成多个处理器核心（Core），各核心共享最后一级 Cache 和主存，通过片内互连通信。多核实现的是真正的\textbf{并行}（不是并发）。

\textbf{408 了解即可：} SISD/SIMD/MIMD（Flynn 分类法）、共享 Cache 的一致性协议（MESI 协议基本思想）、Amdahl 定律（加速比上限由不可并行部分决定）。
\end{mydef}

\section{习题}

\subsection{基础过关题}

\begin{enumerate}

\item \textbf{（真题风格·选择题）} 下列关于指令流水线设计方法的说法中，\textbf{错误}的是
\begin{center}
\begin{tabular}{ll}
(A) 指令流水线可以提高 CPU 的吞吐率 \\[6pt]
(B) 指令流水线可以提高单条指令的执行速度 \\[6pt]
(C) 指令流水线不能提高单条指令的执行速度 \\[6pt]
(D) 指令流水线可能会增加指令的执行时间
\end{tabular}
\end{center}

\ifshowsolutions\par\noindent\textbf{答案：}(B)

\textbf{解析：} 流水线通过指令间重叠执行提高吞吐率（单位时间完成的指令数），但单条指令的执行时间（latency）不会缩短——流水段寄存器引入的额外开销甚至可能使单条指令延迟略有增加。选项 (C) 正确，(D) 正确（流水段寄存器带来额外时延）。\fi

\item \textbf{（真题风格·选择题）} 下列有关指令流水线的叙述中，\textbf{错误}的是
\begin{center}
\begin{tabular}{ll}
(A) 流水段寄存器会增加单条指令的执行时间 \\[6pt]
(B) 数据冒险可以通过转发技术解决 \\[6pt]
(C) 控制冒险可以通过分支预测来缓解 \\[6pt]
(D) 指令流水线可以消除结构冒险
\end{tabular}
\end{center}

\ifshowsolutions\par\noindent\textbf{答案：}(D)

\textbf{解析：} 结构冒险是由硬件资源冲突引起的（如单端口存储器同时被取指和访存争用），解决手段是增加硬件资源（如分离指令 Cache 和数据 Cache），而非流水线本身能够消除的。转发（forwarding）可解决多数 RAW 数据冒险，分支预测可缓解控制冒险。\fi

\item \textbf{（真题风格·选择题）} 在经典五级顺序流水线（IF、ID、EX、MEM、WB）中，假定已配置常规的 EX/MEM、MEM/WB 转发通路，以下哪种紧邻的数据相关仍需要通过阻塞（插入气泡）解决？
\begin{center}
\begin{tabular}{ll}
(A) 相邻两条 ALU 指令之间的 RAW 冒险 \\[6pt]
(B) Load 指令与其后紧跟的 ALU 指令之间的 RAW 冒险 \\[6pt]
(C) 两条 ALU 指令之间的 WAR 冒险 \\[6pt]
(D) 两条 ALU 指令之间的 WAW 冒险
\end{tabular}
\end{center}

\ifshowsolutions\par\noindent\textbf{答案：}(B)

\textbf{解析：} 相邻 ALU 指令间的 RAW 可通过转发完全解决——结果在 EX 阶段就绪即可转发到下一拍的 EX 输入。(C)(D) WAR 和 WAW 在经典五级流水线中不会发生（按序写回，且读在写之前）。Load-use 冒险中，lw 的数据在 MEM 阶段才就绪，即使转发也只能送到下一拍的 EX 阶段，因此紧跟的 ALU 指令必须阻塞 1 个周期。\fi

\item \textbf{（真题风格·选择题）} 以下关于流水线冒险的说法，\textbf{正确}的是
\begin{center}
\begin{tabular}{ll}
(A) 结构冒险是由于多条指令存在数据依赖引起的 \\[6pt]
(B) 数据冒险中，WAR 和 WAW 在五级流水线中不会发生 \\[6pt]
(C) 分支预测可以彻底消除控制冒险 \\[6pt]
(D) 所有数据冒险都可以通过转发技术消除，无需阻塞
\end{tabular}
\end{center}

\ifshowsolutions\par\noindent\textbf{答案：}(B)

\textbf{解析：} (A) 结构冒险由硬件资源冲突引起，与数据依赖无关。(B) 经典五级流水线按序发射、按序写回，写操作都在 WB 段且只需 1 拍，故 WAR 和 WAW 不可能发生；若引入乱序执行则可能出现。(C) 分支预测只能缓解、不能「彻底」消除控制冒险。(D) Load-use 冒险即使用转发也需阻塞 1 拍。\fi

\item \textbf{（微程序控制器计算题）} 某微程序控制器的控制存储器（CM）容量为 $256\times 32$ 位（$256$ 个微指令字，每字 $32$ 位）。微操作控制字段采用直接编码，其中下地址字段占 $8$ 位，判别测试字段占 $3$ 位，其余位每位直接对应一个微操作控制信号。试回答：
\begin{enumerate}
    \item[(1)] 该微程序控制器最多可控制多少个微操作信号？
    \item[(2)] 假定 256 个字均可用于各机器指令的专属微程序，每条机器指令固定占用 $5$ 条微指令，不另计共享的取指微程序，则最多可支持多少条机器指令？
\end{enumerate}

\ifshowsolutions\par\noindent\textbf{答案：}

(1) 每个微指令字 $32$ 位，下地址字段占 $8$ 位，判别测试字段占 $3$ 位，用于直接控制微操作的位数为 $32-8-3=21$ 位。直接编码中每位控制一个微操作，故最多直接控制 $21$ 个微操作信号。

(2) CM 共 $256$ 条微指令。每条机器指令需 $5$ 条微指令（不含取指公操作），故最大机器指令数 $= 256 \div 5 = 51$（向下取整）。
\fi

\item \textbf{（流水线性能计算——填空题）} 某经典五级流水线处理器，各段执行时间分别为：IF = 10 ns，ID = 8 ns，EX = 10 ns，MEM = 10 ns，WB = 8 ns。
\begin{enumerate}
    \item[(1)] 该流水线的时钟周期应取 \underline{\qquad} ns；
    \item[(2)] 执行 $1000$ 条指令（不发生阻塞）需要 \underline{\qquad} ns；
    \item[(3)] 相比非流水线（每条指令 $10+8+10+10+8=46$ ns），加速比为 \underline{\qquad}（保留两位小数）。
\end{enumerate}

\ifshowsolutions\par\noindent\textbf{答案：}

(1) 时钟周期 $= \max\{10,8,10,10,8\} = 10$ ns。

(2) $k=5$, $n=1000$：总时间 $= kT + (n-1)T = 5\times 10 + 999\times 10 = 10040$ ns。

(3) 非流水线时间 $= n \times (10+8+10+10+8) = 1000 \times 46 = 46000$ ns。加速比 $S = 46000 / 10040 \approx 4.58$。
\fi

\end{enumerate}

\subsection{真题风格综合训练}

\begin{enumerate}[resume]

\item \textbf{（真题风格·选择题）} 某计算机的指令流水线由 4 个功能段组成，各段经过的时间分别为 $\Delta t$、$2\Delta t$、$\Delta t$、$3\Delta t$。忽略流水寄存器开销，连续执行 $n$ 条相同指令，则该流水线相对于顺序执行的加速比为
\begin{center}
\begin{tabular}{ll}
(A) $\dfrac{7n}{3n+9}$ & (B) $\dfrac{7n}{3n+6}$ \\[10pt]
(C) $\dfrac{7n}{3n+3}$ & (D) $\dfrac{7n}{3n+12}$
\end{tabular}
\end{center}

\ifshowsolutions\par\noindent\textbf{答案：}(A)

\textbf{解析：} 非流水线时间 $T_{\text{seq}} = n\times(\Delta t+2\Delta t+\Delta t+3\Delta t)=7n\Delta t$。流水线时钟周期取最长段 $T=\max\{\Delta t,2\Delta t,\Delta t,3\Delta t\}=3\Delta t$。流水线执行时间 $T_{\text{pipe}} = 4T + (n-1)T = (n+3)T = (n+3)\times 3\Delta t = (3n+9)\Delta t$。加速比 $S = \dfrac{7n\Delta t}{(3n+9)\Delta t} = \dfrac{7n}{3n+9} = \dfrac{7n}{3(n+3)}$。

当 $n$ 较大时 $S$ 趋近于 $7/3$，但本题问有限 $n$ 下的精确表达式，不能用极限值或「最接近」选项代替。故选 (A)。
\fi

\item \textbf{（真题风格·选择题）} 在经典五级流水线（IF/ID/EX/MEM/WB）中，考虑以下指令序列：
\begin{lstlisting}
I1: lw  R1, 0(R2)
I2: add R3, R1, R4
I3: sub R5, R6, R7
I4: sw  R8, 0(R9)
\end{lstlisting}
假定寄存器堆允许同一周期先写回、后读出。若仅考虑数据冒险，且不使用转发技术，则 CPU 需要插入的总阻塞（气泡）周期数为
\begin{center}
\begin{tabular}{ll}
(A) $0$ & (B) $1$ \\[6pt]
(C) $2$ & (D) $3$
\end{tabular}
\end{center}

\ifshowsolutions\par\noindent\textbf{答案：}(C)

\textbf{解析：} I1（lw）与 I2（add）之间的 RAW：I1 的数据在 MEM 结束才写入，但 I2 在 EX 阶段就需要 R1。无转发时，lw 的结果在 WB 阶段（第 5 拍）才写回寄存器堆，而 add 在 ID 阶段（第 3 拍）就需要读 R1。因此 I2 必须等到 I1 写回完成后才能 ID，需要阻塞 2 个周期。I2-I4 间无数据依赖。I3 和 I4 之间也无依赖（sw 读 R8，I3 不写 R8；sw 读 R9，无人写 R9）。故仅 I1$\to$I2 间需要 2 个气泡。总阻塞 $=2$。\fi

\item \textbf{（真题风格·综合题）} 某 16 位计算机采用单总线 CPU 结构，数据通路示意图如下（文字描述）：

系统包含：PC、MAR、MDR、IR、通用寄存器 R0--R7、ALU、暂存器 Y 和 Z。ALU 有两个输入端 A 和 B（A 来自暂存器 Y，B 来自总线），一个输出端连接到暂存器 Z。所有部件通过单总线交换数据。PC 具有自增功能（PC $\to$ PC+1）。主存储器的读写由 MDR 和 MAR 配合完成。所有控制信号高电平有效。

回答以下问题：
\begin{enumerate}
    \item[(1)] 写出取指周期的微操作序列（数据流动路径及有效控制信号）。
    \item[(2)] 写出指令 \texttt{ADD R1, R2}（功能：R1 $\gets$ R1 + R2）在\textbf{执行周期}的微操作序列。
    \item[(3)] 写出指令 \texttt{LOAD R3, (R4)}（寄存器间接寻址，功能：R3 $\gets$ Mem[R4]）的\textbf{完整}指令周期微操作（取指 + 执行）。
\end{enumerate}

\ifshowsolutions\par\noindent\textbf{答案：}

(1) 取指周期微操作序列（单总线，每次只能传送一个数据）：
\begin{enumerate}[label=T$\arabic*$:,leftmargin=*]
    \item PC $\to$ MAR, PC+1 $\to$ PC（PC 送出地址到 MAR，同时自增）
    \item MEM(MAR) $\to$ MDR（读主存）
    \item MDR $\to$ IR（指令码送指令寄存器）
\end{enumerate}
控制信号：PCout, MARin, PCinc; MemRead, MDRin; MDRout, IRin。

(2) \texttt{ADD R1, R2} 执行周期：
\begin{enumerate}[label=T$\arabic*$:,leftmargin=*]
    \item R2 $\to$ Y（通过总线送一个操作数到 ALU 的 A 端暂存器 Y）
    \item R1 $\to$ B（送另一操作数到 ALU 的 B 端），ALU 做加法，结果 $\to$ Z（暂存到 Z）
    \item Z $\to$ R1（结果从 Z 写回 R1）
\end{enumerate}
控制信号：R2out, Yin; R1out, ALU\_add, Zin; Zout, R1in。

(3) \texttt{LOAD R3, (R4)} 的完整周期：
\begin{itemize}
    \item \textbf{取指周期（同(1)）：} 3 拍。
    \item \textbf{执行周期：}
    \begin{enumerate}[label=T$\arabic*$:,leftmargin=*]
        \item R4 $\to$ MAR
        \item MEM(MAR) $\to$ MDR（读主存）
        \item MDR $\to$ R3（数据写入目标寄存器）
    \end{enumerate}
\end{itemize}
共计 6 拍。\texttt{(R4)} 是寄存器间接寻址，R4 已直接给出有效地址，不需要再从主存读取一次「间接地址」，因此没有独立的间址周期；R4 $\to$ MAR 是该 LOAD 指令执行周期的一部分。注意：本题假定一次主存读在一拍内完成。
\fi

\item \textbf{（综合题·指令流水线分析）} 某经典五级流水线处理器（IF/ID/EX/MEM/WB），执行以下指令序列：
\begin{lstlisting}
I1: add  R1, R2, R3    % R1 <- R2 + R3
I2: lw   R4, 0(R1)     % R4 <- Mem[R1 + 0]
I3: sub  R5, R1, R6    % R5 <- R1 - R6
I4: add  R7, R4, R5    % R7 <- R4 + R5
I5: sw   R7, 0(R8)     % Mem[R8 + 0] <- R7
\end{lstlisting}

\begin{enumerate}
    \item[(1)] 指出上述指令序列中存在的所有数据冒险，说明冒险类型。
    \item[(2)] 若\textbf{不存在}转发机制，需要插入多少个阻塞周期？画出流水线时空图。
    \item[(3)] 若有完整的数据转发（包括向存储指令的数据输入转发），但紧邻的 Load-use 冒险仍需阻塞 1 拍，需要插入多少个阻塞周期？
    \item[(4)] 能否在保持程序语义的前提下交换 I3 与 I4？说明理由。
\end{enumerate}

\ifshowsolutions\par\noindent\textbf{答案：}

(1) 数据冒险：
\begin{itemize}
    \item I1 $\to$ I2（RAW on R1）：I1 在 EX 段末产生 R1，I2 在 EX 段初需要 R1。有转发 $\to$ 转发可解决；无转发 $\to$ 需等待 I1 的 WB。
    \item I1 $\to$ I3（RAW on R1）：I1 在 EX 段末产生 R1，I3 在 EX 段初需要 R1。同上。
    \item I2 $\to$ I4（RAW on R4）：I2 在 MEM 段末获得 R4，I4 在 EX 段需要 R4。二者之间隔着 I3，因此在完整转发下不需阻塞；若 I4 紧跟 I2，才构成需要 1 拍阻塞的 Load-use 冒险。
    \item I3 $\to$ I4（RAW on R5）：I3 在 EX 段末产生 R5，I4 在 EX 段初需要 R5。有转发可解决；无转发需等待 WB。
\end{itemize}

(2) 无转发时，消费者必须到 ID 阶段读取到生产者已写回的值。假定 WB 前半拍写、ID 后半拍读，则逐周期安排为：
\begin{center}
\small
\begin{tabular}{|c|l|}
\hline
指令 & 所在阶段（数字为周期；ID 后的 $*$ 表示因相关而停留） \\
\hline
I1 & IF1, ID2, EX3, MEM4, WB5 \\
I2 & IF2, ID3$^*$, ID4$^*$, ID5, EX6, MEM7, WB8 \\
I3 & IF3--5（随前端冻结）, ID6, EX7, MEM8, WB9 \\
I4 & IF6, ID7$^*$, ID8$^*$, ID9, EX10, MEM11, WB12 \\
I5 & IF7--9（随前端冻结）, ID10$^*$, ID11$^*$, ID12, EX13, MEM14, WB15 \\
\hline
\end{tabular}
\end{center}
I2 等待 I1 写 R1 产生 2 拍阻塞；I4 同时等待 I2 的 R4 和 I3 的 R5，最终由较晚的 I3 写回决定，共产生 2 拍，而不是把两条相关分别相加；I5 等待 I4 写 R7 再产生 2 拍。总阻塞为 $2+2+2=6$ 拍。

(3) 有完整转发时，I1$\to$I2、I1$\to$I3、I3$\to$I4 和 I4$\to$I5 均可旁路。I2 与 I4 之间已有一条 I3，I2 的装入数据在 I4 进入 EX 前已经可用，因此不是「紧邻 Load-use」，总阻塞为 0 拍。

(4) 不能交换。I3 写 R5，而 I4 读 R5，二者之间存在真数据相关（RAW）。若令 I4 先执行，它会读到 R5 的旧值并改变 R7 以及随后存入主存的数据，程序语义发生变化。指令调度只能跨越不存在此类依赖的独立指令。
\fi

\end{enumerate}
