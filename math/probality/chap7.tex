\section{参数统计与假设检验}
\subsection{参数统计}
\begin{mydef}{估计量，估计值与点估计}{estimator, estimate value and point estimate}
    从总体\(X\)中抽取样本\(X_1,X_2,\cdots, X_n\), 相应的样本值\(x_1,x_2,\cdots,x_n\),
    构造一个适当的统计量\(\widehat{\theta}(X_1, X_2, \cdots , X_n)\),
    则\(\widehat{\theta}(X_1, X_2, \cdots , X_n)\)称为\(X\)的一个估计量,
    相应的\(\widehat{\theta}(x_1, x_2, \cdots, x_n)\)称为\(X\)的一个估计值,
    而\(\widehat{\theta}(x_1, x_2, \cdots, x_n)\)是\(X\)的一个点估计.
\end{mydef}

\begin{mydef}{矩估计}{moment estimate}
    设\(\theta\)是\(X\)的一个未知参数, 若\(\widehat{\theta}\)是\(X\)的一个矩估计,
    则\(\widehat{\theta}\)称为\(X\)的一个矩估计量.
    \begin{enumerate}
        \item 计算总体\(X\)的前\(k\)阶原点矩
        \item 计算样本\(X_1,X_2,\cdots, X_n\)的前\(k\)阶原点矩，令样本矩=总体矩，
              即得
              \[
              \frac{1}{n} \sum_{i=1}^n X_i^k = E(X^l)
              \]
        \item 求解上述方程, 即可得\(\widehat{\theta}\)的矩估计量.
    \end{enumerate}
\end{mydef}

\begin{mydef}{最大似然估计}{maximum likelihood estimate}
    设\(X_1,X_2,\cdots, X_n\)是\(X\)的一个样本, 则\(X_1,X_2,\cdots, X_n\)的最大似然估计量为
    \[
        \widehat{\theta} = \arg\max_{\theta} L(\theta) = \arg\max_{\theta} \prod_{i=1}^n f(x_i; \theta)
    \]
    其中\(L(\theta)\)称为\(X_1,X_2,\cdots, X_n\)的最大似然函数.
    求解步骤为
    \begin{enumerate}
        \item 写出\(L(\theta)\)的函数表达式.
        \item 对\(L(\theta)\)或\(\ln L(\theta)\)求导，
              并令导数等于零，
              即可得\(\widehat{\theta}\)的最大似然估计量.若求导无解，则利用单调性来求
              解.
    \end{enumerate}
\end{mydef}

\begin{conclusion}{最大似然估计的性质}{maximum likelihood estimate property}
    最大似然估计具有不变性, 即若\(u(\theta)\)是\(\theta\)的一个函数, 则\(\widehat{\theta}\)
    也是\(u(\widehat{\theta})\)的最大似然估计\\
    \boxed{\text{例}} \\
    \[
    \widehat{\sigma^2} = \frac{1}{n}\sum_{i}^{n}(X_i - \bar{X})^2
    \]可得
    \[
    \widehat{\sigma} = \sqrt{\frac{1}{n}\sum_{i}^{n}(X_i - \bar{X})^2}
    \]
\end{conclusion}

\begin{conclusion}{估计量的评选标准}{estimator selection standard}
    \begin{enumerate}
        \item 无偏性，即\(E(\widehat{\theta}) = E(\theta)\)
        \item 有效性，即若\(D(\widehat{\theta_1}) \leq D(\widehat{\sigma_2})\),则称\(\widehat{\theta_1}\)
              比\(\widehat{\theta_2}\)更有效.
        \item 一致性，即当\(n \to \infty\)时，\(\widehat{\theta_n}\)
              依概率收敛于\(\theta\), 则称\(\widehat{\theta_n}\)是\(\theta\)的一个一致估计量.
    \end{enumerate}
\end{conclusion}

\begin{mydef}{区间估计}{interval estimate}
    区间估计就是用区间来刻画未知参数所在的范围, 即\(P(\theta \in I) = 1 - \alpha\),
    其中\(I\)是一个区间, \(\alpha\)是一个给定的显著性水平.
    \begin{enumerate}
        \item 单侧置信区间 \\
        对于给定值\(\alpha\), 单侧置信区间为
        \[
        P(\theta \leq \widehat{\theta}) = 1 - \alpha
        \]
        \item 双侧置信区间 \\
        对于给定值\(\alpha\), 双侧置信区间为
        \[
        P(\theta \in I) = 1 - \alpha
        \]即
        \[
        P(\theta_1 < \theta < \theta_2) = 1 - \alpha
        \]
    \end{enumerate}
\end{mydef}

\begin{conclusion}{对区间估计的一些理解}{interval estimate understanding}
    正态总体均值或方差总体还是依据估计及分布来求，或许可以才这方面入手来推出均值差，方差比的置信区间.\\
    总的来说，区间估计本质还是估计，还是根据统计量来求 \\
    我们根据前文所提的样本统计量来进行区间估计 \\
    \boxed{例} \\
    当\(\sigma^2\)已知时，使用统计量
    \[
    \frac{\overline{X} - \mu}{\frac{\sigma}{\sqrt{n}}}
     \sim N(0,1)
    \]
    当\(\sigma^2\)未知时，使用统计量
    \[
    \frac{\overline{X} - \mu}{s\sqrt{n}}
    \sim t(n-1)
    \]
    设统计量为\(L\)，则根据单边或双边写出矩法方程
    \[
    |L| < k \text{或} L > k \text{或} L < k
    \]
    其中\(k\)不同情况下的\(\alpha\)分位点 \\
    下表给出了常见的单正态总体均值方差的置信区间 \\
    双正态根据前文的双正态总体的统计量来求
\end{conclusion}

\subsection{假设检验}
\begin{mydef}{假设检验}{hypothesis test}
    假设检验就是根据样本数据来判断总体参数是否等于某个特定值的过程.
\end{mydef}

\begin{mydef}{原假设与备择假设}{null hypothesis and alternative hypothesis}
    原假设\(H_0\)是指总体参数满足某个条件的假设, 而备择假设\(H_1\)是指与\(H_0\)对立的假设
\end{mydef}

\begin{mydef}{显著性水平与显著性检验}{significance level and significance test}
    显著性水平\(\alpha\)是指在假设检验中, 拒绝原假设的概率.
    若\(P(\text{拒绝 }H_0 | H_0 \text{ 为真}) = \alpha\), 则称\(\alpha\)是检验的显著性水平.
\end{mydef}

\begin{conclusion}{假设检验的一般步骤}{hypothesis test general steps}
    \begin{enumerate}
        \item 确定原假设与备择假设
        \item 确定显著性水平
        \item 计算检验统计量
        \item 确定拒绝区域
        \item 计算p值
        \item 做出判断
    \end{enumerate}
\end{conclusion}
