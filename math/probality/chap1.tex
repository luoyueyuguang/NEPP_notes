\section{随机事件及其概率}
\subsection{随机事件}

\begin{mydef}{随机试验}{random experiment}
    随机试验满足以下三个条件：
    \begin{enumerate}
        \item 可以在相同的条件下重复进行；
        \item 每次试验的结果不确定；
        \item 所有可能的结果都明确可知且不止一个。
    \end{enumerate}
\end{mydef}

\begin{mydef}{样本空间}{sample space}
    样本空间是指随机试验所有可能结果的集合，通常用 $\Omega$ 表示。
\end{mydef}

\begin{mydef}{随机事件}{random event}
    随机事件是样本空间中属于给定事件域（$\sigma$-代数）的集合，通常用大写字母表示，如 $A, B, C$ 等。在有限或可列的初等模型中，通常可把样本空间的所有子集都作为事件。
    \newline
    \newline
    每次试验中必定发生的事件称为必然事件，记作 $\Omega$；不可能发生的事件称为不可能事件，记作 $\emptyset$。
\end{mydef}

\subsection{事件的运算法则}
\begin{formula}{事件的运算法则}{laws of event operations}
    \begin{enumerate}
        \item 交换律：$A \cup B = B \cup A$，$A \cap B = B \cap A$
        \item 结合律：$A \cup (B \cup C) = (A \cup B) \cup C$，$A \cap (B \cap C) = (A \cap B) \cap C$
        \item 分配律：$A \cup (B \cap C) = (A \cup B) \cap (A \cup C)$，$A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$
        \item 德摩根定律：$(A \cup B)^c = A^c \cap B^c$，$(A \cap B)^c = A^c \cup B^c$
        \item 吸收律 ：$\text{若} A \subset B,\text{则} A \cup B = B$，$A \cap B = A$
        \item 对立律：$A \cup A^c = \Omega$，$A \cap A^c = \emptyset$
        \item 蕴涵律：$A \cap B = \emptyset \Rightarrow A \subset B^c$，$B \subset A^{c}$
        \item 差积律：$A - B = A \cap B^c = A - (A \cap B)$
    \end{enumerate}
\end{formula}

\subsection{概率的定义}
\begin{mydef}{概率的描述性定义}{descriptive definition of probability}
    概率是指在大量重复的随机试验中，某事件发生的频率趋近于一个固定值，这个固定值即为该事件的概率。
\end{mydef}

\begin{mydef}{概率的公理化定义}{axiomatic definition of probability}
    概率是定义在样本空间 $\Omega$ 的事件域 $\mathcal F$ 上的集合函数 $P:\mathcal F\to[0,1]$，满足以下公理：
    \begin{enumerate}
        \item 非负性：对于任意事件 $A$，$P(A) \geq 0$。
        \item 归一性：$P(\Omega) = 1$。
        \item 可列可加性：对于任意互不相交的事件序列 $\{A_i\}$，有 $P(\cup_{i=1}^{\infty} A_i) = \sum_{i=1}^{\infty} P(A_i)$。
    \end{enumerate}
\end{mydef}

\begin{mydef}{概率的统计定义}{statistical definition of probability}
    概率是指在大量重复的随机试验中，某事件发生的相对频率趋近于一个固定值，这个固定值即为该事件的概率。
\end{mydef}

\begin{mydef}{概率的几何定义}{geometric definition of probability}
    当样本点相对于长度、面积或体积测度在有限测度区域 $\Omega$ 上均匀分布时，事件 $A$ 的概率可表示为
    $P(A)=\dfrac{m(A)}{m(\Omega)}$。若分布不均匀，则应按相应概率密度积分，不能直接使用测度之比。
\end{mydef}

\begin{mydef}{古典概型}{classical probability model}
    古典概型是指在有限样本空间中，每个基本事件发生的概率相等的情况。设样本空间 $\Omega$ 包含 $n$ 个基本事件，事件 $A$ 包含 $m$ 个基本事件，则事件 $A$ 的概率为：
    \[
    P(A) = \frac{m}{n}
    \]
\end{mydef}

\begin{formula}{古典概型的计算方法}{classical probability calculation}
    \begin{enumerate}
        \item 随机分配问题
        \newline
        将 $n$ 个不同物品分别放入 $m$ 个不同盒子，每个盒子的容量不限。若每个物品都可独立选择任一盒子，则共有
        \[
        m^n
        \]
        种放法。若另加“每盒至多放一个物品”的限制（要求 $n\leq m$），则共有
        \[
        A_m^n=\frac{m!}{(m-n)!}
        \]
        种放法。计数前必须先分清物品、盒子是否可区分，以及是否允许空盒、是否限制盒子容量；不同模型不能混用同一个公式。
        \item 简单随机抽样问题
        \newline
        $\Omega$ 包含 $N$ 个元素，如果各元素被选中的概率相等，则 $\Omega$ 的抽样被称作简单随机抽样。
        \newline
        \begin{tablebox}
        {简单随机抽样的计算方法}
        \centering
        \begin{tabular}{|c|c|p{4cm}|}
        \hline
        抽样方式 & 公式 & 说明 \\
        \hline
        有放回有序抽样 & $N^n$ & 每次抽样后元素仍然可选，依次选择 $n$ 个元素 \\
        \hline
        无放回有序抽样 & $A_{N}^{n} = \frac{N!}{(N-n)!}$ & 每次抽样后元素不再可选，依次选择 $n$ 个元素 \\
        \hline
        无放回抽样（组合） & $C_{N}^{n} = \frac{N!}{n!(N-n)!}$ & 每次抽样后元素不再可选，选择 $n$ 个元素且不考虑顺序 \\
        \hline
        \end{tabular}
        \label{tab:simple_random_sampling}
        \end{tablebox}
        \item 其他方法
        \newline
        简单问题也应先判断是否放回、是否计序，再选择直接列举或排列组合方法。
        \begin{enumerate}
            \item 直接列举法：对于小规模的样本空间，可以直接列举所有可能的事件。
            \item 排列组合法：对于有序或无序的事件，可以使用排列组合的方法来计算概率。
            \item 容斥原理：对于多个事件的并集，可以使用容斥原理来计算概率。
        \end{enumerate}
        \item \boxed{\textbf{例}}
        将 $n$ 个不同物品放入 $m$ 个不同盒子：
        \begin{enumerate}
            \item 每盒容量不限：每个物品有 $m$ 种选择，共有 $m^n$ 种放法。
            \item 每盒至多一个物品：当 $n\leq m$ 时，相当于从 $m$ 个盒子中有序选出 $n$ 个，共有 $A_m^n$ 种放法。
            \item 若物品或盒子不可区分，计数模型将发生变化，不能直接套用上述两个公式。
        \end{enumerate}
    \end{enumerate}
\end{formula}

\begin{mydef}
{几何概型}{geometric probability}
    几何概型是指在几何空间中，事件的概率通过测度来定义。通常用于处理连续样本空间中的事件。
\end{mydef}

\begin{formula}{几何概型的计算方法}{geometric probability calculation}
    若样本点相对于测度 $m$ 在 $\Omega$ 上均匀分布，且 $0<m(\Omega)<\infty$，则事件 $A$ 的概率为：
    \[
    P(A) = \frac{m(A)}{m(\Omega)}
    \]
    其中，$m(A)$ 和 $m(\Omega)$ 分别表示事件 $A$ 和样本空间 $\Omega$ 的测度。

    \boxed{
        \textbf{
            落到计算上，就是长度、面积或体积的比值。
            这可以和高数结合起来
        }
    }

\end{formula}

\subsection{概率论性质与公式}

\begin{conclusion}{概率论的性质}{properties of probability theory}
    \begin{enumerate}
        \item 有界性\\
        对于任一事件 $A$，有 $0 \leq P(A) \leq 1$，且 $P(\emptyset) = 0, P(\Omega) = 1$。
        \item 单调性\\
        如果 $A \subset B$，则 $P(B-A) = P(B) - P(A)$，且 $P(B) \geq P(A)$。
    \end{enumerate}
\end{conclusion}

\begin{formula}{概率论的公式}{formulas of probability theory}
    \begin{enumerate}
        \item 加法公式\\
        如果 $A_1, A_2, \ldots, A_n$ 是互斥的事件，则 $P(A_1 \cup A_2 \cup \ldots \cup A_n) = P(A_1) + P(A_2) + \ldots + P(A_n)$。
        \item 求逆公式\\
        $P(\bar{A}) = 1 - P(A)$。
        \item 减法公式 \\
        如果 $A$ 是事件 $B$ 的子集，则 $P(B-A) = P(B) - P(A)$。
        \item 乘法公式 \\
        $P(A_1 \cap A_2 \cap \ldots \cap A_n) = P(A_1)P(A_2|A_1)P(A_3|A_1 \cap A_2) \cdots P(A_n|A_1 \cap \ldots \cap A_{n-1})$。特别地，如果 $A_1, A_2, \ldots, A_n$ 相互独立，则 $P(A_1 \cap A_2 \cap \ldots \cap A_n) = P(A_1)P(A_2) \cdots P(A_n)$。
        \item 条件概率公式 \\
        如果 $P(B) > 0$，则 $P(A|B) = \frac{P(A \cap B)}{P(B)}$。
        \item 全概率公式 \\
        设 $B_1, B_2, \ldots, B_n$ 为完备事件组（互斥且 $\bigcup_{i=1}^{n} B_i = \Omega$），$P(B_i) > 0$，则 $P(A) = \sum_{i=1}^{n} P(B_i)P(A|B_i)$。
        \item 贝叶斯公式 \\
        设 $\{A_i\}$ 是完备事件组（互斥且 $\bigcup_i A_i = \Omega$），$P(A_i) > 0$，且 $P(B)>0$，则\\
        $P(A_{k} | B) =  \frac{P(A_{k})P(B|A_{k})}{\sum_{i} P(A_i)P(B|A_i)}$
        \item n重伯努利试验 \\
        $P_{n}(A = k) = C_{n}^{k}p^{k}(1-p)^{n-k}$
    \end{enumerate}
\end{formula}

\subsection{事件的独立性}

\begin{mydef}{事件的独立性}{independence of events}
    如果 $A$ 和 $B$ 是事件，且 $P(A \cap B) = P(A)P(B)$，则称 $A$ 和 $B$ 是独立的。
    直观上可理解为$A$与$B$在概率上互不影响；严格定义仍是上式。只有在相应条件概率有定义时，才能等价地表述为一个事件的发生不改变另一个事件的概率。
\end{mydef}

\begin{conclusion}{互不相容}{mutually exclusive}
    如果 $A$ 和 $B$ 是事件，且 $A \cap B = \emptyset$，则称 $A$ 和 $B$ 是互不相容的。
    互不相容只表示两个事件不能同时发生，并不表示二者“概率上不相关”或相互独立。特别地，当 $P(A)>0$、$P(B)>0$ 时，互不相容反而意味着二者不独立。
\end{conclusion}

\begin{mydef}{独立与不相容的联系}{independence and mutually exclusive}
    $P(A) > 0, P(B) > 0, A\textrm{与}B\textrm{独立} \Rightarrow A\textrm{与}B\textrm{相容}$ \\
    $A\textrm{与}B\textrm{互不相容},P(A) >0, P(B) > 0 \Rightarrow A \textrm{与}B\textrm{不互相独立}$
\end{mydef}

\begin{conclusion}{四对事件}{four in}
    若$A$与$B$, $\bar{A}$与$B$,$A$与$\bar{B}$,$\bar{A}$与$\bar{B}$有一对相互独立，
    则其他三对相互独立
    
\end{conclusion}

\begin{conclusion}{事件A与B相互独立的充要条件}{necessary and sufficient conditions for independence}
    %在左边
    % \vspace{
    % $A\textrm{与}B\textrm{独立} \Leftrightarrow$
    % }
    % %在右边
    % \[
    % \begin{cases}
    %     P(AB) = P(A)P(B) \\
    %     P(B|A) = P(B),\quad P(A) > 0 \\
    %     P(A | B) = P(A),\quad P(B) > 0 \\
    %     P(B|A) = P(B | \hat{A}),\quad 0 < P(A) < 1
    % \end{cases}
    % \]
    \[
% 左边文本
\parbox{6.5em}{% 调整宽度使文本垂直居中
    $A$与$B$独立 $\Leftrightarrow$
}
% 右边条件组
\quad
\begin{cases}
    1.P(AB) = P(A)P(B) \\
    2.P(B|A) = P(B),\quad P(A) > 0 \\
    P(A | B) = P(A),\quad P(B) > 0 \\
    3.P(B|A) = P(B | \bar{A}),\quad 0 < P(A) < 1\\
    4.P(B|A) + P(\bar{B}|\bar{A}) = 1,\quad 0 < P(A) < 1
\end{cases}
\]
\end{conclusion}

\begin{conclusion}{事件A,B,C相互独立的条件}{necessary and sufficient conditions for independence2}
    $$
    \begin{cases}
       P(AB) = P(A)P(B) \\
       P(AC) = P(A)P(C) \\
       P(BC) = P(B)P(C) \\
       P(ABC) = P(A)P(B)P(C) \\
    \end{cases}
    \Leftrightarrow A,B,C\textrm{相互独立}
    $$
\end{conclusion}

\begin{conclusion}{n个独立事件的概率}{probability of n independent events}
    若$A_1,A_2,\ldots,A_n$相互独立，则
    $
    P\!\left(\bigcap_{i=1}^{n}A_i\right) = \prod_{i=1}^{n}P(A_i)
    $
    \\
    $
    P\!\left(\bigcup_{i=1}^{n}A_i\right) = 1 - \prod_{i=1}^{n}[1-P(A_i)]
    $
\end{conclusion}

\begin{conclusion}{独立的性质}{properties of independence}
   若$A_1,A_2,\ldots,A_n$独立，则
   由$A_{i1},A_{i2},\ldots,A_{in}$经过运算后所得事件$\sigma_{1}(
    A_{i1},A_{i2},\ldots,A_{in}
   )$
   与$A_{ik+1},A_{ik+2},\ldots,A_{in}$经过运算后所得事件$\sigma_{2}(
    A_{ik+1},A_{ik+2},\ldots,A_{in}
   )$也独立, $i1,i2,\ldots,in$为$1,2,\ldots,n$的任何一个排列。
   \\
   即原本独立的事件，各自经过运算后所得事件也独立
\end{conclusion}

\section{习题}

\subsection{基础过关题}

\begin{enumerate}

\item \textbf{（填空题）} 设 $A$, $B$, $C$ 是三个随机事件，则：
\begin{enumerate}[label=(\arabic*)]
    \item ``三个事件中恰有两个发生''表示为 \underline{\qquad};
    \item ``三个事件中至少有两个发生''表示为 \underline{\qquad};
    \item ``三个事件中至少有一个不发生''表示为 \underline{\qquad}.
\end{enumerate}

\ifshowsolutions%
\boxed{\textbf{解}}
(1) $AB\bar{C} \cup A\bar{B}C \cup \bar{A}BC$;
(2) $AB \cup AC \cup BC$;
(3) $\bar{A} \cup \bar{B} \cup \bar{C}$（或 $\overline{ABC}$）.

\fi%
\item \textbf{（计算题）} 袋中有 $5$ 个红球和 $3$ 个白球，外观相同。现不放回地依次取出两个球。
\begin{enumerate}[label=(\arabic*)]
    \item 求第二次取到红球的概率；
    \item 已知第二次取到了红球，求第一次也取到红球的概率.
\end{enumerate}

\ifshowsolutions%
\boxed{\textbf{解}} 设 $A_i = \{\text{第 }i\text{ 次取到红球}\}$，$i=1,2$.

(1) 由全概率公式：
\[
P(A_2) = P(A_1)P(A_2 \mid A_1) + P(\bar{A}_1)P(A_2 \mid \bar{A}_1)
= \frac{5}{8} \cdot \frac{4}{7} + \frac{3}{8} \cdot \frac{5}{7}
= \frac{20 + 15}{56} = \frac{35}{56} = \frac{5}{8}.
\]

(2) 由条件概率定义：
\[
P(A_1 \mid A_2) = \frac{P(A_1 A_2)}{P(A_2)}
= \frac{\frac{5}{8} \cdot \frac{4}{7}}{\frac{5}{8}} = \frac{4}{7}.
\]

\textbf{评注：} 第(1)问的结果等于初始红球比例 $\frac{5}{8}$，这不是巧合——抽签问题中，不放回情形下各人抽中的概率等于初始比例，与抽签顺序无关。

\fi%
\item \textbf{（计算题）} 设 $P(A) = 0.5$，$P(B) = 0.6$，$P(B \mid A) = 0.8$. 求：
\begin{enumerate}[label=(\arabic*)]
    \item $P(AB)$；
    \item $P(A \cup B)$；
    \item $P(A \mid B)$.
\end{enumerate}

\ifshowsolutions%
\boxed{\textbf{解}}
(1) $P(AB) = P(A)P(B \mid A) = 0.5 \times 0.8 = 0.4$.

(2) $P(A \cup B) = P(A) + P(B) - P(AB) = 0.5 + 0.6 - 0.4 = 0.7$.

(3) $P(A \mid B) = \dfrac{P(AB)}{P(B)} = \dfrac{0.4}{0.6} = \dfrac{2}{3}$.

\fi%
\item \textbf{（计算题）} 从一副去掉大小王的 $52$ 张扑克牌中随机抽取 $5$ 张。求：
\begin{enumerate}[label=(\arabic*)]
    \item $5$ 张牌为同花（即花色全部相同）的概率；
    \item $5$ 张牌中至少有一张 A 的概率.
\end{enumerate}

\ifshowsolutions%
\boxed{\textbf{解}} 样本空间为从 $52$ 张牌中任选 $5$ 张，共 $C_{52}^5$ 种等可能结果。

(1) 同花：先选一种花色（$C_4^1$ 种），再从该花色的 $13$ 张牌中选 $5$ 张（$C_{13}^5$ 种）：
\[
P = \frac{C_4^1 \cdot C_{13}^5}{C_{52}^5}
= \frac{4 \times 1287}{2598960}
= \frac{5148}{2598960} \approx 0.00198.
\]

(2) 设 $D = \{\text{没有 A}\}$，则 $|D| = C_{48}^5$（从不含 A 的 $48$ 张中选 $5$ 张）：
\[
P(\text{至少一张 A}) = 1 - P(D)
= 1 - \frac{C_{48}^5}{C_{52}^5}
= 1 - \frac{1712304}{2598960}
\approx 0.3412.
\]

\fi%
\end{enumerate}

\subsection{综合提高题}

\begin{enumerate}[resume]

\item \textbf{（计算题，$\star\star$ 全概率 + 贝叶斯）} 某产品由三个工厂生产，工厂 1、2、3 的产量分别占总产量的 $25\%$、$35\%$、$40\%$，各工厂产品的次品率分别为 $5\%$、$4\%$、$2\%$. 现从市场上随机抽取一件该产品。
\begin{enumerate}[label=(\arabic*)]
    \item 求抽到次品的概率；
    \item 已知抽到的是次品，求它来自工厂 1 的概率.
\end{enumerate}

\ifshowsolutions%
\boxed{\textbf{解}} 设 $B = \{\text{抽到次品}\}$，$A_i = \{\text{产品来自工厂 }i\}$，$i=1,2,3$.
已知 $P(A_1)=0.25$，$P(A_2)=0.35$，$P(A_3)=0.40$，
$P(B \mid A_1)=0.05$，$P(B \mid A_2)=0.04$，$P(B \mid A_3)=0.02$.

(1) 由全概率公式：
\[
\begin{aligned}
P(B)
&=\sum_{i=1}^3P(A_i)P(B\mid A_i)\\
&=0.25\times0.05+0.35\times0.04+0.40\times0.02\\
&=0.0125+0.0140+0.0080=0.0345.
\end{aligned}
\]

(2) 由贝叶斯公式：
\[
P(A_1 \mid B) = \frac{P(A_1)P(B \mid A_1)}{P(B)}
= \frac{0.25 \times 0.05}{0.0345}
= \frac{0.0125}{0.0345} = \frac{25}{69} \approx 0.3623.
\]

\textbf{评注：} 贝叶斯公式的本质是``执果索因''——已知结果（次品），反推原因（来自哪个工厂）的概率。工厂 1 的次品率最高，先验概率仅有 $25\%$，后验概率升至约 $36.23\%$，体现了数据对先验信念的修正。

\fi%
\item \textbf{（判断题，$\star\star$ 独立性的深层理解）} 设 $P(A) > 0$, $P(B) > 0$. 判断以下命题的真伪并简要说理：
\begin{enumerate}[label=(\arabic*)]
    \item 若 $A$ 与 $B$ 互不相容，则 $A$ 与 $B$ 必不独立；
    \item 若 $A$ 与 $B$ 独立，则 $A$ 与 $B$ 必相容（即 $AB \neq \emptyset$）；
    \item $P(A \mid B) = P(A \mid \bar{B})$ 是 $A$ 与 $B$ 独立的充要条件（设 $0 < P(B) < 1$）.
\end{enumerate}

\ifshowsolutions%
\boxed{\textbf{解}}
(1) \textbf{真}. $A$ 与 $B$ 互不相容 $\Rightarrow P(AB) = 0$，但 $P(A)P(B) > 0$，故 $P(AB) \neq P(A)P(B)$，从而 $A$ 与 $B$ 不独立. 即：互不相容且概率均正的事件必不独立.

(2) \textbf{真}. $A$ 与 $B$ 独立 $\Rightarrow P(AB) = P(A)P(B) > 0$，故 $AB \neq \emptyset$，即相容. 也就是说：概率均正的独立事件必相容.

(3) \textbf{真}. 证明：
\begin{align*}
P(A \mid B) = P(A \mid \bar{B})
&\Longleftrightarrow \frac{P(AB)}{P(B)} = \frac{P(A\bar{B})}{1 - P(B)} \\
&\Longleftrightarrow P(AB)(1 - P(B)) = P(A\bar{B})P(B) \\
&\Longleftrightarrow P(AB) = P(AB)P(B) + P(A\bar{B})P(B) \\
&\Longleftrightarrow P(AB) = P(B)(P(AB) + P(A\bar{B})) = P(B)P(A).
\end{align*}
故 $P(A \mid B) = P(A \mid \bar{B})$ 等价于 $P(AB) = P(A)P(B)$，即 $A$ 与 $B$ 独立.

\textbf{评注：} 第(1)(2)条合起来说明：对于概率均正的事件，``独立''和``互不相容''是互斥的概念，不能兼得. 第(3)条给出了独立性的等价刻画：$B$ 发生与否不影响 $A$ 的条件概率.

\fi%
\item \textbf{（计算题，$\star\star\star$ 几何概型——会面问题）} 甲、乙两人约定在 $[0, T]$ 时间段内到达某处见面，先到者等待 $t$（$0 < t < T$）时间后若对方仍未到则离开。设两人到达时刻相互独立且均服从 $[0, T]$ 上的均匀分布。求两人能成功见面的概率.

\ifshowsolutions%
\boxed{\textbf{解}} 设甲到达时刻为 $x$，乙到达时刻为 $y$，则 $(x, y) \in \Omega = [0, T] \times [0, T]$.
两人能见面当且仅当 $|x - y| \leq t$.

$\Omega$ 是边长为 $T$ 的正方形，面积为 $T^2$.
满足 $|x - y| \leq t$ 的区域是正方形内夹在两条直线 $x - y = \pm t$ 之间的带状区域.
区域外为两个等腰直角三角形，直角边长均为 $T - t$，总面积 $(T - t)^2$.
故有利区域的面积为：
\[
m(A) = T^2 - (T - t)^2 = T^2 - (T^2 - 2Tt + t^2) = 2Tt - t^2.
\]

因此
\[
P = \frac{m(A)}{m(\Omega)} = \frac{2Tt - t^2}{T^2} = \frac{t(2T - t)}{T^2} = 1 - \left(1 - \frac{t}{T}\right)^2.
\]

\textbf{评注：} 当 $t = T$ 时 $P = 1$（必然见面）；当 $t \to 0$ 时 $P \approx \frac{2t}{T}$. 几何概型的核心是将随机现象转化为几何区域上的均匀分布，用长度、面积或体积之比计算概率.

\fi%
\end{enumerate}

\subsection{真题风格与综合训练}

\begin{enumerate}[resume]

\item \textbf{（\boxed{\textbf{真题风格}}，计算题，贝叶斯公式——信号传输）} 某通信系统中，发送信号为``0''和``1''两种。设发送``0''的概率为 $0.6$，发送``1''的概率为 $0.4$. 由于信道噪声干扰，发送``0''时收到``0''的概率为 $0.9$（即误码率 $0.1$），发送``1''时收到``1''的概率为 $0.8$（即误码率 $0.2$）.
\begin{enumerate}[label=(\arabic*)]
    \item 求收到信号``0''的概率；
    \item 已知收到信号``0''，求实际发送的也是``0''的概率.
\end{enumerate}

\ifshowsolutions%
\boxed{\textbf{解}} 设 $A_0 = \{\text{发送 }0\}$，$A_1 = \{\text{发送 }1\}$，$B_0 = \{\text{收到 }0\}$.
已知：$P(A_0) = 0.6$，$P(A_1) = 0.4$，$P(B_0 \mid A_0) = 0.9$，$P(B_0 \mid A_1) = 0.2$.

(1) 由全概率公式：
\[
\begin{aligned}
P(B_0)
&=P(A_0)P(B_0\mid A_0)+P(A_1)P(B_0\mid A_1)\\
&=0.6\times0.9+0.4\times0.2=0.62.
\end{aligned}
\]

(2) 由贝叶斯公式：
\[
P(A_0 \mid B_0) = \frac{P(A_0)P(B_0 \mid A_0)}{P(B_0)}
= \frac{0.6 \times 0.9}{0.62}
= \frac{0.54}{0.62} = \frac{27}{31} \approx 0.8710.
\]

\textbf{评注：} 先验概率 $P(A_0) = 0.6$，收到``0''后后验概率提升到约 $0.871$——观测信息修正了我们对实际发送信号的信念，这是贝叶斯推断的核心思想.

\fi%
\item \textbf{（\boxed{\textbf{真题风格}}，证明题，$\star\star$ 独立性 + 条件概率）} 设 $0 < P(A) < 1$，$0 < P(B) < 1$，且
\[
P(A \mid B) + P(\bar{A} \mid \bar{B}) = 1.
\]
证明：$A$ 与 $B$ 相互独立.

\boxed{\textbf{证}} 由条件概率定义：
\[
P(A \mid B) + P(\bar{A} \mid \bar{B})
= \frac{P(AB)}{P(B)} + \frac{P(\bar{A}\bar{B})}{P(\bar{B})} = 1.
\]

由德摩根律，$\bar{A}\bar{B} = \overline{A \cup B}$，故 $P(\bar{A}\bar{B}) = 1 - P(A \cup B)$.
又 $P(A \cup B) = P(A) + P(B) - P(AB)$，因此
\[
P(\bar{A}\bar{B}) = 1 - P(A) - P(B) + P(AB).
\]

代入得：
\[
\frac{P(AB)}{P(B)} + \frac{1 - P(A) - P(B) + P(AB)}{1 - P(B)} = 1.
\]

记 $x = P(AB)$，$a = P(A)$，$b = P(B)$，则
\[
\frac{x}{b} + \frac{1 - a - b + x}{1 - b} = 1.
\]

通分：
\[
x(1 - b) + b(1 - a - b + x) = b(1 - b).
\]

展开得：
\[
x - xb + b - ab - b^2 + bx = b - b^2.
\]

化简得 $x + b - ab - b^2 = b - b^2$，即 $x = ab$.
故 $P(AB) = P(A)P(B)$，$A$ 与 $B$ 相互独立. \hfill $\square$

\item \textbf{（\boxed{\textbf{真题风格}}，计算题，$\star\star$ 独立性 + 条件概率混合）} 甲、乙两人独立地对同一目标各射击一次，甲命中的概率为 $0.6$，乙命中的概率为 $0.5$. 现已知目标被命中，求目标是被甲命中的概率.

\ifshowsolutions%
\boxed{\textbf{解}} 设 $A = \{\text{甲命中}\}$，$B = \{\text{乙命中}\}$，则 $A, B$ 独立且 $P(A) = 0.6$，$P(B) = 0.5$.

目标被命中 $\Leftrightarrow A \cup B$，其概率为：
\[
P(A \cup B) = P(A) + P(B) - P(AB)
= 0.6 + 0.5 - 0.6 \times 0.5
= 1.1 - 0.3 = 0.8.
\]

所求为条件概率：
\[
P(A \mid A \cup B) = \frac{P(A \cap (A \cup B))}{P(A \cup B)}
= \frac{P(A)}{P(A \cup B)}
= \frac{0.6}{0.8} = \frac{3}{4} = 0.75.
\]

\textbf{评注：} $A \cap (A \cup B) = A$ 始终成立（$A \subseteq A \cup B$）。已知目标被命中后，甲命中的条件概率 $0.75$ 高于其无条件概率 $0.6$——这是因为``目标被命中''排除了两人都未命中的情形，使甲的实际贡献占比增大.

\fi%
\item \textbf{（\boxed{\textbf{真题风格}}，计算题，$\star\star$ 抽签问题 + 全概率 + 贝叶斯）} 有 $10$ 个签，其中 $3$ 个有奖。甲、乙、丙三人依次不放回地各抽一签。求：
\begin{enumerate}[label=(\arabic*)]
    \item 甲中奖的概率；
    \item 乙中奖的概率；
    \item 已知乙中奖，求甲也中奖的概率.
\end{enumerate}

\ifshowsolutions%
\boxed{\textbf{解}} 设 $A, B, C$ 分别表示甲、乙、丙中奖的事件.

(1) 显然 $P(A) = \dfrac{3}{10}$.

(2) 乙中奖有两种互斥途径：甲中奖后乙中奖，或甲未中奖后乙中奖。由全概率公式：
\[
P(B) = P(A)P(B \mid A) + P(\bar{A})P(B \mid \bar{A})
= \frac{3}{10} \cdot \frac{2}{9} + \frac{7}{10} \cdot \frac{3}{9}
= \frac{6 + 21}{90} = \frac{27}{90} = \frac{3}{10}.
\]

(3) 由贝叶斯公式：
\[
P(A \mid B) = \frac{P(A)P(B \mid A)}{P(B)}
= \frac{\frac{3}{10} \cdot \frac{2}{9}}{\frac{3}{10}}
= \frac{2}{9}.
\]

\textbf{评注：} 第(1)(2)问结果相同并非偶然——不放回抽签中，每人中奖概率等于奖品初始比例 $\frac{3}{10}$，与抽签顺序无关，此即``抽签的公平性''。第(3)问中已知乙中奖，甲中奖概率降至 $\frac{2}{9}$：因为乙中奖``消耗''了一个奖品，留给甲的奖品减少了一个.

\fi%
\end{enumerate}
