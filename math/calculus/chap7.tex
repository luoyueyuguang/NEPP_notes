\section{向量代数与空间解析几何}
\subsection{空间直角坐标系}
\begin{mydef}{空间直角坐标系}{SpaceRightAngleCoordinateSystem}
    从空间\(O\)中作三条互相垂直的数轴，分别记为\(x\)轴、\(y\)轴和\(z\)轴。这三条数轴的公共原点称为空间直角坐标系的\textbf{原点}，记作\(O\)。这样建立的坐标系称为空间直角坐标系。其中，\(x
    \)轴、\(y\)轴和\(z\)轴分别称为空间直角坐标系的\textbf{基线}，\(x\)轴的正方向为\textbf{正\(x\)向}，\(y\)轴的正方向为\textbf{正\(y\)向}，
    \(z\)轴的正方向为\textbf{正\(z\)向}。若三个坐标轴的正向满足右手法则，则称为右手空间直角坐标系。空间中的“卦限”由三个坐标面的分割定义，不能把整个坐标系称为“第一象限”。
    \begin{tikzpicture}
        \draw[->] (0,0,0) -- (3,0,0) node[anchor=north east]{$x$轴};
        \draw[->] (0,0,0) -- (0,3,0) node[anchor=north west]{$y$轴};
        \draw[->] (0,0,0) -- (0,0,3) node[anchor=south]{$z$轴};
    \end{tikzpicture}
    如果曲面\(S\)与三元方程\(F(x, y, z) = 0\)或\(z = f(x, y)\)有如下关
    \begin{enumerate}
        \item \(S\)上的点\(P(x, y, z)\)满足方程\(F(x, y, z) = 0\)；
        \item 不在曲面上的点\(P(x, y, z)\)不满足方程
    \end{enumerate}
    则称这个三元方程是曲面\(S\)的\textbf{方程}。曲面\(S\)称为三元方程\(F(x, y, z) = 0\)或\(z = f(x, y)\)的图形。 \\
    空间曲线\(C\)可以看作是曲面\(
        \begin{cases}
            F(x, y, z) = 0 \\
            G(x, y, z) = 0 \
        \end{cases}
    \)的交线，\(
        \begin{cases}
            F(x, y, z) = 0 \\
            G(x, y, z) = 0 \
        \end{cases}
    \)被称为曲线\(C\)的一般方程。\\

    \begin{mydef}{曲线$C$的参数方程}{CurveParameterEquation}
        对于空间曲线\(C\)上的一点\(P(x, y, z)\)，分别表示为\(x = x(t), y = y(t), z = z(t)\)，其中\(t\)称为参数，则称三元方程组
        为空间曲线\(C\)的\textbf{参数方程}。其中，\(x = x(t), y = y(t), z = z(t)\)称为空间曲线\(C\)上点\(P(x, y, z)\)的\textbf{坐标函数}。
    \end{mydef}
\end{mydef}
\subsection{向量代数}
\begin{mydef}{向量}{Vector}
    既有大小又有方向的量称为向量。\\
    通常用\(\mathbf{a}, \mathbf{b}, \vec{ab}\)等形式表示向量\\
    在空间坐标系中，若向量\(\mathbf{a}\) = \(\vec{OM}\),点\(M\)的坐标为\((x, y, z)\)，则向量\(\mathbf{a}\)可以表示为
    \(\mathbf{a} = x \hat{\imath} + y \hat{\jmath} + z \hat{k}\)的形式，其中\(x, y, z\)分别是向量在基线上的投影。\\
    也可以用\(M\)的坐标表示为\(\mathbf{a}  = (x, y, z)\)的形式。\\
    起点为\(O\)，终点为\(P\)的向量\(\vec{OP}\)称为点\(P\)的向径，常用\(\mathbf{r}, \mathbf{r_0}\)表示。\\
    向量\(\mathbf{a}\)的大小(长度)称为向量的\textbf{模}，记作\(|\mathbf{a}|\)，\(|\mathbf{a}|\)的大小为\(
        \sqrt{x^2 + y^2 + z^2}
    \)。
    模为\(1\)的向量称为\textbf{单位向量}，模为零
    的向量称为\textbf{零向量}，记作\(0\)或\(\mathbf{0}\)。\\
    方向相反或相同的向量称为\textbf{共线向量}。任意两个自由向量总可以平移到同一平面内；“共面”通常用于三个或更多向量，指它们可平移到同一个平面内。\\
    从同一点出发的两个非零向量\(\mathbf{a}\)和\(\mathbf{b}\),它们所成的那个不超过\(\pi\)的角称为向量\(\mathbf{a}\)和
    \(\mathbf{b}\)的\textbf{夹角}，记作\(|\angle \mathbf{a}, \mathbf{b}|\)。\\
    向量\(\mathbf{a}\)与三个坐标向量\(\hat{\imath}, \hat{\jmath}, \hat{k}\)的夹角分别称为向量\(\mathbf{a}\)的\textbf{方向角}。\\
    方向角的余弦称为向量\(\mathbf{a}\)的\textbf{方向余弦}，记作\(cos \alpha, cos \beta, cos \gamma\)，
    \[
        \cos \alpha = \frac{x}{\sqrt{x^2 + y^2 + z^2}}, \cos \beta = \frac{y}{\sqrt{x^2 + y^2 + z^2}}, \cos \gamma = \frac{z}{\sqrt{x^2
        + y^2 + z^2}}
    \]
    且方向余弦\(\cos \alpha, \cos \beta, \cos \gamma\)的取值范围就是单位向量\(
        \frac{\mathbf{a}}{\mathbf{|a|}}
    \)的坐标\(\frac{\mathbf{a}}{\mathbf{|a|}} = (\cos \alpha, \cos \beta, \cos \gamma)\)。\\
    与向量\(\mathbf{a}\)的方向余弦成比例的一组数称为向量\(\mathbf{a}\)的一组\textbf{方向数}。\\
\end{mydef}
\begin{formula}{向量运算的定义及运算公式}{VectorOperation}
    设\(\mathbf{a}_j = x_j \hat{\imath} + y_j \hat{\jmath} + z_j \hat{k} = (x_j, y_j, z_j)\)，其中\(j = 1, 2\)。则有以下运算公式：
    \begin{enumerate}
        \item 向量加法：\(\mathbf{a}_1 + \mathbf{a}_2 = (x_1 + x_2, y_1 + y_2, z_1 + z_2)\)；
        \item 向量数乘：\(\lambda \mathbf{a}_1 = (\lambda x_1, \lambda y_1, \lambda z_1)\)；
        \item 数量积(点积，内积)：\(\mathbf{a}_1 \cdot \mathbf{a}_2 = x_1 x_2 + y_1 y_2 + z_1 z_2\)；
        \item 向量积(叉积，外积)：\(\mathbf{a}_1 \times \mathbf{a}_2 = (y_1 z_2 - y_2z_1,z_1 x_2 - x_1 z_2, x_1 y_2 - x_2 y_1)\)；
        \[
            \mathbf{a}_1 \times \mathbf{a}_2 = \begin{vmatrix}
                \hat{\imath} & \hat{\jmath} & \hat{k}\\
                x_1 & y_1 & z_1\\
                x_2 & y_2 & z_2\
            \end{vmatrix}
        \]
        \item 向量混合积(三元积)：\(\mathbf{a}_1 \cdot (\mathbf{a}_2 \times \mathbf{a}_3)\)
        \[
            \mathbf{a}_1 \cdot (\mathbf{a}_2 \times \mathbf{a}_3) = \begin{vmatrix}
                x_1 & y_1 & z_1\\
                x_2 & y_2 & z_2\\
                x_3 & y_3 & z_3\
                \end{vmatrix}
        \]
    \end{enumerate}
\end{formula}

\begin{formula}{向量的运算法则}{VectorOperationLaw}
    \begin{enumerate}
        \item 加法与数乘
        \[
            \begin{aligned}
                &\mathbf{a} + \mathbf{b} = \mathbf{b} + \mathbf{a} &\qquad (\mathbf{a} + \mathbf{b}) + \mathbf{c} = \mathbf{a} + (\mathbf{b} + \mathbf{c})\\
                &\lambda(\mu \mathbf{a}) = (\lambda \mu) \mathbf{a} &\qquad (\lambda + \mu) \mathbf{a} = \lambda \mathbf{a} + \mu \mathbf{a}\\
                &\lambda(\mathbf{a} + \mathbf{b}) = \lambda \mathbf{a} + \lambda \mathbf{b} &\qquad (\lambda + \mu) \mathbf{a} = \lambda \mathbf{a} + \mu \mathbf{a}\\
                &\mathbf{a} + (-\mathbf{a}) = \mathbf{0} &\qquad \lambda\mathbf{a}=\mathbf0 \Rightarrow \lambda=0\text{或}\mathbf a=\mathbf0\\
            \end{aligned}
        \]
        \item 数量积\\
        \(\mathbf{a} \cdot \mathbf{b} = \mathbf{b} \cdot \mathbf{a}\)，
        \(\mathbf{a} \cdot (\mathbf{b} + \mathbf{c}) = \mathbf{a} \cdot \mathbf{b} + \mathbf{a} \cdot \mathbf{c}\),
        \(\lambda (\mathbf{a} \cdot \mathbf{b}) = \lambda \mathbf{a} \cdot \mathbf{b}\)。
        \item 向量积 \\
        \(\mathbf{a} \times \mathbf{b} = -\mathbf{b} \times \mathbf{a}\)，
        \((\lambda \mathbf{a} \times \mathbf{b}) = \lambda (\mathbf{a} \times \mathbf{b})\) \\
        \(\mathbf{a} \times (\mathbf{b} + \mathbf{c}) = \mathbf{a} \times \mathbf{b} + \mathbf{a} \times \mathbf{c}\),
        \(\mathbf{a} \times \mathbf{a} = 0\)
        \item 混合积 \\
        \[
            \begin{aligned}
            &(\mathbf{a}, \mathbf{b}, \mathbf{c}) = (\mathbf{b}, \mathbf{c}, \mathbf{a}) = (\mathbf{c}, \mathbf{a}, \mathbf{b}) \\
            &(\mathbf{a}, \mathbf{a}, \mathbf{b}) = (\mathbf{a}, \mathbf{b}, \mathbf{a}) = (\mathbf{a}, \mathbf{b}, \mathbf{b}) = 0 \\ 
            &(\mathbf{a}, \mathbf{b}, \mathbf{c}) = - (\mathbf{b}, \mathbf{a}, \mathbf{c}) \\
            &(\lambda \mathbf{a}, \mathbf{b}, \mathbf{c}) = \lambda (\mathbf{a}, \mathbf{b}, \mathbf{c}) \\
            &(\mathbf{a_1} + \mathbf{a_2}, \mathbf{b}, \mathbf{c}) = (\mathbf{a_1}, \mathbf{b}, \mathbf{c}) + (\mathbf{a_2}, \mathbf{b}, \mathbf{c})
            \end{aligned}
        \]
        \end{enumerate}
\end{formula}

\begin{conclusion}{加法与数乘的几何应用}{VectorOperationLawGeo}
    \begin{enumerate}
        \item 建立坐标系，建立直线方程，建立平面方程
        \item 过点\(P_0\)方向向量为\(\mathbf{S}\)的直线的向量方程是\(
            \mathbf{r} - \mathbf{r_0} = t \mathbf{S}
        \)
        \item 过点\(P_0\)与\(\mathbf{U_1}\), \(\mathbf{U_2}\)都平行的平面的向量方程是\(
            \mathbf{r} - \mathbf{r_0} = t_1 \mathbf{U_1} + t_2 \mathbf{U_2}
        \)
    \end{enumerate}
\end{conclusion}
\begin{conclusion}{数量积的几何应用}{DotProductGeo}
    \begin{enumerate}
        \item 求模\(|\mathbf{a}| = \sqrt{\mathbf{a}^2}\)进而求两点距离
        \item 求两个向量的夹角\(\theta\)，进而可求直线，直线与平面，两个平面之间的夹角
        \item 判定垂直关系，使用\(\mathbf{a_1} \cdot \mathbf{a_2} = 0\)，进而可求垂直直线，垂直平面
        \item 求点到平面的距离
        \item 建立平面方程 \(
        \mathbf{n} \cdot (\mathbf{r} - \mathbf{r_0}) = 0
        \)，其中\( \mathbf{n} = (A, B, C)\)是平面的法向量，\(\mathbf{r_0}\)是平面上的一个点。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{向量积的几何应用}{CrossProductGeo}
    \begin{enumerate}
        \item 求平行四边形(三角形)面积(以\(\mathbf{a}, \mathbf{b}\)为邻边的平行四边形面积为
        \(|a \times b|\),进而求点到直线的距离
        \[
            d=\frac{\lVert\overrightarrow{P_1P_0}\times\mathbf S\rVert}{\lVert\mathbf S\rVert}
        \]
        \item 求两个平面交线的方向向量\(\mathbf{S}\)，从而可把直线的一般方程化为直线的标准方程。设这两个
        平面的法向量分别为\(\mathbf{n_1}\), \(\mathbf{n_2}\)，则交线与\(\mathbf{n_1}, \mathbf{n_2}\)都要垂直，故可取\(\mathbf{S} = \mathbf{n_1} \times \mathbf{n_2}\)。
        \item 判断平行
        \[
            \mathbf{a_1} \parallel \mathbf{a_2} \Leftrightarrow \mathbf{a_1} \times \mathbf{a_2} = 0
            \Leftrightarrow \frac{x_1}{x_2} = \frac{y_1}{y_2} = \frac{z_1}{z_2}
            \Leftrightarrow \text{存在实数}\lambda,\text{使得} a_1 = \lambda a_2
        \]  
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{向量混合积的几何应用}{TripleProductGeo}
    \begin{enumerate}
        \item 判断三个向量是否共面 \\
        向量\(\mathbf{a_1} = \{ x_1, y_1, z_1\}, \mathbf{a_2} = \{ x_2, y_2, z_2\}, \mathbf{a_3} = \{ x_3, y_3, z_3\}\)共面，则有混合积\(
        (\mathbf{a_1}, \mathbf{a_2}, \mathbf{a_3}) = 0
        Leftrightarrow\)存在不全为零的实数\(\lambda_1, \lambda_2, \lambda_3\)使得\(
        \lambda_1 \mathbf{a_1} + \lambda_2 \mathbf{a_2} + \lambda_3 \mathbf{a_3} = 0
        \Leftrightarrow\)
        \[
        \begin{vmatrix}
            x_1 & y_1 & z_1\\
            x_2 & y_2 & z_2\\
            x_3 & y_3 & z_3\
        \end{vmatrix} = 0
        \]
        \item 建立平面方程 \\
        若\(\mathbf{U_1}, \mathbf{U_2}\)不平行，则过点\(P_0\)与\(\mathbf{U_1}, \mathbf{U_2}\)都平行的平面的向量方程是
        \((\vec{\mathbf{P_0P}}, \mathbf{U}_1, \mathbf{U}_2) = 0\),用坐标表示，即
        \[
            \begin{vmatrix}
                x - x_0 & y - y_0 & z - z_0\\
                x_1 & y_1 & z_1\\
                x_2 & y_2 & z_2
            \end{vmatrix}
            = 0
        \]
        \item 求平行六面体的体积 \\
        以\(\mathbf{a}, \mathbf{b}, \mathbf{c}\)为棱的平行六面体的体积是\(
        |(\mathbf{a},\mathbf{b}, \mathbf{c})|\)
        \item 判断两条直线是否异面，并求两条异面直线公垂线的长 \\
        设\(L_1, L_2\)是两条异面直线，方向向量分别为\(\mathbf{U_1}, \mathbf{U_2}\),\(L_2\)投影在平行于\(L_2\)的平面上且交于\(L_1\)的点为\(\mathbf{Q}\),\(
            \mathbf{Q}
        \)为\(L_2\)上点\(P\)在平面上的投影，\(\bar{PQ}\)便是公垂线
        则这两条直线的公垂线长为
        \[
            d = \frac{
                |(
                    \overrightarrow{\mathbf{P_1P_2}}, \mathbf{U}_1, \mathbf{U}_2
                )|
            }{
                |\mathbf{U}_1 \times \mathbf{U}_2|
            }
        \]
        \item 建立异面直线公垂线的一般方程
        \[
            \begin{cases}
                (\overrightarrow{\mathbf{P_1P}}, \mathbf{U}_1, \mathbf{U}_1 \times \mathbf{U}_2) = 0\\
                (\overrightarrow{\mathbf{P_2P}}, \mathbf{U}_2, \mathbf{U}_1 \times \mathbf{U}_2) = 0\
            \end{cases}
        \]
        (
            异面直线\(L_1, L_2\)的公垂线既在过\(L_1\)且平行于\(\mathbf{U}_1 \times \mathbf{U}_2\)的平面内，也在过\(L_2\)且平行于\(\mathbf{U}_1 \times \mathbf{U}_2\)的平面内。)
    \end{enumerate}
\end{conclusion}

\subsection{空间解析几何}

\begin{conclusion}{平面方程的基本形式}{PlaneEquationBasicForm}
    \begin{enumerate}
        \item 点法式：\(A(x - x_0) + B(y - y_0) + C(z - z_0) = 0\)
        \item 一般式：\(Ax+By+Cz+D=0\)，其中\(A,B,C\)不全为零
        \item 向量式：\(\mathbf{r} - \mathbf{r_0} = t_1\mathbf{U_1} + t_2\mathbf{U_2}\)，
        其中\(
            \mathbf{r_0} = \overrightarrow{OM} = (x_0, y_0, z_0)\)，\(\mathbf{U_1}, \mathbf{U_2}\)是平面上任意两个不共线的向量，\(t_1, t_2\)为参数
            ,\(\mathbf{r} = \overrightarrow{OP} = (x,y,z)\)，\(P\)是平面上任一点
        \item 参数式：
        \[
            \begin{cases}
                x = x_0 + X_1t_1 + X_2t_2\\
                y = y_0 + Y_1t_1 + Y_2t_2\\
                z = z_0 + Z_1t_1 + Z_2t_2\
            \end{cases}
        \]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{确定平面方程的思路}{PlaneEquationBasicFormMethod}
    \begin{enumerate}
        \item 点法式：已知平面上的一个点\(P_0(x_0, y_0, z_0)\)和它的法向量\(\mathbf{n}(A, B, C)\)，则
        \[
            A(x - x_0) + B(y - y_0) + C(z - z_0) = 0
        \]
        \item 已知平面上的一个点\(P_0(x_0, y_0, z_0)\)和它的两个不共线的向量\(\mathbf{U_1}, \mathbf{U_2}\)，则
        \[
            \begin{vmatrix}
                x - x_0 & y - y_0 & z - z_0\\
                x_1 & y_1 & z_1\\
                x_2 & y_2 & z_2\
            \end{vmatrix} = 0
        \]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{直线方程的基本形式}{LineEquationBasicForm}
    \begin{enumerate}
        \item 一般式(交面式)\(
            \begin{cases}
                A_1x + B_1y + C_1z + D_1 = 0\\
                A_2x + B_2y + C_2z + D_2 = 0\
            \end{cases}
        \)其中\(A_1, B_1, C_1\)，\(A_2, B_2, C_2\)不平行
        \item 参数式：方向向量为\((l, m, n)\)\\
        \(
            \begin{cases}
                x = x_0 + tl \\
                y = y_0 + tm \\
                z = z_0 + tn\
            \end{cases}
        \)
        \item 对称式(标准式)\(
            \dfrac{x - x_0}{l} = \dfrac{y - y_0}{m} = \dfrac{z - z_0}{n}
        \)
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{确定直线方程的思路}{LineEquationBasicFormMethod}
    \begin{enumerate}
        \item 两个不平行的平面相交于一条直线，则选择一般式
        \item 已知平面上一点\(M(x_0, y_0, z_0)\)和它的方向向量\((l, m, n)\)，则选择参数式或标准式
        \item 投影直线方程\\
        设直线\(L\)与平面\(\pi\)，过直线\(L\)在\(\pi\)上的投影方程的计算方法为
        将直线\(L\)换为一般式的形态，用平面束方程求出与\(\pi\)垂直的的平面\(\Pi\)的方程，
        联立两个方程组，即为投影直线方程
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{两个平面间的关系}{PlaneRelation}
    设两个平面\(\Pi_1, \Pi_2\)，\(
    \Pi_1: A_1x + B_1y + C_1z + D_1 = 0\)，\(
    \Pi_2: A_2x + B_2y + C_2z + D_2 = 0\)则
    \begin{enumerate}
        \item 两平面的法向量成比例时，两平面平行或重合。具体地，若存在\(k\neq0\)使\((A_1,B_1,C_1)=k(A_2,B_2,C_2)\)，则再比较\(D_1\)与\(kD_2\)：不等时平行而不重合，相等时重合。这样写可避免比例式分母为零的问题。
        \item \(Pi_1 \perp Pi_2 \Leftrightarrow \mathbf{n_1} \perp \mathbf{n_2} \Leftrightarrow A_1A_2 + B_1B_2 + C_1C_2 = 0\)
        \item \(Pi_1\)与\(Pi_2\)的夹角\(\theta\)(不大于\(\pi/2\)的法向量夹角)，则
        \[
            \cos \theta = \dfrac{|\mathbf{n_1} \cdot \mathbf{n_2}|}{|\mathbf{n_1}||\mathbf{n_2}|} =
            \dfrac{|A_1A_2 + B_1B_2 + C_1C_2|}{\sqrt{A_1^2 + B_1^2 + C_1^2}\sqrt{A_2^2 + B_2^2 + C_2^2}}
        \]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{两条直线间的关系}{LineRelation}
    设两条直线\(L_1, L_2\)，方向向量为\(\mathbf{S_1}, \mathbf{S_2}\) \\
    \(L_1: \dfrac{x - x_1}{l_1} = \dfrac{y - y_1}{m_1} = \dfrac{z - z_1}{n_1}\)，\(L_2: \dfrac{x - x_2}{l_2} = \dfrac{y - y_2}{m_2} = \dfrac{z - z_2}{n_2}\)，则
    
    \begin{enumerate}
        \item \(L_1 \parallel L_2 \Leftrightarrow \mathbf{S_1} \parallel \mathbf{S_2}\)，即存在\(k\ne0\)使\(\mathbf S_1=k\mathbf S_2\)，且\((x_1,y_1,z_1)\)不在\(L_2\)上。也可用\(\mathbf S_1\times\mathbf S_2=\mathbf0\)判断方向平行，以避免比例式出现零分母。
        \item 若\(\mathbf S_1\cdot\mathbf S_2=0\)，则两直线的方向夹角为\(90^\circ\)。只有在两直线还相交时，才称两条空间直线互相垂直；异面直线也可能方向正交。
        \item \(L_1\)与\(L_2\)的夹角\(\theta\)(不大于\(\pi/2\)的方向向量夹角)，则
        \[
            \cos \theta = \dfrac{|\mathbf{S_1} \cdot \mathbf{S_2}|}{|\mathbf{S_1}||\mathbf{S_2}|} =
            \dfrac{|l_1l_2 + m_1m_2 + n_1n_2|}{\sqrt{l_1^2 + m_1^2 + n_1^2}\sqrt{l_2^2 + m_2^2 + n_2^2}}
        \]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{直线与平面的关系}{LinePlaneRelation}
    设直线\(L: \dfrac{x - x_0}{l} = \dfrac{y - y_0}{m} = \dfrac{z - z_0}{n}\)，平面\(\Pi: Ax + By + Cz + D = 0\)。
    \(L\)的方向向量为\(\mathbf{S}\), \(\mathbf{n}\)为平面的法向量。
    则
    \begin{enumerate}
        \item \(L \parallel \Pi \Leftrightarrow Al + Bm + Cn = 0\)且\(
            Ax_0 + By_0 + Cz_0 + D \neq 0
        \)
        \item \(L \perp \Pi \Leftrightarrow \mathbf{S} \parallel \mathbf{n}\)，即存在\(k\ne0\)使\((l,m,n)=k(A,B,C)\)
        \item \(L\)与\(\Pi\)的夹角
        \(\theta = \dfrac{\pi}{2} -\mathbf{S}\text{与}\mathbf{n}\text{的夹角：}\)
           \[ \sin \theta = sin( \dfrac{\pi}{2} - <\mathbf{S}, \mathbf{n}> ) = \dfrac{|\mathbf{S} \cdot \mathbf{n}|}{|\mathbf{S}||\mathbf{n}|}
             = \dfrac{|Al + Bm + Cn|}{\sqrt{l^2 + m^2 + n^2}\sqrt{A^2 + B^2 + C^2}}\]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{平面束方程}{PlaneBundleEquation}
    通过直线\(L:\begin{cases}
    A_1x + B_1y + C_1z + D_1 = 0\\
    A_2x + B_2y + C_2z + D_2 = 0\
\end{cases}\)的平面束方程为\\
\(
    \lambda(A_1x + B_1y + C_1z + D_1) + \mu(A_2x + B_2y + C_2z + D_2) = 0\),\\
    其中\(\lambda,\mu\)为任意实数，但不能同时为零。
\end{conclusion}

\begin{formula}{关于距离的计算公式}{DistanceFormula}
    \begin{enumerate}
        \item 两点\(P_1, P_2\)的距离\(d = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2 + (z_2 - z_1)^2}\)
        \item 点\(P_0(x_0, y_0, z_0)\)到直线\(L: \dfrac{x - x_1}{l} = \dfrac{y - y_1}{m} = \dfrac{z - z_1}{n}\)的距离
        为\[
            d = |\overrightarrow{P_0P_1}| \sin <
                \overrightarrow{P_0P_1}, \mathbf{S} 
            > = 
            \dfrac{|\mathbf{S} \times \overrightarrow{P_0P_1}|}{|\mathbf{S}|} =
            \dfrac{
                \left|\begin{vmatrix}
                    i & j & k \\
                    x_1 - x_0 & y_1 - y_0 & z_1 - z_0 \\
                    l & m & n\
                \end{vmatrix}\right|
            }{\sqrt{l^2 + m^2 + n^2}}
        \]
        其中\(P_1\)是直线\(L\)上的定点，\(\mathbf{S}\)为直线\(L\)的方向向量。
        \item 点\(P_0(x_0, y_0, z_0)\)到平面\(\Pi: Ax + By + Cz + D = 0\)的距离
        \[
            d = |\overrightarrow{P_0P_1}| \cos <\overrightarrow{P_0P_1}, \mathbf{n}> =
            \dfrac{|\mathbf{n} \cdot \overrightarrow{P_0P_1}|}{|\mathbf{n}|} =
            \dfrac{|Ax_0 + By_0 + Cz_0 + D|}{\sqrt{A^2 + B^2 + C^2}}
        \]
        其中\(P_1\)是平面\(\Pi\)上的定点，\(\mathbf{n}\)为平面的法向量。
    \end{enumerate}
\end{formula}

\begin{mydef}{球面}{SphericalSurface}
    设\(P_0(x_0, y_0, z_0)\)为球心，\(R\)为半径，则球面方程为
    \[
        (x - x_0)^2 + (y - y_0)^2 + (z - z_0)^2 = R^2
    \]
\end{mydef}

\begin{mydef}{旋转曲面}{RotationalSurface}
    设\(L\)是\(xOz\)平面上一条曲线，其方程为\(\begin{cases}
        f(x, z) = 0 \\
        y = 0\
    \end{cases}\), 则以\(L\)绕\(z\)轴旋转得到旋转曲面，设\(P(x, y, z)\)为旋转曲面上的点，由点\(
    P_0(x_0, 0, z_0)\)旋转而来(点\(M(0, 0, z)\)是圆心),由
    \(
        |x_0| = |\overrightarrow{MP_0}| = 
        ||\overrightarrow{MP}| = \sqrt{x^2 + y^2}, z_0 = z
    \)得到旋转面方程为\(
    f(\pm\sqrt{x^2 + y^2}, z) = 0
    \)

    同理，绕\(y\)轴旋转得到的曲面方程为\(f(\pm\sqrt{x^2 + z^2}, y) = 0\)，绕\(x\)轴旋转得到的曲面方程为\(f(x, \pm\sqrt{y^2 + z^2}) = 0\)。
    由参数方程\(
    x = f(t), y = g(t), z = h(t)\)得到的旋转曲面的参数方程为\[
        \begin{cases}
            x = \sqrt{f^2(t) + g^2(t)}\cos \theta \\
            y = \sqrt{f^2(t) + g^2(t)}\sin \theta \\
            z = h(t)
        \end{cases}
        \alpha < t < \beta,0 \leq \theta \leq 2\pi
    \]
\end{mydef}

\begin{conclusion}{旋转曲面方程的直角坐标系求法}{RotationalSurfaceEquationInCartesianCoordinateSystem}
    \begin{enumerate}
        \item 将旋转的曲线h化成参数方程
        \item 根据与旋转轴的距离进行联立
        \item 消去参数\(t\)，得到直角坐标方程。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{柱面}{CylinderSurface}
    \(\Gamma\)是一条空间曲线，直线\(L\)沿\(\Gamma\)平行移动所产生的曲面称为柱面。
    \(\Gamma\)为柱面的准线,\(L\)为柱面的母线。
    \begin{enumerate}
        \item 若准线方程为\(\begin{cases}
            f(x, y) = 0 \\
            z = c\
            \end{cases}\)
        \begin{enumerate}
            \item 当母线平行于\(z\)轴时，柱面方程为\(f(x, y) = 0\)；
            \item 当母线的方向向量是\(\mathbf{S} = (l, m, n)\)且\(n\ne0\)时，柱面方程为\[
                f\left(x - \dfrac{l}{n}(z-c), y - \dfrac{m}{n}(z-c)\right) = 0
            \]
        \end{enumerate}
        \item 若准线方程为\(\begin{cases}
            x = f(t) \\
            y = g(t) \\
            z = h(t)\
        \end{cases}, t \in (\alpha, \beta)\)，母线的方向向量是\(\mathbf{S} = (l, m, n)\)，则柱面方程为\[
            \begin{cases}
                x = f(t) + \lambda l \\
                y = g(t) + \lambda m \\
                z = h(t) + \lambda n
                \end{cases}, t \in (\alpha, \beta), \lambda \text{为任意实数}
            \]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{空间曲线在坐标面上的投影}{SpaceCurveProjectionOnCoordinatePlane}
    设\(\Gamma\)是一条空间曲线，\(\Pi\)是一张平面，对于\(\Gamma\)上任意一点\(P\),令\(\Gamma(P)\)
    是点\(P\)在平面\(\Pi\)上的投影点，即\(
    \Pi(P) \in \Pi, \overrightarrow{\Pi(P)P} \perp \Pi
    \)，所有投影点的集合称为\(\Gamma\)在平面\(\Pi\)上的投影曲线。\\
    由垂线构成的曲面是以\(\Gamma\)为准线的柱面，称为\(\Gamma\)到\(\Pi\)的
    投影柱面。\\
    \begin{enumerate}
        \item 如\(
        \Gamma:\begin{cases}
            f(x, y, z) = 0 \\
            g(x, y, z) = 0\
        \end{cases}
        \)消去\(z\)得到\(\varphi(x, y) = 0\),这是以\(\Gamma\)为准线，母线平行于
        \(z\)轴的柱面方程。
        而\(
            \begin{cases}
                \varphi(x, y) = 0 \\
                z = 0\
            \end{cases}
        \)是\(\Gamma\)在\(xOy\)平面上的投影曲线方程。
        \item 如\(\begin{cases}
            x = f(t) \\
            y = g(t) \\
            z = h(t)\
        \end{cases}\),则\(
            \begin{cases}
                x = f(t) \\
                y = g(t) \\
                z = 0\
            \end{cases}
        \)是\(\Gamma\)是在\(xOy\)平面上的投影曲线方程
    \end{enumerate}
\end{conclusion}

\subsection{基本方法与技巧}
\begin{conclusion}{求平面方程的基本方法}{PlaneEquationBasicMethod}
    求平面需注意怎么求法向量
    \begin{enumerate}
        \item 过一定点并与直线垂直的平面，法向量即为直线方向向量。
        \item 过一定点且平行于已知平面方程，法向量即为平面法向量
        \item 过一定点且与两给定直线平行的平面，法向量同时垂直这两条方向向量,即\(\mathbf{n} = \mathbf{a} \times \mathbf{b}\)
        \item 过一定点且分别垂直于两个已知平面的方程，法向量即为这两个平面方程的法向量的向量积
        \item 过一定点、平行于一条直线且垂直于一个平面的平面，若直线方向向量为\(\mathbf a\)、已知平面法向量为\(\mathbf n_0\)，则可取法向量\(\mathbf n=\mathbf a\times\mathbf n_0\)（要求叉积非零）
        \item 过一定点且过一定直线，将直线化为对称式方程，求出直线上的点与定点的向量\(\overrightarrow{\mathbf{M_1M_0}}\),
        进行叉乘得到法向量，即\(\mathbf{n} = \overrightarrow{\mathbf{M_1M_0}} \times \mathbf{a}\)
        \item 过两点且垂直于一个已知平面的平面，其法向量可由两点连线的方向向量与已知平面的法向量叉乘得到（要求二者不平行）。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{求直线方程的基本方法}{LineEquationBasicMethod}
    求直线方程需注意怎么求方向向量
    \begin{enumerate}
        \item 过一点且平行于一直线的直线，方向向量即为直线方向向量
        \item 过一点且垂直于平面的直线方程，方向向量即为平面的法向量
        \item 过一点且垂直于两条直线的直线，方向向量即为两条直线的方向向量的叉乘
        \item 过一点，且平行于两个平面的直线，方向向量即为两平面法向量的叉乘
        \item 过一点、平行于一个平面且垂直于一条已知直线的直线，其方向向量可取为平面法向量与已知直线方向向量的叉积（要求叉积非零）
    \end{enumerate}
\end{conclusion}

\subsection{基础过关题}

\textbf{1.（计算题）} 已知向量 \(\mathbf{a}=(2,-1,3)\)，\(\mathbf{b}=(1,4,-2)\)，求：
\begin{enumerate}[label=(\arabic*)]
    \item \(\mathbf{a}\cdot \mathbf{b}\)；
    \item \(\mathbf{a}\times \mathbf{b}\)；
    \item \(\mathbf{a}\) 与 \(\mathbf{b}\) 的夹角 \(\theta\)。
\end{enumerate}

\vspace{0.5cm}

\boxed{\textbf{解}}
\begin{enumerate}[label=(\arabic*)]
    \item \(\mathbf{a}\cdot\mathbf{b}=2\times 1+(-1)\times 4+3\times(-2)=2-4-6=-8\)。
    \item \(\mathbf{a}\times\mathbf{b}=\begin{vmatrix}
        \hat{\imath} & \hat{\jmath} & \hat{k} \\
        2 & -1 & 3 \\
        1 & 4 & -2
    \end{vmatrix}=(-2-12)\hat{\imath}-(-4-3)\hat{\jmath}+(8+1)\hat{k}=(-10,7,9)\)。
    \item \(|\mathbf{a}|=\sqrt{4+1+9}=\sqrt{14},\;|\mathbf{b}|=\sqrt{1+16+4}=\sqrt{21}\)，
    \[
        \cos\theta=\frac{\mathbf{a}\cdot\mathbf{b}}{|\mathbf{a}||\mathbf{b}|}
        =\frac{-8}{\sqrt{14}\cdot\sqrt{21}}=\frac{-8}{7\sqrt{6}},
    \]
    故 \(\theta=\arccos\left(-\dfrac{8}{7\sqrt{6}}\right)\)。
\end{enumerate}


\textbf{2.（计算题）} 求过三点 \(P(1,2,3),\;Q(2,0,1),\;R(-1,1,2)\) 的平面方程。

\vspace{0.5cm}

\boxed{\textbf{解}}
\[
    \overrightarrow{PQ}=(1,-2,-2),\quad \overrightarrow{PR}=(-2,-1,-1).
\]
法向量
\[
    \mathbf{n}=\overrightarrow{PQ}\times\overrightarrow{PR}
    =\begin{vmatrix}
        \hat{\imath} & \hat{\jmath} & \hat{k} \\
        1 & -2 & -2 \\
        -2 & -1 & -1
    \end{vmatrix}
    =(2-2)\hat{\imath}-(-1-4)\hat{\jmath}+(-1-4)\hat{k}
    =(0,5,-5).
\]
取 \(\mathbf{n}=(0,1,-1)\)，由点法式得
\[
    0\cdot(x-1)+1\cdot(y-2)+(-1)\cdot(z-3)=0\;\Longrightarrow\;y-z+1=0.
\]
即平面方程为 \(y-z+1=0\)。


\textbf{3.（计算题）} 求过点 \((1,-2,3)\) 且方向向量为 \((2,3,-1)\) 的直线的对称式方程与参数式方程。

\vspace{0.5cm}

\boxed{\textbf{解}}
对称式方程：
\[
    \frac{x-1}{2}=\frac{y+2}{3}=\frac{z-3}{-1}.
\]
参数式方程：
\[
    \begin{cases}
        x=1+2t,\\
        y=-2+3t,\\
        z=3-t,
    \end{cases}\quad t\in\mathbb{R}.
\]


\textbf{4.（计算题）} 求点 \(P(1,2,3)\) 到平面 \(2x-y+2z-5=0\) 的距离。

\vspace{0.5cm}

\boxed{\textbf{解}}
由点到平面的距离公式：
\[
    d=\frac{|Ax_0+By_0+Cz_0+D|}{\sqrt{A^2+B^2+C^2}}
    =\frac{|2\times 1-1\times 2+2\times 3-5|}{\sqrt{2^2+(-1)^2+2^2}}
    =\frac{|2-2+6-5|}{3}=\frac{1}{3}.
\]


\subsection{综合提高题}

\textbf{5.（计算题，\(\star\star\star\)）} 求两异面直线
\[
    L_1:\frac{x-1}{2}=\frac{y+1}{-1}=\frac{z-3}{1},\qquad
    L_2:\frac{x}{1}=\frac{y-2}{2}=\frac{z+1}{-2}
\]
的公垂线长度。

\vspace{0.5cm}

\boxed{\textbf{解}}
\(L_1\) 过点 \(P_1(1,-1,3)\)，方向向量 \(\mathbf{s}_1=(2,-1,1)\)；\\
\(L_2\) 过点 \(P_2(0,2,-1)\)，方向向量 \(\mathbf{s}_2=(1,2,-2)\)。

首先验证异面：
\[
    \mathbf{s}_1\times\mathbf{s}_2=\begin{vmatrix}
        \hat{\imath} & \hat{\jmath} & \hat{k} \\
        2 & -1 & 1 \\
        1 & 2 & -2
    \end{vmatrix}
    =(2-2)\hat{\imath}-(-4-1)\hat{\jmath}+(4+1)\hat{k}
    =(0,5,5)\neq\mathbf{0},
\]
故不平行。又
\[
    \overrightarrow{P_1P_2}=(-1,3,-4),
\]
\[
\begin{aligned}
(\overrightarrow{P_1P_2},\mathbf{s}_1,\mathbf{s}_2)
&=\begin{vmatrix}
    -1 & 3 & -4 \\
    2 & -1 & 1 \\
    1 & 2 & -2
\end{vmatrix}\\
&=-1(2-2)-3(-4-1)+(-4)(4+1)\\
&=0+15-20=-5\neq0,
\end{aligned}
\]
故两直线异面。

公垂线长度：
\[
    d=\frac{|(\overrightarrow{P_1P_2},\mathbf{s}_1,\mathbf{s}_2)|}{|\mathbf{s}_1\times\mathbf{s}_2|}
    =\frac{5}{\sqrt{0^2+5^2+5^2}}=\frac{5}{5\sqrt{2}}=\frac{\sqrt{2}}{2}.
\]


\textbf{6.（计算题，\(\star\star\)）} 将直线 \(L:\begin{cases}
    x+y+z-1=0,\\
    2x-y+3z+2=0
\end{cases}\) 化为对称式方程。

\vspace{0.5cm}

\boxed{\textbf{解}}
两平面的法向量分别为 \(\mathbf{n}_1=(1,1,1)\)，\(\mathbf{n}_2=(2,-1,3)\)，直线的方向向量
\[
    \mathbf{s}=\mathbf{n}_1\times\mathbf{n}_2=
    \begin{vmatrix}
        \hat{\imath} & \hat{\jmath} & \hat{k} \\
        1 & 1 & 1 \\
        2 & -1 & 3
    \end{vmatrix}
    =(3+1)\hat{\imath}-(3-2)\hat{\jmath}+(-1-2)\hat{k}
    =(4,-1,-3).
\]

求直线上一点：令 \(z=0\)，得
\[
    \begin{cases}
        x+y=1,\\
        2x-y=-2,
    \end{cases}
\]
相加得 \(3x=-1\Rightarrow x=-\frac{1}{3}\)，\(y=\frac{4}{3}\)。取点 \(P_0\left(-\dfrac{1}{3},\dfrac{4}{3},0\right)\)。

对称式方程：
\[
    \frac{x+\frac{1}{3}}{4}=\frac{y-\frac{4}{3}}{-1}=\frac{z}{-3}.
\]


\textbf{7.（计算题，\(\star\star\)）} 求 \(xOz\) 平面上的曲线 \(\begin{cases}z=x^2,\\ y=0\end{cases}\) 绕 \(z\) 轴旋转所得旋转曲面的方程，并说明其类型。

\vspace{0.5cm}

\boxed{\textbf{解}}
将曲线中的 \(x\) 替换为 \(\pm\sqrt{x^2+y^2}\)（保留 \(z\) 不变），得
\[
    z=(\pm\sqrt{x^2+y^2})^2=x^2+y^2.
\]
故旋转曲面方程为 \(z=x^2+y^2\)，它是一个旋转抛物面（椭圆抛物面的特例）。


\textbf{8.（计算题，\(\star\star\star\)）} 求通过直线 \(L:\dfrac{x-1}{2}=\dfrac{y}{-1}=\dfrac{z-2}{1}\) 且与平面 \(\Pi:x+2y-z+1=0\) 垂直的平面方程。

\vspace{0.5cm}

\boxed{\textbf{解}}
先将直线 \(L\) 化为一般式。由 \(\dfrac{x-1}{2}=\dfrac{y}{-1}\) 得 \(-x+1=2y\Rightarrow x+2y-1=0\)；
由 \(\dfrac{x-1}{2}=\dfrac{z-2}{1}\) 得 \(x-1=2z-4\Rightarrow x-2z+3=0\)。
故
\[
    L:\begin{cases}
        x+2y-1=0,\\
        x-2z+3=0.
    \end{cases}
\]

设所求平面方程为
\[
    \lambda(x+2y-1)+\mu(x-2z+3)=0,
\]
即 \((\lambda+\mu)x+2\lambda y-2\mu z+(-\lambda+3\mu)=0\)。其法向量 \(\mathbf{n}=(\lambda+\mu,2\lambda,-2\mu)\)。

与 \(\Pi\) 垂直 \(\Longrightarrow \mathbf{n}\perp\mathbf{n}_\Pi=(1,2,-1)\)：
\[
    (\lambda+\mu)\times 1+2\lambda\times 2+(-2\mu)\times(-1)=0
    \;\Longrightarrow\; \lambda+\mu+4\lambda+2\mu=0
    \;\Longrightarrow\; 5\lambda+3\mu=0.
\]
取 \(\lambda=3,\;\mu=-5\)，代入得
\[
    -2x+6y+10z-18=0\;\Longrightarrow\; -x+3y+5z-9=0.
\]
（验证：\(\mathbf{n}=(-1,3,5)\)，\(\mathbf{n}\cdot\mathbf{n}_\Pi=-1+6-5=0\)；直线上的点 \((1,0,2)\) 满足 \(-1+0+10-9=0\)，说明平面确实过该直线。）


\subsection{真题风格与综合训练}

\textbf{9.（\boxed{\textbf{真题风格}}，计算题，\(\star\star\star\)）}\par
判断下列四点是否共面：
\[
A(1,2,3),\quad B(2,4,5),\quad
C(0,0,0),\quad D(-1,-1,-3).
\]
若不共面，求以这四点为顶点的四面体体积。

\vspace{0.5cm}

\boxed{\textbf{解}}
取 \(C\) 为起点，构造向量：
\[
    \overrightarrow{CA}=(1,2,3),\quad \overrightarrow{CB}=(2,4,5),\quad \overrightarrow{CD}=(-1,-1,-3).
\]
计算混合积：
\[
    (\overrightarrow{CA},\overrightarrow{CB},\overrightarrow{CD})=
    \begin{vmatrix}
        1 & 2 & 3 \\
        2 & 4 & 5 \\
        -1 & -1 & -3
    \end{vmatrix}.
\]
\[
\begin{aligned}
(\overrightarrow{CA},\overrightarrow{CB},\overrightarrow{CD})
&=1(-12+5)-2(-6+5)+3(-2+4)\\
&=-7+2+6=1\ne0.
\end{aligned}
\]
故四点不共面，四面体体积为
\[
V=\frac16\left|(\overrightarrow{CA},\overrightarrow{CB},\overrightarrow{CD})\right|=\frac16.
\]


\textbf{10.（\boxed{\textbf{真题风格}}，计算题，\(\star\star\star\star\)）} 求过点 \(P(1,1,1)\) 且同时满足以下两个条件的直线方程：
\begin{enumerate}[label=(\arabic*)]
    \item 与直线 \(L_1:\dfrac{x-2}{1}=\dfrac{y-3}{2}=\dfrac{z}{3}\) 垂直；
    \item 与直线 \(L_2:\dfrac{x}{2}=\dfrac{y-1}{1}=\dfrac{z+1}{-1}\) 相交。
\end{enumerate}

\vspace{0.5cm}

\boxed{\textbf{解}}
设所求直线的方向向量为 \(\mathbf{d}=(d_1,d_2,d_3)\)。

由条件 (1)：\(\mathbf{d}\perp\mathbf{s}_1=(1,2,3)\)，故
\[
    d_1+2d_2+3d_3=0. \tag{1}
\]

由条件 (2)：\(L_2\) 过点 \(Q(0,1,-1)\)，方向为 \(\mathbf{s}_2=(2,1,-1)\)。\\
所求直线与 \(L_2\) 相交 \(\Longleftrightarrow\) \(\overrightarrow{PQ},\mathbf{s}_2,\mathbf{d}\) 共面，即
\[
    (\overrightarrow{PQ},\mathbf{s}_2,\mathbf{d})=0.
\]
其中 \(\overrightarrow{PQ}=Q-P=(-1,0,-2)\)。\\
（注意：\(P(1,1,1),\;Q(0,1,-1)\)，\(\overrightarrow{PQ}=(-1,0,-2)\)）
\[
\begin{aligned}
    \begin{vmatrix}
        -1 & 0 & -2 \\
        2 & 1 & -1 \\
        d_1 & d_2 & d_3
    \end{vmatrix}
    &=-(d_3+d_2)-2(2d_2-d_1)\\
    &=2d_1-5d_2-d_3=0. \qquad (2)
\end{aligned}
\]

由 (1) 得 \(d_3=-\frac{1}{3}(d_1+2d_2)\)，代入 (2)：
\[
    2d_1-5d_2+\frac{1}{3}(d_1+2d_2)=0
    \;\Longrightarrow\; 6d_1-15d_2+d_1+2d_2=0
    \;\Longrightarrow\; 7d_1-13d_2=0.
\]
取 \(d_2=7\)，则 \(d_1=13\)，\(d_3=-\frac{1}{3}(13+14)=-9\)。
\begin{center}
（或直接解齐次线性方程组 \(\begin{cases}d_1+2d_2+3d_3=0\\ 2d_1-5d_2-d_3=0\end{cases}\)）
\end{center}
故方向向量 \(\mathbf{d}=(13,7,-9)\)。

所求直线：
\[
    \frac{x-1}{13}=\frac{y-1}{7}=\frac{z-1}{-9}.
\]
（可验证：\(\mathbf{d}\cdot\mathbf{s}_1=13+14-27=0\)；交点可由 \(P+t\mathbf{d}=Q+v\mathbf{s}_2\) 解得 \(t=1,\;v=7\)，交点为 \((14,8,-8)\)。）


\textbf{11.（\boxed{\textbf{真题风格}}，证明/计算题，\(\star\star\star\)，综合）} 设 \(\mathbf{a}=(a_1,a_2,a_3),\;\mathbf{b}=(b_1,b_2,b_3),\;\mathbf{c}=(c_1,c_2,c_3)\) 为三个不共面的向量。
\begin{enumerate}[label=(\arabic*)]
    \item 证明：以 \(\mathbf{a},\mathbf{b},\mathbf{c}\) 为棱的平行六面体的体积 \(V=|(\mathbf{a},\mathbf{b},\mathbf{c})|\)；
    \item 若 \(\mathbf{a}=(1,0,1),\;\mathbf{b}=(1,1,0),\;\mathbf{c}=(0,1,1)\)，求以此三向量为棱的平行六面体的体积。
\end{enumerate}

\vspace{0.5cm}

\boxed{\textbf{解}}
\begin{enumerate}[label=(\arabic*)]
    \item 平行六面体的体积 = 底面积 \(\times\) 高。以 \(\mathbf{a},\mathbf{b}\) 为邻边的平行四边形面积 \(S=|\mathbf{a}\times\mathbf{b}|\)，
    高为 \(\mathbf{c}\) 在 \(\mathbf{a}\times\mathbf{b}\) 方向上的投影长度：
    \[
        h=|\mathbf{c}|\cdot|\cos\varphi|=\frac{|\mathbf{c}\cdot(\mathbf{a}\times\mathbf{b})|}{|\mathbf{a}\times\mathbf{b}|},
    \]
    故
    \[
        V=S\cdot h=|\mathbf{a}\times\mathbf{b}|\cdot\frac{|\mathbf{c}\cdot(\mathbf{a}\times\mathbf{b})|}{|\mathbf{a}\times\mathbf{b}|}
        =|(\mathbf{a},\mathbf{b},\mathbf{c})|.
    \]
    \item 计算混合积：
    \[
        (\mathbf{a},\mathbf{b},\mathbf{c})=
        \begin{vmatrix}
            1 & 0 & 1 \\
            1 & 1 & 0 \\
            0 & 1 & 1
        \end{vmatrix}
        =1\cdot\begin{vmatrix}1&0\\1&1\end{vmatrix}
        -0+1\cdot\begin{vmatrix}1&1\\0&1\end{vmatrix}
        =1\cdot 1+1\cdot 1=2.
    \]
    故体积 \(V=|2|=2\)。
\end{enumerate}
