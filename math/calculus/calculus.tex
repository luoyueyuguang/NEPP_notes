\section{微积分综合复习}

\subsection{基础综合题}

\textbf{1.} 设 $f(x)=\begin{cases}
\dfrac{\sin x}{x}, & x \neq 0,\\
1, & x=0,
\end{cases}$ $g(x)=\displaystyle\int_{0}^{x} e^{-t^{2}}\mathrm{d}t$，求极限 $\lim\limits_{x\to 0}\dfrac{g(f(x))-g(1)}{x^{2}}$。

\boxed{\textbf{解}}
由 $f(x)$ 定义知 $f(0)=1$，且 $\lim\limits_{x\to 0}f(x)=\lim\limits_{x\to 0}\frac{\sin x}{x}=1=f(0)$，故 $f(x)$ 在 $x=0$ 处连续。
由积分上限函数求导公式，$g'(x)=e^{-x^{2}}$。
所求极限为 $\frac{0}{0}$ 型，由洛必达法则：
\[
\lim_{x\to 0}\frac{g(f(x))-g(1)}{x^{2}} = \lim_{x\to 0}\frac{g'(f(x))\cdot f'(x)}{2x}.
\]
$g'(f(0))=g'(1)=e^{-1}$。又 $f'(x)=\frac{x\cos x-\sin x}{x^{2}}\;(x\neq 0)$，在 $x=0$ 处：
\[
f'(0)=\lim_{x\to 0}\frac{\frac{\sin x}{x}-1}{x}=\lim_{x\to 0}\frac{\sin x-x}{x^{2}}=\lim_{x\to 0}\frac{\cos x-1}{2x}=0.
\]
由泰勒展开 $\sin x=x-\frac{x^{3}}{6}+o(x^{3})$，$f(x)=\frac{\sin x}{x}=1-\frac{x^{2}}{6}+o(x^{2})$，故 $f'(x)=-\frac{x}{3}+o(x)$。
因此
\[
\lim_{x\to 0}\frac{g'(f(x))f'(x)}{2x}=\frac{e^{-1}}{2}\lim_{x\to 0}\frac{f'(x)}{x}=\frac{e^{-1}}{2}\cdot\left(-\frac{1}{3}\right)=-\frac{1}{6e}.
\]

\textbf{2.} 求曲线 $y=\displaystyle\int_{0}^{x} t e^{-t}\mathrm{d}t$ 的拐点坐标。

\boxed{\textbf{解}}
$y'=xe^{-x}$，$y''=e^{-x}-xe^{-x}=e^{-x}(1-x)$。令 $y''=0$ 得 $x=1$。
当 $x<1$ 时 $y''>0$（凹）；当 $x>1$ 时 $y''<0$（凸），故 $x=1$ 为拐点横坐标。
拐点纵坐标：
\[
y(1)=\int_{0}^{1}te^{-t}\mathrm{d}t = \left[-te^{-t}\right]_{0}^{1}+\int_{0}^{1}e^{-t}\mathrm{d}t=-e^{-1}-[e^{-t}]_{0}^{1}=-e^{-1}-e^{-1}+1=1-\frac{2}{e}.
\]
故拐点坐标为 $\bigl(1,\,1-\frac{2}{e}\bigr)$。

\textbf{3.} 计算二重积分 $\displaystyle\iint_{D} \sqrt{x^{2}+y^{2}}\,\mathrm{d}x\mathrm{d}y$，其中 $D$ 由 $x^{2}+y^{2}\leq 2x$ 确定。

\boxed{\textbf{解}}
$D$ 可改写为 $(x-1)^{2}+y^{2}\leq 1$，是以 $(1,0)$ 为心、半径为 $1$ 的圆。用极坐标 $x=r\cos\theta$，$y=r\sin\theta$，则边界 $r=2\cos\theta$，$\theta\in[-\frac{\pi}{2},\frac{\pi}{2}]$。
\[
\iint_{D}\sqrt{x^{2}+y^{2}}\,\mathrm{d}x\mathrm{d}y = \int_{-\pi/2}^{\pi/2}\mathrm{d}\theta\int_{0}^{2\cos\theta} r\cdot r\,\mathrm{d}r
= \int_{-\pi/2}^{\pi/2} \frac{8\cos^{3}\theta}{3}\,\mathrm{d}\theta.
\]
由对称性，
\[
I = \frac{16}{3}\int_{0}^{\pi/2}\cos^{3}\theta\,\mathrm{d}\theta = \frac{16}{3}\cdot\frac{2}{3} = \frac{32}{9}.
\]

\textbf{4.} 求幂级数 $\displaystyle\sum_{n=1}^{\infty} \frac{n}{3^{n}}x^{2n}$ 的收敛域与和函数。

\boxed{\textbf{解}}
令 $t=x^{2}$，考虑 $\sum_{n=1}^{\infty} \frac{n}{3^{n}}t^{n}$。收敛半径 $R=\lim\limits_{n\to\infty}\bigl|\frac{a_{n}}{a_{n+1}}\bigr|=\lim_{n\to\infty}\frac{n/3^{n}}{(n+1)/3^{n+1}}=3\lim_{n\to\infty}\frac{n}{n+1}=3$。
当 $t=3$ 时 $\sum n$ 发散；$t=-3$ 时 $\sum(-1)^{n}n$ 发散。故 $\sum_{n=1}^{\infty}\frac{n}{3^{n}}t^{n}$ 收敛域为 $(-3,3)$。
由 $t=x^{2}<3$ 得 $|x|<\sqrt{3}$，收敛域为 $(-\sqrt{3},\sqrt{3})$。

利用 $\sum_{n=1}^{\infty}nt^{n-1}=\frac{1}{(1-t)^{2}}\;(|t|<1)$，得
\[
\sum_{n=1}^{\infty} n t^{n} = \frac{t}{(1-t)^{2}},
\]
故
\[
S(x)=\sum_{n=1}^{\infty}\frac{n}{3^{n}}x^{2n}=\sum_{n=1}^{\infty}n\Bigl(\frac{x^{2}}{3}\Bigr)^{n}
=\frac{x^{2}/3}{(1-x^{2}/3)^{2}}=\frac{3x^{2}}{(3-x^{2})^{2}},\quad |x|<\sqrt{3}.
\]

\textbf{5.} 求微分方程 $y''-3y'+2y=2xe^{x}$ 的通解。

\boxed{\textbf{解}}
齐次方程特征方程为 $r^{2}-3r+2=0$，$r_{1}=1$，$r_{2}=2$，通解 $y_{h}=C_{1}e^{x}+C_{2}e^{2x}$。
右端 $2xe^{x}$ 中 $e^{x}$ 对应特征根 $r=1$（一重根），故设特解 $y^{*}=x(Ax+B)e^{x}=(Ax^{2}+Bx)e^{x}$。
令 $u=Ax^2+Bx$，则 $y^*=ue^x$，从而
\[
\begin{aligned}
(y^*)'&=(u'+u)e^x
=\bigl[Ax^2+(2A+B)x+B\bigr]e^x,\\
(y^*)''&=(u''+2u'+u)e^x\\
&=\bigl[Ax^2+(4A+B)x+2A+2B\bigr]e^x.
\end{aligned}
\]
代入 $y''-3y'+2y=2xe^{x}$，比较 $e^{x}$ 的系数：
\[
[Ax^{2}+(4A+B)x+(2A+2B)]-3[Ax^{2}+(2A+B)x+B]+2(Ax^{2}+Bx)=2x.
\]
整理得：$(-2A)x+(2A-B)=2x$，故 $-2A=2$，$2A-B=0$，解得 $A=-1$，$B=-2$。
特解 $y^{*}=(-x^{2}-2x)e^{x}$。通解：
\[
y=C_{1}e^{x}+C_{2}e^{2x}-(x^{2}+2x)e^{x}.
\]

\textbf{6.} 计算曲线积分 $\displaystyle I=\oint_{L} (e^{x}\sin y-2y)\mathrm{d}x+(e^{x}\cos y+3x)\mathrm{d}y$，其中 $L$ 为圆周 $x^{2}+y^{2}=1$（逆时针方向）。

\boxed{\textbf{解}}
记 $P=e^{x}\sin y-2y$，$Q=e^{x}\cos y+3x$，则
\[
\frac{\partial Q}{\partial x}-\frac{\partial P}{\partial y}=e^{x}\cos y+3-(e^{x}\cos y-2)=5.
\]
由 Green 公式，$D: x^{2}+y^{2}\leq 1$，
\[
I=\iint_{D}5\,\mathrm{d}x\mathrm{d}y=5\cdot\pi\cdot1^{2}=5\pi.
\]

\textbf{7.} 设 $z=f(xy,\,x^{2}+y^{2})$，其中 $f$ 具有二阶连续偏导数，求 $\dfrac{\partial^{2}z}{\partial x\partial y}$。

\boxed{\textbf{解}}
令 $u=xy$，$v=x^{2}+y^{2}$。则
\[
\frac{\partial z}{\partial x}=f_{1}'\cdot y+f_{2}'\cdot 2x.
\]
再对 $y$ 求偏导：
\[
\begin{aligned}
\frac{\partial^{2}z}{\partial x\partial y}
&= f_{1}' + y\bigl(f_{11}''\cdot x+f_{12}''\cdot 2y\bigr) + 2x\bigl(f_{21}''\cdot x+f_{22}''\cdot 2y\bigr)\\
&= f_{1}' + xy\,f_{11}'' + (2y^{2}+2x^{2})f_{12}'' + 4xy\,f_{22}''.
\end{aligned}
\]

\textbf{8.} 讨论反常积分 $\displaystyle\int_{0}^{+\infty} \frac{\ln(1+x)}{x^{p}}\,\mathrm{d}x$ 的敛散性与参数 $p$ 的关系。

\boxed{\textbf{解}}
积分有两个瑕点：$x=0$（当 $p>0$ 时）与 $x=+\infty$。分别讨论。

在 $x\to 0^{+}$ 处，$\ln(1+x)\sim x$，故被积函数 $\sim x^{1-p}$。$\int_{0}^{\delta}x^{1-p}\mathrm{d}x$ 收敛 $\Leftrightarrow 1-p>-1 \Leftrightarrow p<2$。

在 $x\to+\infty$ 处，$\ln(1+x)\sim\ln x$，被积函数 $\sim \frac{\ln x}{x^{p}}$。$\int_{M}^{+\infty}\frac{\ln x}{x^{p}}\mathrm{d}x$ 收敛 $\Leftrightarrow p>1$（当 $p=1$ 时 $\int\frac{\ln x}{x}\mathrm{d}x=\frac{1}{2}(\ln x)^{2}\to\infty$ 发散）。

综上，积分收敛当且仅当 $1<p<2$。


\subsection{提高综合题}

\textbf{9.}（极限与级数综合）设 $a_{1}=1$，$a_{n+1}=\dfrac{1}{2}\bigl(a_{n}+\dfrac{2}{a_{n}}\bigr)\;(n\geq 1)$。
\begin{enumerate}
\item[(1)] 证明 $\{a_{n}\}$ 收敛并求极限；
\item[(2)] 判断级数 $\displaystyle\sum_{n=1}^{\infty}(a_{n}-\sqrt{2})$ 的敛散性。
\end{enumerate}

\boxed{\textbf{解}}
(1) 易知 $a_{n}>0$。由均值不等式，$a_{n+1}=\frac{1}{2}(a_{n}+\frac{2}{a_{n}})\geq \sqrt{a_{n}\cdot\frac{2}{a_{n}}}=\sqrt{2}$，故 $n\geq 2$ 时 $a_{n}\geq\sqrt{2}$。
考察单调性：$a_{n+1}-a_{n}=\frac{1}{2}(\frac{2}{a_{n}}-a_{n})=\frac{2-a_{n}^{2}}{2a_{n}}\leq 0\;(n\geq 2)$，故 $\{a_{n}\}_{n\geq2}$ 单调递减有下界，收敛。设极限为 $A$，则 $A=\frac{1}{2}(A+\frac{2}{A})$，解得 $A=\sqrt{2}$。

(2) 令 $b_{n}=a_{n}-\sqrt{2}\to 0$。由递推：
\[
b_{n+1}=a_{n+1}-\sqrt{2}=\frac{a_{n}^{2}+2-2\sqrt{2}a_{n}}{2a_{n}}=\frac{(a_{n}-\sqrt{2})^{2}}{2a_{n}}=\frac{b_{n}^{2}}{2a_{n}}.
\]
故 $\frac{b_{n+1}}{b_{n}}=\frac{b_{n}}{2a_{n}}\to 0$，由比值判别法知 $\sum b_{n}$ 收敛（且收敛很快）。

\textbf{10.}（多元极值与条件极值）求函数 $f(x,y)=x^{2}+y^{2}-xy-x-y$ 在闭区域 $D=\{(x,y)\mid x^{2}+y^{2}\leq 4\}$ 上的最大值与最小值。

\boxed{\textbf{解}}
先求 $D$ 内部的驻点。令
\[
\begin{cases}
f_{x}=2x-y-1=0,\\
f_{y}=2y-x-1=0,
\end{cases}
\]
解得 $x=y=1$。此时 $f(1,1)=1+1-1-1-1=-1$。$f_{xx}=2$，$f_{yy}=2$，$f_{xy}=-1$，$H=f_{xx}f_{yy}-f_{xy}^{2}=3>0$ 且 $f_{xx}>0$，故为极小值点。

再在边界 $x^{2}+y^{2}=4$ 上用 Lagrange 乘数法。令 $L=x^{2}+y^{2}-xy-x-y+\lambda(x^{2}+y^{2}-4)$。
\[
\begin{cases}
L_{x}=2x-y-1+2\lambda x=0,\\
L_{y}=2y-x-1+2\lambda y=0,\\
x^{2}+y^{2}=4.
\end{cases}
\]
由前两式消去 $\lambda$：$(2x-y-1)y=(2y-x-1)x$，整理得
\[
(x-y)(x+y+1)=0.
\]

若 $x=y$，代入 $x^{2}+y^{2}=4$ 得 $x=y=\pm\sqrt{2}$。$f(\sqrt{2},\sqrt{2})=4-2-\sqrt{2}-\sqrt{2}=2-2\sqrt{2}$；$f(-\sqrt{2},-\sqrt{2})=2+2\sqrt{2}$。

若 $x+y=-1$，则
\[
x^2+y^2=4,\qquad
2xy=(x+y)^2-(x^2+y^2)=1-4=-3,
\]
故 $xy=-\frac32$，并且
\[
f=4-xy-(x+y)=4+\frac32+1=\frac{13}{2}.
\]
对应边界点为
\[
(x,y)=\left(\frac{-1+\sqrt7}{2},\frac{-1-\sqrt7}{2}\right)
\quad\text{或}\quad
\left(\frac{-1-\sqrt7}{2},\frac{-1+\sqrt7}{2}\right).
\]

比较各候选值可得
\[
-1<2-2\sqrt2<2+2\sqrt2<\frac{13}{2}.
\]
故最小值为 $-1$，在 $(1,1)$ 处取得；最大值为 $\frac{13}{2}$，
在上述两个满足 $x+y=-1$ 的边界点处取得。

\textbf{11.}（曲线积分与曲面积分综合）计算曲面积分
\[
I=\iint_{S} x^{3}\mathrm{d}y\mathrm{d}z+y^{3}\mathrm{d}z\mathrm{d}x+z^{3}\mathrm{d}x\mathrm{d}y,
\]
其中 $a>0$，$S$ 为球面 $x^{2}+y^{2}+z^{2}=a^{2}$ 的外侧。

\boxed{\textbf{解}}
由 Gauss 公式，设 $\Omega$ 为球体 $x^{2}+y^{2}+z^{2}\leq a^{2}$，
\[
I=\iiint_{\Omega}\bigl(3x^{2}+3y^{2}+3z^{2}\bigr)\mathrm{d}V = 3\iiint_{\Omega}(x^{2}+y^{2}+z^{2})\mathrm{d}V.
\]
用球坐标：$x=r\sin\varphi\cos\theta$，$y=r\sin\varphi\sin\theta$，$z=r\cos\varphi$。
\[
I=3\int_{0}^{2\pi}\mathrm{d}\theta\int_{0}^{\pi}\sin\varphi\,\mathrm{d}\varphi\int_{0}^{a} r^{2}\cdot r^{2}\,\mathrm{d}r
=3\cdot 2\pi\cdot 2\cdot\frac{a^{5}}{5}=\frac{12\pi a^{5}}{5}.
\]

\textbf{12.}（微分方程与定积分综合）设 $f(x)$ 在 $[0,+\infty)$ 上连续可导，且满足
\[
f'(x)+f(x)-\frac{1}{x+1}\int_{0}^{x} f(t)\mathrm{d}t=0,\quad f(0)=1.
\]
求 $f(x)$，并计算 $\displaystyle\int_{0}^{+\infty} f(x)\mathrm{d}x$。

\boxed{\textbf{解}}
令 $F(x)=\int_{0}^{x}f(t)\mathrm{d}t$，则 $F'(x)=f(x)$，$F''(x)=f'(x)$，$F(0)=0$，$F'(0)=1$。代入方程得
\[
F''(x)+F'(x)-\frac{1}{x+1}F(x)=0.
\]
由原方程
$f'=-f+I/(x+1)$ 可知其右端连续可导，因而 $f$ 实际上二阶可导。令
$I(x)=\int_{0}^{x}f(t)\mathrm{d}t$，原方程为
\[
f'(x)+f(x)-\frac{I(x)}{x+1}=0.\tag{1}
\]
对 (1) 求导：
\[
f''+f'-\frac{f(x)(x+1)-I(x)}{(x+1)^{2}}=0
\;\Longrightarrow\;
f''+f'-\frac{f(x)}{x+1}+\frac{I(x)}{(x+1)^{2}}=0.\tag{2}
\]
由 (1) 得 $\frac{I(x)}{x+1}=f'+f$，代入 (2)：
\[
f''+f'-\frac{f}{x+1}+\frac{f'+f}{x+1}=0
\;\Longrightarrow\;
f''+\frac{x+2}{x+1}f'=0.
\]
令 $p=f'$，则 $p'+\frac{x+2}{x+1}p=0$，分离变量得
\[
\frac{\mathrm{d}p}{p}=-\Bigl(1+\frac{1}{x+1}\Bigr)\mathrm{d}x
\;\Longrightarrow\;
\ln|p|=-x-\ln(x+1)+C
\;\Longrightarrow\;
p(x)=C\cdot\frac{e^{-x}}{x+1}.
\]
由原方程在 $x=0$ 处：$f'(0)+f(0)-0=0\Rightarrow f'(0)=-1$，故 $C=-1$，即 $f'(x)=-\dfrac{e^{-x}}{x+1}$。结合 $f(0)=1$，积分得
\[
f(x)=1-\int_{0}^{x}\frac{e^{-t}}{t+1}\mathrm{d}t.
\]
令 $J(x)=(x+1)(f'(x)+f(x))$。由上式可验得 $J'(x)=f(x)$ 且 $J(0)=0$，
所以 $J(x)=I(x)$；因此所得函数满足原积分微分方程，而非仅满足求导后的必要条件。

所求 $\int_{0}^{+\infty}f(x)\mathrm{d}x$。由 $f'(x)<0$ 知 $f(x)$ 严格递减。$f(+\infty)=1-\int_{0}^{\infty}\frac{e^{-t}}{t+1}\mathrm{d}t$。
注意到 $\int_{0}^{\infty}\frac{e^{-t}}{t+1}\mathrm{d}t<\int_{0}^{\infty}e^{-t}\mathrm{d}t=1$，故 $f(+\infty)=L>0$。
因 $\lim\limits_{x\to+\infty}f(x)=L>0$，由极限保号性，$\int_{0}^{+\infty}f(x)\mathrm{d}x$ 发散到 $+\infty$。

（注：也可直接由 $f(x)>f(+\infty)>0$ 及比较判别法得出积分发散。）

\textbf{13.}（中值定理与极限综合）设 $f(x)$ 在 $[0,1]$ 上连续，在 $(0,1)$ 内可导，$f(0)=0$，$f(1)=1$。证明：
\begin{enumerate}
\item[(1)] 存在 $\xi\in(0,1)$ 使得 $f(\xi)=1-\xi$；
\item[(2)] 存在 $\eta\in(0,1)$ 使得 $f'(\eta)=2\eta$。
\end{enumerate}

\boxed{\textbf{解}}
(1) 令 $\varphi(x)=f(x)+x-1$。$\varphi(0)=-1<0$，$\varphi(1)=1>0$。由零点定理，存在 $\xi\in(0,1)$ 使 $\varphi(\xi)=0$，即 $f(\xi)=1-\xi$。

(2) 令 $h(x)=f(x)-x^{2}$。$h(0)=f(0)-0=0$，$h(1)=f(1)-1=0$。由 Rolle 定理，存在 $\eta\in(0,1)$ 使 $h'(\eta)=0$，即 $f'(\eta)-2\eta=0$，故 $f'(\eta)=2\eta$。

\textbf{14.}（级数与积分综合）将函数 $f(x)=\arctan x$ 展开为 $x$ 的幂级数，并求 $\displaystyle\sum_{n=0}^{\infty}\frac{(-1)^{n}}{2n+1}$ 的和。

\boxed{\textbf{解}}
$f'(x)=\dfrac{1}{1+x^{2}}=\sum_{n=0}^{\infty}(-1)^{n}x^{2n}$，$|x|<1$。逐项积分：
\[
f(x)=\int_{0}^{x}\frac{\mathrm{d}t}{1+t^{2}}=\sum_{n=0}^{\infty}(-1)^{n}\frac{x^{2n+1}}{2n+1},\quad |x|<1.
\]
该幂级数在 $x=\pm1$ 处均由 Leibniz 判别法收敛，故收敛区间为 $[-1,1]$。
当 $x=1$ 时级数变为 $\sum_{n=0}^{\infty}\frac{(-1)^{n}}{2n+1}$，为交错级数，Leibniz 判别法知其收敛。由 Abel 定理，
\[
\arctan 1 = \frac{\pi}{4} = \sum_{n=0}^{\infty}\frac{(-1)^{n}}{2n+1}.
\]
即 $\displaystyle\sum_{n=0}^{\infty}\frac{(-1)^{n}}{2n+1}=\frac{\pi}{4}$。


\subsection{证明题专项}

\textbf{15.}（中值定理证明）设 $f(x)$ 在 $[0,1]$ 上二阶可导，$f(0)=f(1)=0$，且 $|f''(x)|\leq M$。证明：$|f'(x)|\leq \dfrac{M}{2}$ 对任意 $x\in[0,1]$ 成立。

\boxed{\textbf{解}}
先取任意 $x\in(0,1)$，由 Taylor 公式在 $x$ 处展开：
\[
f(0)=f(x)+f'(x)(0-x)+\frac{f''(\xi_{1})}{2}(0-x)^{2}=f(x)-xf'(x)+\frac{f''(\xi_{1})}{2}x^{2},\quad \xi_{1}\in(0,x).
\]
\[
f(1)=f(x)+f'(x)(1-x)+\frac{f''(\xi_{2})}{2}(1-x)^{2},\quad \xi_{2}\in(x,1).
\]
两式相减：$f(1)-f(0)=f'(x)+\frac{f''(\xi_{2})}{2}(1-x)^{2}-\frac{f''(\xi_{1})}{2}x^{2}$。因为 $f(0)=f(1)=0$，
\[
f'(x)=\frac{f''(\xi_{1})}{2}x^{2}-\frac{f''(\xi_{2})}{2}(1-x)^{2}.
\]
故
\[
|f'(x)|\leq \frac{M}{2}\bigl[x^{2}+(1-x)^{2}\bigr]=\frac{M}{2}\bigl[2x^{2}-2x+1\bigr].
\]
$g(x)=x^{2}+(1-x)^{2}=2x^{2}-2x+1$ 在 $[0,1]$ 上最大值在 $x=0$ 或 $x=1$ 处取得，$g(0)=g(1)=1$。故 $|f'(x)|\leq \frac{M}{2}$。

最后处理端点。由 $|f''|\leq M$ 及 Lagrange 中值定理，$f'$ 在 $[0,1]$ 上为 Lipschitz 连续函数；令上述内点 $x$ 分别趋于 $0^+$ 和 $1^-$，即可得到端点处同样的估计。

\textbf{16.}（不等式证明）证明：当 $x>0$ 时，$\dfrac{x}{1+x} < \ln(1+x) < x$。

\boxed{\textbf{解}}
令 $f(t)=\ln(1+t)$。对任意 $x>0$，在 $[0,x]$ 上用 Lagrange 中值定理：
\[
\ln(1+x)-\ln(1+0)=\frac{1}{1+\xi}\cdot x,\quad \xi\in(0,x).
\]
由于 $0<\xi<x$，有 $\frac{1}{1+x}<\frac{1}{1+\xi}<1$，故
\[
\frac{x}{1+x} < \frac{x}{1+\xi} < x,
\]
即 $\frac{x}{1+x} < \ln(1+x) < x$。

\textbf{17.}（级数证明）设 $a_{n}>0$，$\displaystyle\sum_{n=1}^{\infty} a_{n}$ 收敛。证明：
\begin{enumerate}
\item[(1)] $\displaystyle\sum_{n=1}^{\infty} \frac{\sqrt{a_{n}}}{n}$ 收敛；
\item[(2)] $\displaystyle\sum_{n=1}^{\infty} a_{n}^{2}$ 收敛。
\end{enumerate}

\boxed{\textbf{解}}
(1) 由 Cauchy-Schwarz 不等式（或 $2ab\leq a^{2}+b^{2}$），
\[
\frac{\sqrt{a_{n}}}{n} \leq \frac{1}{2}\Bigl(a_{n}+\frac{1}{n^{2}}\Bigr).
\]
由于 $\sum a_{n}$ 收敛且 $\sum \frac{1}{n^{2}}$ 收敛，由比较判别法知 $\sum \frac{\sqrt{a_{n}}}{n}$ 收敛。

(2) 由 $\sum a_{n}$ 收敛知 $a_{n}\to 0$，故存在 $N$ 使 $n>N$ 时 $0<a_{n}<1$。此时 $a_{n}^{2}<a_{n}$，由比较判别法知 $\sum a_{n}^{2}$ 收敛。

\textbf{18.}（综合证明）设 $f(x)$ 在 $[0,\pi]$ 上连续，且 $\displaystyle\int_{0}^{\pi} f(x)\sin x\,\mathrm{d}x = \int_{0}^{\pi} f(x)\cos x\,\mathrm{d}x = 0$。证明：$f(x)$ 在 $(0,\pi)$ 内至少有两个零点。
\boxed{\textbf{解}}
由 $\int_{0}^{\pi}f(x)\sin x\,\mathrm{d}x=0$ 和 $\sin x>0$ 在 $(0,\pi)$ 上，可知 $f(x)$ 在 $(0,\pi)$ 内至少变号一次（否则 $f$ 定号，乘积定号，积分为零只能 $f\equiv 0$，此时显然有无限多个零点）。故 $f$ 在 $(0,\pi)$ 内至少有一个零点，记为 $x_{1}$。

假设 $f$ 在 $(0,\pi)$ 内仅有一个零点 $x_1$。由连续性，$f$ 在
$(0,x_1)$ 和 $(x_1,\pi)$ 上分别保持固定符号。由于 $\sin x>0$ 且
$\int_0^\pi f(x)\sin x\,\mathrm dx=0$，两侧符号必须相反；否则积分不可能为零。

现考虑组合：
\[
\int_{0}^{\pi} f(x)\sin(x-x_{1})\mathrm{d}x
= \cos x_{1}\int_{0}^{\pi}f(x)\sin x\,\mathrm{d}x - \sin x_{1}\int_{0}^{\pi}f(x)\cos x\,\mathrm{d}x = 0.
\]
函数 $\sin(x-x_{1})$ 在 $x=x_{1}$ 处变号：$(0,x_{1})$ 上 $\sin(x-x_{1})<0$，$(x_{1},\pi)$ 上 $\sin(x-x_{1})>0$。

由于 $f(x)$ 与 $\sin(x-x_1)$ 都在 $x_1$ 两侧变号，乘积
$f(x)\sin(x-x_1)$ 在 $(0,x_1)$ 和 $(x_1,\pi)$ 上同号，且不恒为零，
所以其积分不可能为零，与上式矛盾。

故 $f$ 在 $(0,\pi)$ 内至少有两个零点。
