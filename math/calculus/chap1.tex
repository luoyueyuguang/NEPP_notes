\section{函数与数列的极限}
\subsection{极限}
\begin{mydef}{数列极限的定义}{definition of limit}
    \(\lim\limits_{n \to \infty} x_{n} = A ~
    \Leftrightarrow ~ \forall   \bold \varepsilon > 0
    ,\exists ~ \text{正整数}N,
    \text{使得当} n ~>~ N \text{时}
    \lvert x_{n} - A \rvert <  \bold \varepsilon.\) \\
    若数列\(\{x_{n}\}\)存在极限，又称数列\({x_{n}}\)收敛，
    否则数列\(\{x_{n}\}\)发散。
\end{mydef}

\begin{mydef}{无穷远处函数极限的定义}{definition of limit at infinity}
    \(\lim\limits_{x \to \infty} f(x) = A ~
    \Leftrightarrow ~ \forall  \bold \varepsilon > 0
    ,\exists ~ \text{正数}X
    \text{使得当} \lvert x \rvert  >  X \text{时}
    \lvert f(x) - A \rvert <  \bold \varepsilon.\) \\
    单侧极限：\\
    \(\lim\limits_{x \to +\infty} f(x) = A ~
    \Leftrightarrow ~ \forall  \bold \varepsilon > 0
    ,\exists ~ \text{正数}X
    \text{使得当} x > X \text{时}
    \lvert f(x) - A \rvert <  \bold \varepsilon.\) \\
    \(\lim\limits_{x \to -\infty} f(x) = A ~
    \Leftrightarrow ~ \forall  \bold \varepsilon > 0
    ,\exists ~ \text{正数}X
    \text{使得当} x < -X \text{时}
    \lvert f(x) - A \rvert <  \bold \varepsilon.\)
\end{mydef}

\begin{mydef}{函数极限的定义}{definition of limit2}
    \(\lim\limits_{x \to x_0} f(x) = A ~
    \Leftrightarrow ~ \forall  \bold \varepsilon > 0
    ,\exists ~ \text{正数} \delta
    \text{使得当} 0 < \lvert x - x_0 \rvert  <  \delta \text{时},
    \lvert f(x) - A \rvert <  \bold \varepsilon.\)
    类似可定义左极限与右极限。
\end{mydef}

\begin{conclusion}{数列极限的基本性质}{conclusion of limit}
    \begin{enumerate}
        \item 数列极限的不等式 \\
        设\(\lim\limits_{n \to \infty} x_{n} = A\)，
        \(\lim\limits_{n \to \infty} y_{n} = B\) \\
        若\(A > B \)，\( \exists N\), 当\(n > N\)时有
        \(x_{n} > y_{n}\); \\ 若\(n > N\)时有
        \(x_{n} < y_{n}\),则\(A \geq B\)。
        \item 收敛数列的有界性 \\
        设\(\{x_{n}\}\)是收敛数列，
        \(\lim\limits_{n \to \infty} x_{n} = A\)，
        则\(\{x_{n}\}\)有界。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{函数极限的基本性质}{conclusion of limit2}
    \begin{enumerate}
        \item 函数极限的不等式 \\
        设\(\lim\limits_{x \to x_0} f(x) = A\)，
        \(\lim\limits_{x \to x_0} g(x) = B\) \\
        若\(A > B \)，\( \exists \delta > 0,
        \text{使得当} 0 < \lvert x - x_0 \rvert  <  \delta \text{时}
        \text{有} f(x) > g(x),\)\\
        若\(\exists \delta > 0,
        \text{使得当} 0 < \lvert x - x_0 \rvert  <  \delta \text{时}
        \text{有} f(x) \geq g(x)\), 则\(A \geq B\). \\
        \(\text{若}f(x) > g(x),\text{仍然为}A \geq B\) \\
        函数极限的保号性 \\
        设\(\lim\limits_{x \to x_0} f(x) = A\)， \\ 
        若\(A > 0\),则\(\exists \delta > 0,
        \text{使得当} 0 < \lvert x - x_0 \rvert  <  \delta \text{时}
        \text{有} f(x) > 0\). \\
        若\(\exists \delta > 0,
        \text{使得当} 0 < \lvert x - x_0 \rvert  <  \delta \text{时}
        \text{有} f(x) \geq 0\), 则\(A \geq 0\).
        \item 存在极限的函数的局部有界性 \\
        设\(\lim\limits_{x \to x_0} f(x) = A\)，
        则\(f(x)\)在\(x_0\)的某空心邻域\(\bigcup_{0}(x_0, \delta) =
        \{x | 0 < |x - x_0| < \delta\}\text{内有界，即}
        \exists \delta > 0 \text{与} M > 0,\text{使得}
        \text{当} 0 < \lvert x - x_0 \rvert  <  \delta \text{时}
        \text{有} \lvert f(x) \rvert  \leq  M\).
    \end{enumerate}
\end{conclusion}

\begin{formula}{重要的极限}{important limit}
    \[
        \begin{aligned}
        &\lim\limits_{x \to 0} \frac{\sin x}{x} = 1,\\
        &\lim\limits_{x \to 0} (1+x)^{\frac{1}{x}} = e \text{或} \\
        &\lim\limits_{x \to \infty} (1+\frac{1}{x})^{x} = e, \\
        &\lim\limits_{x \to 0} \frac{\ln(1+x)}{x} = 1
        \end{aligned}
    \]
\end{formula}

\begin{conclusion}{夹逼定理}{bracket theorem}
    \begin{enumerate}
        \item 数列
        \(\exists N\), 当\(n > N\)时有\(y_n \leq x_n \leq z_n\)，
        \(\lim\limits_{n \to \infty} y_{n} = a\)，
        \(\lim\limits_{n \to \infty} z_{n} = a\)，
        则
        \(\lim\limits_{n \to \infty} x_{n} = a\)。
        \item 函数
        \(\exists \delta > 0\), 当\(0 < \lvert x - x_0 \rvert  <  \delta\)时有
        \(h(x) \leq f(x) \leq g(x)\),又\(\lim\limits_{x \to x_0} h(x) = A\),
        \(\lim\limits_{x \to x_0} g(x) = A\)，
        则\(\lim\limits_{x \to x_0} f(x) = A\)。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{单调有界数列必收敛定理}{monotone and bounded sequence must converge}
    若数列\(\{x_n\}\)单调上升有上界，即\(x_{n+1} \geq x_n\)且\(\exists M\), 使得对所有\(n\)有\(x_n \leq M\)，即
    存在一个数\(a\),
    使得\(\lim\limits_{n \to \infty} x_n = a\)，且\(x_n \leq a\). \\
    若数列\(\{x_n\}\)单调下降有下界，即\(x_{n+1} \leq x_n\)且\(\exists M\), 使得对所有\(n\)有\(x_n \geq M\)，即
    存在一个数\(a\),
    使得\(\lim\limits_{n \to \infty} x_n = a\)，且\(x_n \geq a\).
\end{conclusion}

\begin{conclusion}{极限存在的充要条件}{limit exists if and only if}
    \begin{enumerate}
        \item 函数极限 \\
        \(\lim\limits_{x \to x_0} f(x) = A \Leftrightarrow
        \lim\limits_{x \to x_0^-} f(x) = A =
        \lim\limits_{x \to x_0^+} f(x) = A\) \\
        即左右极限都存在且相等。
        \item 数列极限 \\
        数列\(\{x_n\}\)收敛于\(A\)的充要条件是它的任意子列都收敛于\(A\)。特别地，若偶数项子列与奇数项子列都收敛于同一个\(A\)，则原数列收敛于\(A\)。
    \end{enumerate}
\end{conclusion}

\begin{formula}{证明极限不存在的常用方法}{proof of limit not existing}
    \begin{enumerate}
        \item \(x \to x_0\)时，左右极限不相等则极限不存在 \\
        \(x \to \infty\)时，\(-\infty \text{与} +\infty\)极限不相等，则极限不存在
        \item 若存在\(x_n\to x_0\)（且\(x_n\neq x_0\)），使\(f(x_n)\)不趋于候选极限\(A\)，则\(\lim\limits_{x\to x_0}f(x)\neq A\)。\\
        若存在\(x_n\to x_0\)、\(y_n\to x_0\)（各项均不等于\(x_0\)），且\(f(x_n)\)、\(f(y_n)\)分别趋于两个不同的极限，则\(\lim\limits_{x\to x_0}f(x)\)不存在。
        \item 利用结论， \(\lim\limits_{x \to x_0} f(x) = A\),\(\lim\limits_{x \to x_0} g(x)\)不存在。
        则\(\lim\limits_{x \to x_0} [f(x) + g(x)]\)不存在。若\(A\)不为零，
        则\(\lim\limits_{x \to x_0} [f(x)\cdot g(x)]\)不存在。
    \end{enumerate}
\end{formula}

\subsection{求极限的方法}

\begin{conclusion}{极限的四则运算法则}{limit of sum and product}
    设\(\lim\limits_{x\to x_0}f(x)=A\)、\(\lim\limits_{x\to x_0}g(x)=B\)，其中\(A,B\)为有限数，则
    \begin{enumerate}
        \item \(\lim\limits_{x \to x_0} [f(x) \pm g(x)] = A\pm B\)；
        \item \(\lim\limits_{x \to x_0} [f(x)g(x)] = AB\)；
        \item 当\(B\neq0\)时，\(\lim\limits_{x \to x_0} \dfrac{f(x)}{g(x)} = \dfrac{A}{B}\)。
    \end{enumerate}
    \textbf{注：}若某个分项极限不存在、为无穷大，或商的分母极限为零，不能机械套用上述法则，应化简后具体分析；例如\(x+(-x)\equiv0\)，不能由两个分项均无界推出和无界。
\end{conclusion}

\begin{conclusion}{幂指数函数的极限运算法则}{limit of power and exponential function}
    设在所考察的去心邻域内\(f(x)>0\)，\(\lim\limits_{x\to x_0}f(x)=A\)，\(\lim\limits_{x\to x_0}g(x)=B\)。当\(A>0\)且\(B\)为有限数时，由连续性有
    \[
        \lim\limits_{x\to x_0}f(x)^{g(x)}=A^B.
    \]
    其余情形需分别判断：
    \begin{enumerate}
        \item 当\(0<A<1\)或\(A>1\)，且\(g(x)\to+\infty\)时，
        \[
            \begin{cases}
                0, & 0 < A < 1 \\
                +\infty, & A > 1 \\
            \end{cases}
        \]
        \item 当\(A=0\)且\(B\neq0\)为有限数时，若幂指函数在邻域内有定义，则
        \[
            \lim\limits_{x\to x_0}f(x)^{g(x)}=
            \begin{cases}0,&B>0,\\+\infty,&B<0;\end{cases}
        \]
        \item 当\(A=+\infty\)且\(B\neq0\)为有限数时，
        \[
            \lim\limits_{x\to x_0}f(x)^{g(x)}=
            \begin{cases}+\infty,&B>0,\\0,&B<0.\end{cases}
        \]
        \item \(1^\infty\)、\(0^0\)、\(\infty^0\)为未定式，通常令\(f(x)^{g(x)}=\exp[g(x)\ln f(x)]\)后研究指数的极限。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{运算法则不适用的未定式}{limit of undefined function}
    \(\frac{0}{0}, \frac{\infty}{\infty}, 0 \cdot \infty, \infty - \infty,
    0^0, \infty^0, 1^\infty\)
\end{conclusion}

\begin{conclusion}{利用函数的连续性求极限}{limit of function by continuity}
    \(\lim\limits_{x \to x_0} f(x) = A\)，若\(f(x)\)在\(x_0\)处连续，则\(f(x)\)在\(x_0\)处的极限为\(A = f(x_0)\)。
\end{conclusion}

\begin{conclusion}{利用变量替换法求极限}{limit of function by substitution}
    \begin{enumerate}
    \item \(\lim\limits_{x \to +\infty} \varphi(x) = +\infty \text{且} \lim\limits_{u \to \infty} f(u) = A\), \\
    则 \(\lim\limits_{x \to +\infty} f[\varphi(x)] \xlongequal{x= \varphi(x)}
    \lim\limits_{u \to \infty} f(u) = A.\)
    \item \(\lim\limits_{x \to x_0} \varphi(x) = u_0\)且\(f(u)\)在\(u_0\)处连续， \\
    则\(\lim\limits_{x \to x_0} f[\varphi(x)] \underset{x \to x_0, u \to u_0}{\xlongequal{u= \varphi(x)}}
    \lim\limits_{u \to u_0} f(u) = A.\)
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{利用等价无穷小代换求极限}{limit by equivalent infinitesimals}
    若\(x \to a\)时有,无穷小\(\alpha(x) \sim \alpha^*(x), \beta(x) \sim \beta^*(x)\)， \\
    则在相关表达式有定义时，
    \[
      \frac{\alpha(x)u(x)}{\beta(x)v(x)}
      \sim \frac{\alpha^*(x)u(x)}{\beta^*(x)v(x)}.
    \]
    因而两者的极限同时存在时相等。注意等价无穷小在极限点的值通常均为\(0\)，不能把最后结果写成在\(x=a\)处直接代入的\(0/0\)。
\end{conclusion}

\begin{conclusion}{利用洛必达法则求极限}{limit of function by l'hospital rule}
    设\(f,g\)在点\(a\)的某个去心邻域内可导，且\(g'(x)\neq0\)。若\(f(x)/g(x)\)为\(0/0\)型或\(\infty/\infty\)型，并且
    \[
        \lim\limits_{x\to a}\frac{f'(x)}{g'(x)}=A
    \]
    存在（\(A\)也可为无穷大），则
    \[
        \lim\limits_{x\to a}\frac{f(x)}{g(x)}=A.
    \]
    单侧极限和\(x\to\infty\)情形有相应版本。导数之比极限不存在时，原极限仍可能存在，不能反向使用本法则。
\end{conclusion}

\begin{conclusion}{求数列极限的方法}{limit of sequence}
    \begin{enumerate}
        \item 利用函数极限求数列极限,将所求数列\(\{y_n\}\)看作函数\(f(x)\)在数列\(\{x_n\}\)上的取值,
        则\(\lim\limits_{n \to \infty} y_n = \lim\limits_{n \to \infty} f(x_n) = \lim\limits_{x \to +\infty} f(x) = A\)
        \item 使用放缩法,处理余项
        \item 对递推数列\(y_{n+1}=f(y_n)\)，应先证明数列收敛（常用单调有界法）并确认\(f\)在候选极限处连续；随后才可令\(n\to\infty\)得到必要方程\(A=f(A)\)。固定点方程本身不能证明数列收敛。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{其他求极限的方法}{other methods to calculate limit}
    \begin{enumerate}
        \item 利用导数定义求极限
        \item 利用泰勒公式求极限
        \item 利用定积分求极限
    \end{enumerate}
\end{conclusion}

\subsection{无穷小}
\begin{mydef}{无穷小的定义}{infinitesimal}
    \begin{enumerate}
        \item 数列
        若\(\lim\limits_{n \to \infty} x_n = 0,\)则称数列\({x_n}\)为无穷小，记为\(x_n = o(1)(n \to \infty)\)
        \item 函数
        若\(\lim\limits_{x \to x_0} f(x) = 0,\)则称函数\(f(x)\)在\(x_0\)处为无穷小，记为\(f(x) = o(1)(x \to x_0)\)
    \end{enumerate}
\end{mydef}

\begin{mydef}{无穷大的定义}{infinite}
    \begin{enumerate}
        \item 数列 \\
        若\(\forall M > 0, \exists N, \forall n > N, |x_n| > M\),则称数列\({x_n}\)为无穷大，记为\(\lim\limits_{n \to \infty} x_n = \infty\)
        \item 函数 \\
        若\(\forall M > 0, \exists X > 0, \forall |x| > X, |f(x)| > M\),则称\(x \to \infty\)时\(f(x)\)为无穷大量，记为\(\lim\limits_{x \to \infty} f(x) = \infty\) \\
        若\(\forall M > 0, \exists \delta > 0, 0 < |x - x_0| < \delta\)时有\(|f(x)| > M\),则称\(x \to x_0\)时\(f(x)\)为无穷大量，记为\(\lim\limits_{x \to x_0} f(x) = \infty\)
    \end{enumerate}
\end{mydef}

\begin{conclusion}{无穷小的关系}{infinitesimal relation}
    \begin{enumerate}
        \item 与极限的关系 \\
        \(\lim\limits_{x \to x_0} f(x) = A \Leftrightarrow f(x) = A + \alpha(x),
        \lim\limits_{x \to x_0} \alpha(x) = 0\)
        \item 与无穷大的关系
        若\(f(x)\)为无穷小，若\(f(x) \neq 0\),则\(\frac{1}{f(x)}\)为无穷大，反之亦成立。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{无穷小的运算性质}{infinitesimal properties}
    \begin{enumerate}
        \item 有限个无穷小的代数和仍为无穷小
        \item 有限个无穷小的代数积仍为无穷小
        \item 有界变量与无穷小的乘积仍为无穷小
    \end{enumerate}
\end{conclusion}

\begin{mydef}{无穷小阶的定义}{infinitesimal order}
    设\(\alpha(x)\)、\(\beta(x)\)在同一极限过程中均为无穷小，且\(\beta(x)\neq0\)。若
    \[
      \lim\frac{\alpha(x)}{\beta(x)^k}=c\neq0,
    \]
    则称\(\alpha\)是关于\(\beta\)的\(k\)阶无穷小；若该比值为\(0\)，称\(\alpha\)是\(\beta^k\)的高阶无穷小；若比值为无穷大，称其为低阶无穷小。特别地，\(\alpha\sim\beta\)表示同阶且比值趋于\(1\)。
\end{mydef}

\begin{formula}{常见的等价无穷小}{common infinitesimal}
    当\(x \to 0\)时，有
    \begin{enumerate}
        \item \(\mathrm{sin}x \sim x\)
        \item \(\mathrm{tan}x \sim x\)
        \item \(\mathrm{arcsin}x \sim x\)
        \item \(\mathrm{arctan}x \sim x\)
        \item \(1 - \mathrm{cos}x \sim \frac{1}{2}x^2\)
        \item \(\mathrm{e}^x - 1 \sim x\)
        \item \(\mathrm{ln}(1 + x) \sim x\)
        \item \(a^x-1 \sim x\ln{a}(a > 0, a \neq 1)\)
        \item \({(1 + \beta x)}^\alpha -1 \sim \alpha \beta x\)
        \item \(\log_a{(1 + x)} \sim \frac{1}{\ln a}x(a > 0, a \neq 1)\)
        \item \(\sqrt[n]{1+x} - 1 \sim \frac{1}{n}x\)
    \end{enumerate}
\end{formula}

\begin{conclusion}{等价无穷小的重要性质}{important infinitesimal properties}
    \begin{enumerate}
        \item 等价无穷小具有自反性、对称性、传递性。
        \item 等价无穷小代换只适用于乘除运算，不适用于加减运算。\\
        \boxed{\text{注}} 若\(f(x) \sim f^*(x), g(x) \sim g^*(x)\)，则
        \[
            \lim \frac{f(x)}{g(x)} = \lim \frac{f^*(x)}{g^*(x)},
            \quad \lim f(x)h(x) = \lim f^*(x)h(x)
        \]
        但\(\lim [f(x) - g(x)]\)一般不能直接代换。
        \item 若\(\alpha \sim \tilde{\alpha}, \beta \sim \tilde{\beta}\)，且\(\lim \frac{\tilde{\alpha}}{\tilde{\beta}}\)存在，
        则\(\lim \frac{\alpha}{\beta} = \lim \frac{\tilde{\alpha}}{\tilde{\beta}}\)。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{确定无穷小阶的方法}{method to determine infinitesimal order}
    若\(\lim \frac{f(x)}{(x-a)^k} = c \neq 0\)（\(c\)为非零常数），则称\(f(x)\)是\(x \to a\)时的\textbf{\(k\)阶无穷小}。
    常用方法：
    \begin{enumerate}
        \item \textbf{等价代换法}：利用已知等价无穷小直接判定阶数；
        \item \textbf{泰勒展开法}：将函数展开为泰勒公式，最低次幂即为无穷小的阶；
        \item \textbf{求导法}：若\(f(a) = f'(a) = \cdots = f^{(k-1)}(a) = 0\)而\(f^{(k)}(a) \neq 0\)，
        则\(f(x)\)是\(x \to a\)时的\(k\)阶无穷小。
    \end{enumerate}
\end{conclusion}


\subsection{函数的连续性}
\begin{mydef}{连续性}{continuity}
    设函数\(f(x)\)在点\(x_0\)的某邻域内有定义，若
    \[
        \lim_{x \to x_0} f(x) = f(x_0)
    \]
    则称\(f(x)\)在点\(x_0\)处\textbf{连续}。\\
    等价表述：\(f(x)\)在\(x_0\)处连续 \(\Leftrightarrow\) \(\lim_{\Delta x \to 0} \Delta y = 0\)，
    其中\(\Delta y = f(x_0 + \Delta x) - f(x_0)\)。\\
    若\(f(x)\)在区间\(I\)的每一点都连续，则称\(f(x)\)在区间\(I\)上连续。
\end{mydef}

\begin{mydef}{间断点的定义}{discontinuity point}
    若函数\(f(x)\)在点\(x_0\)处不连续，则称\(x_0\)为\(f(x)\)的\textbf{间断点}。间断点出现的三种情况：
    \begin{enumerate}
        \item \(f(x)\)在\(x_0\)处无定义；
        \item \(f(x)\)在\(x_0\)处有定义但\(\lim_{x \to x_0} f(x)\)不存在；
        \item \(f(x)\)在\(x_0\)处有定义且极限存在，但\(\lim_{x \to x_0} f(x) \neq f(x_0)\)。
    \end{enumerate}
\end{mydef}

\begin{conclusion}{间断点的分类}{classification of discontinuity point}
    \begin{enumerate}
        \item \textbf{第一类间断点}：左右极限均存在。\\
        —— 若左右极限相等（即可去间断点），记\(\lim_{x \to x_0} f(x) = A \neq f(x_0)\)或\(f(x_0)\)无定义；\\
        —— 若左右极限不相等（即跳跃间断点），\(\lim_{x \to x_0^-} f(x) \neq \lim_{x \to x_0^+} f(x)\)。
        \item \textbf{第二类间断点}：左右极限至少有一个不存在。\\
        —— 若极限为\(\infty\)，称为无穷间断点；\\
        —— 若极限振荡不存在，称为振荡间断点（如\(\sin\frac{1}{x}\)在\(x=0\)处）。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{判断函数连续性的方法}{method to judge continuity}
    \begin{enumerate}
        \item \textbf{定义法}：验证\(\lim_{x \to x_0} f(x) = f(x_0)\)；
        \item \textbf{左右极限法}：验证\(\lim_{x \to x_0^-} f(x) = \lim_{x \to x_0^+} f(x) = f(x_0)\)；
        \item \textbf{运算法则法}：利用连续函数的四则运算、复合运算保持连续性；
        \item \textbf{初等函数法}：一切初等函数在其定义区间内都是连续的。
    \end{enumerate}
\end{conclusion}

\begin{formula}{连续性运算法则}{continuity operation rule}
    设\(f(x), g(x)\)在点\(x_0\)处连续，则：
    \begin{enumerate}
        \item \(f(x) \pm g(x)\)在\(x_0\)处连续；
        \item \(f(x) \cdot g(x)\)在\(x_0\)处连续；
        \item \(\frac{f(x)}{g(x)}\)在\(x_0\)处连续（要求\(g(x_0) \neq 0\)）；
        \item 若\(u = \varphi(x)\)在\(x_0\)处连续，\(y = f(u)\)在\(u_0 = \varphi(x_0)\)处连续，
        则复合函数\(f(\varphi(x))\)在\(x_0\)处连续。
    \end{enumerate}
\end{formula}

\begin{conclusion}{连续函数的局部保号性质}{local sign property of continuity function}
    若\(f(x)\)在\(x_0\)处连续且\(f(x_0) > 0\)（或\(< 0\)），则存在\(\delta > 0\)，
    使得当\(|x - x_0| < \delta\)时，\(f(x) > 0\)（或\(< 0\)）。\\
    即连续函数在一点处函数值的正负号可以"保持"到该点的一个小邻域内。
\end{conclusion}

\begin{conclusion}{有界闭区间上连续函数的性质}{properties of continuity function on bounded closed interval}
    设\(f(x)\)在闭区间\([a, b]\)上连续，则：
    \begin{enumerate}
        \item \textbf{有界性定理}：\(f(x)\)在\([a, b]\)上有界；
        \item \textbf{最值定理}：\(f(x)\)在\([a, b]\)上必能取到最大值和最小值；
        \item \textbf{介值定理}：\(f(x)\)可以取到介于最小值和最大值之间的任何值；
        \item \textbf{零点定理}：若\(f(a) \cdot f(b) < 0\)，则存在\(\xi \in (a, b)\)使得\(f(\xi) = 0\)。
    \end{enumerate}
    \boxed{\text{注}} 以上性质均要求\textbf{闭区间}和\textbf{连续}两个条件同时满足，缺一不可。
\end{conclusion}

\begin{conclusion}{连续介值定理的应用}{application of intermediate value theorem for continuity function}
    介值定理及其特例——零点定理——是证明方程根存在性的核心工具：
    \begin{enumerate}
        \item \textbf{证明方程有根}：构造辅助函数\(F(x) = f(x) - g(x)\)，验证\(F(a)F(b) < 0\)；
        \item \textbf{证明存在\(\xi\)使\(f(\xi) = c\)}：令\(F(x) = f(x) - c\)，验证\(F(a)F(b) \leq 0\)；
        \item \textbf{与中值定理结合}：用介值定理确定函数值的范围，再用罗尔定理或拉格朗日中值定理推出导数的结论。
    \end{enumerate}
    \boxed{\textbf{例}} 证明方程\(x^3 - 3x + 1 = 0\)在\((0, 1)\)内至少有一个实根。\\
    令\(f(x) = x^3 - 3x + 1\)，\(f(0) = 1 > 0\)，\(f(1) = -1 < 0\)，由零点定理即得。
\end{conclusion}

\subsection{基础过关题}

\textbf{1.} (选择题) 设 \(\lim\limits_{n \to \infty} a_n = a\), \(\lim\limits_{n \to \infty} b_n = b\), 则下列命题中正确的是\underline{\hspace{2cm}}。

\begin{enumerate}
    \item[(A)] 若 \(a_n > b_n\) (\(n = 1,2,\ldots\)), 则 \(a > b\)
    \item[(B)] 若 \(a_n \geq b_n\) (\(n = 1,2,\ldots\)), 则 \(a \geq b\)
    \item[(C)] 若 \(a > b\), 则存在 \(N\), 当 \(n > N\) 时, \(a_n > b_n\)
    \item[(D)] 若 \(\{a_n\}\) 单调递增, 则 \(\{a_n\}\) 必有界
\end{enumerate}

\boxed{\textbf{解}} 由极限的保号性可知, 若 \(a > b\), 则当 \(n\) 充分大时必有 \(a_n > b_n\), 故选 \textbf{(C)}。

(A) 的反例: \(a_n = \frac{1}{n}\), \(b_n = 0\), 则 \(a_n > b_n\) 但 \(\lim a_n = \lim b_n = 0\);
(B) 同理, 严格不等号在取极限后可能退化为等号;
(D) 单调递增不一定有界, 反例 \(a_n = n\)。

\medskip

\textbf{2.} (填空题) \(\displaystyle\lim_{x \to 0} \frac{\sin 3x}{\tan 5x} = \underline{\hspace{2cm}}\)。

\boxed{\textbf{解}} 当 \(x \to 0\) 时, \(\sin 3x \sim 3x\), \(\tan 5x \sim 5x\), 故
\[
\lim_{x \to 0} \frac{\sin 3x}{\tan 5x} = \lim_{x \to 0} \frac{3x}{5x} = \frac{3}{5}.
\]

\medskip

\textbf{3.} (计算题) 求极限 \(\displaystyle\lim_{n \to \infty} \frac{(n+1)(2n-1)}{3n^2 + n - 2}\)。

\boxed{\textbf{解}} 分子分母同除以 \(n^2\):
\[
\lim_{n \to \infty} \frac{(n+1)(2n-1)}{3n^2 + n - 2}
= \lim_{n \to \infty} \frac{\left(1 + \frac{1}{n}\right)\left(2 - \frac{1}{n}\right)}{3 + \frac{1}{n} - \frac{2}{n^2}}
= \frac{1 \cdot 2}{3} = \frac{2}{3}.
\]

\medskip

\textbf{4.} (计算题) 求极限 \(\displaystyle\lim_{x \to 0} \frac{\sqrt{1+2x} - 1}{\ln(1+x)}\)。

\boxed{\textbf{解}} 当 \(x \to 0\) 时,
\[
\sqrt{1+2x} - 1 \sim \frac{1}{2} \cdot 2x = x, \qquad \ln(1+x) \sim x,
\]
故
\[
\lim_{x \to 0} \frac{\sqrt{1+2x} - 1}{\ln(1+x)} = \lim_{x \to 0} \frac{x}{x} = 1.
\]

\medskip

\textbf{5.} (计算题) 求极限 \(\displaystyle\lim_{x \to 0} \frac{\ln(1+2x)}{e^{3x} - 1}\)。

\boxed{\textbf{解}} 当 \(x \to 0\) 时, \(\ln(1+2x) \sim 2x\), \(e^{3x} - 1 \sim 3x\), 故
\[
\lim_{x \to 0} \frac{\ln(1+2x)}{e^{3x} - 1} = \lim_{x \to 0} \frac{2x}{3x} = \frac{2}{3}.
\]

\medskip

\textbf{6.} (计算题) 利用夹逼定理求极限
\[
\lim_{n \to \infty} \left( \frac{n}{n^2+1} + \frac{n}{n^2+2} + \cdots + \frac{n}{n^2+n} \right).
\]

\boxed{\textbf{解}} 记 \(\displaystyle S_n = \sum_{k=1}^{n} \frac{n}{n^2+k}\)。注意到对每个 \(k = 1,2,\ldots,n\), 有
\[
\frac{n}{n^2+n} \leq \frac{n}{n^2+k} \leq \frac{n}{n^2+1},
\]
因此
\[
n \cdot \frac{n}{n^2+n} \leq S_n \leq n \cdot \frac{n}{n^2+1}
\quad\Longrightarrow\quad
\frac{n^2}{n^2+n} \leq S_n \leq \frac{n^2}{n^2+1}.
\]
而
\[
\lim_{n \to \infty} \frac{n^2}{n^2+n} = \lim_{n \to \infty} \frac{n^2}{n^2+1} = 1,
\]
由夹逼定理得 \(\displaystyle\lim_{n \to \infty} S_n = 1\)。

\subsection{综合提高题}

\textbf{7.} (证明题) 用 \(\varepsilon\text{-}N\) 语言证明: \(\displaystyle\lim_{n \to \infty} \frac{2n+1}{3n-2} = \frac{2}{3}\)。

\boxed{\textbf{解}} \(\forall \varepsilon > 0\), 要使当 \(n\) 充分大时成立
\[
\left| \frac{2n+1}{3n-2} - \frac{2}{3} \right| < \varepsilon.
\]

计算:
\[
\left| \frac{2n+1}{3n-2} - \frac{2}{3} \right|
= \left| \frac{3(2n+1) - 2(3n-2)}{3(3n-2)} \right|
= \left| \frac{7}{3(3n-2)} \right|
= \frac{7}{3(3n-2)} \quad (n \geq 1).
\]

要使 \(\displaystyle\frac{7}{3(3n-2)} < \varepsilon\), 只需
\[
3n-2 > \frac{7}{3\varepsilon} \quad\Longrightarrow\quad n > \frac{1}{3}\left(\frac{7}{3\varepsilon} + 2\right).
\]

取 \(\displaystyle N = \left[\frac{7}{9\varepsilon} + \frac{2}{3}\right] + 1\)（其中 \([\,\cdot\,]\) 表示取整）, 则当 \(n > N\) 时,
\[
\left| \frac{2n+1}{3n-2} - \frac{2}{3} \right| < \varepsilon.
\]

由数列极限的定义, \(\displaystyle\lim_{n \to \infty} \frac{2n+1}{3n-2} = \frac{2}{3}\)。

\medskip

\textbf{8.} (计算题) 求极限 \(\displaystyle\lim_{x \to 0} (\cos x)^{\frac{1}{x^2}}\)。

\boxed{\textbf{解}} 这是 \(1^\infty\) 型未定式。利用重要极限 \(\displaystyle\lim_{u \to 0} (1+u)^{\frac{1}{u}} = e\):
\[
(\cos x)^{\frac{1}{x^2}} = \bigl[1 + (\cos x - 1)\bigr]^{\frac{1}{x^2}}
= \Bigl[1 + (\cos x - 1)\Bigr]^{\frac{1}{\cos x - 1} \cdot \frac{\cos x - 1}{x^2}}.
\]

当 \(x \to 0\) 时, \(\cos x - 1 \sim -\dfrac{1}{2}x^2\), 故
\[
\frac{\cos x - 1}{x^2} \to -\frac{1}{2}.
\]

因此
\[
\lim_{x \to 0} (\cos x)^{\frac{1}{x^2}} = e^{-\frac{1}{2}} = \frac{1}{\sqrt{e}}.
\]

\medskip

\textbf{9.} (计算题) 设 \(x_1 = 1\), \(\displaystyle x_{n+1} = \frac{1}{2}\left(x_n + \frac{2}{x_n}\right)\) (\(n = 1,2,\ldots\)), 证明数列 \(\{x_n\}\) 收敛, 并求其极限。

\boxed{\textbf{解}} 先证有下界: 由均值不等式,
\[
x_{n+1} = \frac{1}{2}\left(x_n + \frac{2}{x_n}\right) \geq \sqrt{x_n \cdot \frac{2}{x_n}} = \sqrt{2} \quad (n \geq 1),
\]
故当 \(n \geq 2\) 时, \(x_n \geq \sqrt{2}\)。

再证单调递减（\(n \geq 2\)）:
\[
x_{n+1} - x_n = \frac{1}{2}\left(x_n + \frac{2}{x_n}\right) - x_n
= \frac{2 - x_n^2}{2x_n} \leq 0 \quad (\because\ x_n^2 \geq 2).
\]

故 \(\{x_n\}_{n \geq 2}\) 单调递减且有下界 \(\sqrt{2}\), 由单调有界定理知数列收敛。

设 \(\displaystyle\lim_{n \to \infty} x_n = A\), 对递推式两边取极限:
\[
A = \frac{1}{2}\left(A + \frac{2}{A}\right)
\;\Longrightarrow\; 2A = A + \frac{2}{A}
\;\Longrightarrow\; A^2 = 2
\;\Longrightarrow\; A = \sqrt{2}\ (A > 0).
\]

故 \(\displaystyle\lim_{n \to \infty} x_n = \sqrt{2}\)。

\begin{conclusion}{单调有界定理的应用步骤}{monotone-bounded-steps}
    对于递推数列 \(x_{n+1} = f(x_n)\) 的极限问题, 一般步骤为:
    \begin{enumerate}
        \item 证明数列单调（递增或递减）;
        \item 证明数列有界（上界或下界）;
        \item 由单调有界定理断言极限存在, 设极限为 \(A\);
        \item 对递推式取极限, 解方程 \(A = f(A)\) 得极限值。
    \end{enumerate}
\end{conclusion}

\textbf{10.} (计算题) 求极限 \(\displaystyle\lim_{x \to 0} \frac{\tan x - \sin x}{x^3}\)。

\boxed{\textbf{解}} 

\textbf{解法一（等价无穷小与三角恒等变换）:}
\[
\tan x - \sin x = \tan x(1 - \cos x) = \tan x \cdot 2\sin^2\frac{x}{2}.
\]
当 \(x \to 0\) 时, \(\tan x \sim x\), \(\sin\frac{x}{2} \sim \frac{x}{2}\), 故
\[
\tan x - \sin x \sim x \cdot 2 \cdot \left(\frac{x}{2}\right)^2 = \frac{x^3}{2},
\]
因此 \(\displaystyle\lim_{x \to 0} \frac{\tan x - \sin x}{x^3} = \frac{1}{2}\)。

\textbf{解法二（洛必达法则）:} \(\frac{0}{0}\) 型,
\[
\lim_{x \to 0} \frac{\tan x - \sin x}{x^3}
= \lim_{x \to 0} \frac{\sec^2 x - \cos x}{3x^2}
= \lim_{x \to 0} \frac{1 - \cos^3 x}{3x^2 \cos^2 x}.
\]
当 \(x \to 0\) 时 \(\cos x \to 1\), 分子 \(1 - \cos^3 x = (1 - \cos x)(1 + \cos x + \cos^2 x) \sim \frac{x^2}{2} \cdot 3 = \frac{3x^2}{2}\), 故极限为 \(\frac{3/2}{3} = \frac{1}{2}\)。

\begin{formula}{$\tan x - \sin x$ 的等价无穷小}{tan-minus-sin-equiv}
    当 \(x \to 0\) 时, \(\displaystyle\tan x - \sin x \sim \frac{1}{2}x^3\)。这是一个常用结论, 建议记忆。
\end{formula}

\subsection{真题风格与综合训练}

\textbf{11.}（\boxed{\textbf{真题风格}}，计算题） \(\displaystyle\lim_{x \to 0} \left( \frac{1}{\sin^2 x} - \frac{1}{x^2} \right) = \underline{\hspace{2cm}}\)。

\boxed{\textbf{解}} \(\infty - \infty\) 型未定式, 通分:
\[
\frac{1}{\sin^2 x} - \frac{1}{x^2} = \frac{x^2 - \sin^2 x}{x^2 \sin^2 x}.
\]

当 \(x \to 0\) 时, 分母 \(\sin^2 x \sim x^2\), 故分母 \(\sim x^4\)。

分子 \(x^2 - \sin^2 x = (x - \sin x)(x + \sin x)\)。利用泰勒公式
\[
\sin x = x - \frac{x^3}{6} + o(x^4),
\]
得
\[
x - \sin x \sim \frac{x^3}{6}, \qquad x + \sin x \sim 2x,
\]
故分子 \(\displaystyle\sim \frac{x^3}{6} \cdot 2x = \frac{x^4}{3}\)。

因此
\[
\lim_{x \to 0} \left( \frac{1}{\sin^2 x} - \frac{1}{x^2} \right)
= \lim_{x \to 0} \frac{x^4/3}{x^4} = \frac{1}{3}.
\]

\medskip

\textbf{12.}（\boxed{\textbf{真题风格}}，计算题） 求极限 \(\displaystyle\lim_{n \to \infty} \left( n \tan \frac{1}{n} \right)^{n^2}\)。

\boxed{\textbf{解}} 这是 \(1^\infty\) 型未定式。

令 \(\displaystyle x = \frac{1}{n}\), 当 \(n \to \infty\) 时 \(x \to 0^+\), 原极限化为
\[
L = \lim_{x \to 0^+} \left( \frac{\tan x}{x} \right)^{\frac{1}{x^2}}.
\]

取对数:
\[
\ln L = \lim_{x \to 0^+} \frac{\ln(\tan x) - \ln x}{x^2}.
\]

利用泰勒展开: \(\displaystyle\tan x = x + \frac{x^3}{3} + o(x^3)\), 故
\[
\frac{\tan x}{x} = 1 + \frac{x^2}{3} + o(x^2).
\]
因此
\[
\ln\left( \frac{\tan x}{x} \right)
= \ln\left(1 + \frac{x^2}{3} + o(x^2)\right)
\sim \frac{x^2}{3} \quad (x \to 0).
\]

于是
\[
\ln L = \lim_{x \to 0^+} \frac{x^2/3}{x^2} = \frac{1}{3}
\;\Longrightarrow\; L = e^{\frac{1}{3}} = \sqrt[3]{e}.
\]

\begin{conclusion}{$1^\infty$ 型极限的常用处理方法}{one-inf-limit-method}
    对于 \(\lim f(x)^{g(x)}\) 且 \(\lim f(x) = 1\), \(\lim g(x) = \infty\), 常用方法:
    \[
    \lim f(x)^{g(x)} = \lim \bigl[1 + (f(x) - 1)\bigr]^{\frac{1}{f(x)-1} \cdot (f(x)-1)g(x)}
    = \exp\left\{\lim\, (f(x)-1)g(x)\right\}.
    \]
    或取对数化为 \(\lim g(x) \ln f(x)\) 后利用等价无穷小 \(\ln(1+u) \sim u\)（其中 \(u = f(x)-1 \to 0\)）。
\end{conclusion}
