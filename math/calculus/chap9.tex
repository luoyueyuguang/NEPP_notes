\section{多元函数积分学}
\subsection{多元函数积分的概念与性质}
\begin{mydef}{二重积分}{two-integral}
    平面上有界闭区域$D$上二元函数\(z = f(x, y)\)的二重积分为
    \[
        I = \iint\limits_{D} f(x, y) \, \mathrm{d} x \, \mathrm{d} y = \lim_{\mathrm{d} \to 0} \sum_{i=1}^{n} f(\xi_{i}, \eta_{i}) \Delta \sigma_{i}
    \]
    其中\(\mathrm{d} = \max\limits_{i} \{d_i\}\),\(\mathrm{d_i}\)为小区域\(\Delta \sigma_i\)的直径，\(\Delta \sigma_i\)为第\(i\)个小区域，也表示\(\Delta \sigma_i\)的
    面积。\\
    当二重积分\(
        \iint\limits_{D} f(x, y) \, \mathrm{d} \sigma
    \)存在时,则称\(f(x, y)\)在区域\(D\)上可积。\\
    若\(f(x, y)\)在区域\(D\)上连续，则\(f(x, y)\)在区域\(D\)上可积。即存在二重积分\(
        \iint\limits_{D} f(x, y) \, \mathrm{d} \sigma
        \).\\
    二重积分的几何意义为：当连续函数\(z =f(x, y) \geq 0\)时，二重积分\(I\)以区域\(D\)为底，曲面\(z = f(x, y)\)的
    侧面是以\(D\)的边界为准线，母线平行于\(z\)轴的柱面的曲顶柱体体积。\\
    \[
        \iint\limits_{D} f(x, y) \, \mathrm{d} \sigma = Oxy\text{
            上方的曲顶柱体体积
        } - Oxy\text{下方的曲顶柱体体积}
    \]
\end{mydef}

\begin{mydef}{三重积分}{three-integral}
    空间中有界闭区域\(\Omega\)上三元函数\(z = f(x, y, z)\)的三重积分为
    \[
        J = \iiint\limits_{\Omega} f(x, y, z) \, \mathrm{d} V = \lim_{\mathrm{d} \to 0} \sum_{i=1}^{n} f(\xi_{i}, \eta_{i}, \zeta_{i}) \Delta V_i
    \]
    其中\(\mathrm{d} = \max\limits_{i} \{d_i\}\),\(\mathrm{d_i}\)为小区域\(\Delta V_i\)的直径，\(\Delta V_i\)为第\(i\)个小区域的体积。\\
    当三重积分\(\iiint\limits_{\Omega} f(x, y, z) \, \mathrm{d} V\)存在时，则称\(f(x, y, z)\)在区域\(\Omega\)上可积。\\
    若\(f(x, y, z)\)在区域\(\Omega\)上连续，则\(f(x, y, z)\)在区域\(\Omega\)上可积。即三重积分\(\iiint\limits_{\Omega} f(x, y,z) \, \mathrm{d} V\)一定存在.\\
\end{mydef}

\begin{mydef}{第一类曲线积分(对弧长的曲线积分)}{first-curve-integral}
    \(Oxy\)平面上分段光滑曲线\(L\)上二元函数\(f(x, y)\)的第一类曲线积分为
    \[
        \int_{L} f(x, y) \, \mathrm{d} s = \lim_{\lambda \to 0} \sum_{i=1}^{n} f(\xi_{i}, \eta_{i}) \Delta s_i
    \]
    其中\(\lambda = \max\limits_{i} \{\Delta s_i\}\),\(\Delta s_i\)为第\(i\)段弧长。\\
    第一类曲线积分也被称为对弧长的曲线积分。\(f(x, y)\)称作被积函数，\(L\)称作积分弧段。\\
    若\(f(x, y)\)在分段光滑曲线\(L\)上连续，则\(f(x, y)\)在分段光滑曲线\(L\)上可积。即存在第一类曲线积分\(\int_{L} f(x, y) \, \mathrm{d} s = 
    \lim\limits_{\lambda \to 0} \sum_{i=1}^{n} f(\xi_{i}, \eta_{i}) \Delta s_i\).\\
    如果积分弧段\(L\)是封闭曲线，则相应的对弧长的积分记作\(\oint_{L} f(x, y) \, \mathrm{d} s\)或\(
        \oint_{L} f(x, y, z) \, \mathrm{d} s\
    \).又称环路积分。\\
\end{mydef}

\begin{mydef}{第二类曲线积分(对坐标的曲线积分)}{second-curve-integral}
    \(F = \{
        P(x, y), Q(x, y)
    \}\)在定向分段光滑曲线上的第二类曲线积分
    \[
        \int_{L} P(x, y) \, \mathrm{d} x + Q(x, y) \, \mathrm{d} y
        = \lim_{\lambda \to 0} \sum_{i=1}^{n} \left[P(\xi_{i}, \eta_{i}) \Delta x_i + Q(\xi_{i}, \eta_{i}) \Delta y_i\right]
    \]
    其中\(\lambda\)仍是各小弧段长度的最大值，\(\Delta x_i = x_i - x_{i-1}, \Delta y_i = y_i - y_{i-1}\).\\
    \(x_i, y_i\)分点\(M_i\)的坐标，第二类曲线积分也称作对坐标的曲线积分。记为
    \(
        \int_{L} \mathrm{F} d\mathrm{s}
    \)
    若\(P(x, y)\)和\(Q(x, y)\)在分段光滑曲线\(L\)上连续，则曲线积分\(
        \int_{L} P(x, y) \, \mathrm{d} x + Q(x, y) \, \mathrm{d} y
    \)存在 \\
    将第二类曲线积分推广到空间曲面，则有
    \[
        \begin{aligned}
            \int_{L} \mathbf{F} \cdot \, \mathrm{d}\mathbf{s} &= 
        \int{L} P(x, y, z) \, \mathrm{d} x + Q(x, y, z) \, \mathrm{d} y + R(x, y, z) \, \mathrm{d} z \\
        &= \lim_{\lambda \to 0} \sum_{i=1}^{n} \left[P(\xi_{i}, \eta_{i}) \Delta x_i + Q(\xi_{i}, \eta_{i}) \Delta y_i + R(\xi, \eta_{i}) \Delta z_i\right]
        \end{aligned}
    \]

\end{mydef}
\begin{mydef}{第一类曲面积分(对面积的曲面积分)}{first-surface-integral}
    \(f(x, y, z)\)在分块光滑曲面上\(\Sigma\)上的第一类曲面积分为
    \[
        \iint\limits_{\Sigma} f(x, y, z) \, \mathrm{d} S = \lim_{\lambda \to 0} \sum_{i=1}^{n} f(\xi_{i}, \eta_{i}, \zeta_{i}) \
        \Delta S_i
    \]
    其中\(\lambda\)是各小块曲面的直径的最大值，\(\Delta S_i\)为第\(i\)个小块曲面，同时也代表\(\Delta S_i\)的面积。\\
    \(
    f(x,y,z)\)是被积函数，\(\Sigma\)称作积分曲面。\\
    第一类曲面积分也被称为对面积的曲面积分。\\
\end{mydef}
\begin{mydef}{第二类曲面积分(对坐标的曲面积分)}{second-surface-integral}
    \(F = \{P(x, y, z), Q(x, y, z), R(x, y, z)\}\)在定向分块光滑曲面上\(\Sigma\)上的第二类曲面积分为
    \[
        \begin{aligned}
        \iint\limits_{\Sigma} \mathbf{F} \cdot \, \mathrm{d} \mathbf{S} &= \iint\limits_{\Sigma} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y \\
        &= \iint\limits_{\Sigma} \left[P(x, y, z) \, \mathrm{d} y \mathrm{d} z + Q(x, y, z) \, \mathrm{d} z \mathrm{d} x + R(x, y, z) \, \mathrm{d} x \mathrm{d} y\right] \\
        &
        \text{\small\(= \lim_{\lambda \to 0} \sum_{i=1}^{n} \left[P(\xi_{i}, \eta_{i}, \zeta_{i}) \, (\Delta S_i)_{yz} + Q(\xi_{i}, \eta_{i}
        , \zeta_{i}) \, \Delta (S_i)_{zx} + 
        R(\xi_{i}, \eta_{i}, \zeta_{i}) \, \Delta (S_i)_{xy}\right]\)}
        \end{aligned}
    \]
    其中\((\Delta S_i)_{yz}\)表示第\(i\)个小块曲面在\(y-z\)面上的投影面积，同理可得其余两项。\\
    \(\lambda\)仍是各小块曲面的直径的最大值。\\
    \(P(x, y, z),Q(x, y, z)\)和\(R(x, y, z)\)称作被积函数，\(\Sigma\)称作积分曲面。
    第二类曲面积分也被称为对坐标的曲面积分。\\
    若被积函数在定向分块光滑曲面上连续，则曲面积分\(\iint\limits_{\Sigma} \mathbf{F} \cdot \, \mathrm{d}\mathbf{S}\)存在。\\
\end{mydef}

\begin{conclusion}{关于曲面积分方向的注意事项}{integral-direction}
    \begin{enumerate}
        \item 对于双侧曲面，称有法向量的曲面为有向曲面，根据法向量的方向来说明选择的是哪侧。\\
        对封闭曲面，法向量朝外的称为外侧，法向量朝内的称为内侧。\\
        对曲面\(z = z(x, y)\),与法向量成锐角的曲面称为内侧，与法向量成钝角的曲面称为外侧。\\
        对曲面\(y = y(x, z)\),曲面\(x = x(y, z)\),同理可得。
        \item 曲面\(\Omega\)的法向量\(\mathbf{n}(\mathbf{M}_i) = ({\cos \alpha}_i, {\cos \beta}_i, {\cos \gamma}_i)\)，若\({\cos \gamma}_i > 0\)，则\((\Delta S_i)_{xy}\)为正值，否则为负值。
    \end{enumerate}
\end{conclusion}
\begin{conclusion}{几种多元函数积分的比较}{integral-compare}
    每种积分都是对积分区域进行任意剖分后的相应和式的极限，不管哪种积分和式每一项都有两个乘积因子，其中一个均是被积函数在小区域某点取值，都是用区域的最大值趋于零来无限划分区域，从而得到极限。\\
    不同点在于不同类型的积分区域是不一样的
\end{conclusion}
\begin{conclusion}{多元函数积分的性质}{integral-property}
    设\(f(x, y),g(x, y)\)在有界区域\(D\)上可积，便有如下性质
    \begin{enumerate}
        \item 线性运算性质：\\
        \(\alpha, \beta\)为常数，则
        \[
            \iint\limits_{D} (\alpha f(x, y) \pm \beta g(x, y)) \, \mathrm{d} x \mathrm{d} y = 
            \alpha \iint\limits_{D} f(x, y) \, \mathrm{d} x \mathrm{d} y \pm
            \beta \iint\limits_{D} g(x, y) \, \mathrm{d} x \mathrm{d} y
        \]
        \item 积分区域可加性质 \\
        \[
            \iint\limits_{D_1 + D_2} f(x, y) \, \mathrm{d} x \mathrm{d} y = 
            \iint\limits_{D_1} f(x, y) \, \mathrm{d} x \mathrm{d} y + 
            \iint\limits_{D_2} f(x, y) \, \mathrm{d} x \mathrm{d} y
        \]
        其中\(D\)为两个不相交的区域\(D_1\)和\(D_2\)的并集。\\
        \item 比较定理 \\
        若\(f(x, y), g(x, y)\)在\(D\)上连续，且对任意点\(f(x, y) \substack{\leq\\ \neq} g(x, y)\)，则
        \[
            \iint\limits_{D} f(x, y) \, \mathrm{d} x \mathrm{d} y <
            \iint\limits_{D} g(x, y) \, \mathrm{d} x \mathrm{d} y
        \]
        特别地，若\(m \substack{\leq\\ \neq} f(x, y) \substack{\leq\\ \neq} M\)在\(D\)上恒成立，则
        \[
            m \, A \substack{\leq\\ \neq} 
            \iint\limits_{D} f(x, y) \, \mathrm{d} x \mathrm{d} y \substack{\leq\\ \neq} 
            M \, A
        \]
        其中\(A\)表示区域\(D\)的面积。\\
        又有
        \[
            \left| \iint\limits_{D} f(x, y) \, \mathrm{d} \sigma \right| \leq
            \iint\limits_{D} |f(x, y)| \mathrm{d} \sigma 
        \]
        \item 积分中值定理 \\
        若\(f(x, y)\)在有界区域\(D\)上连续，则存在一点\((\xi, \eta)\)，使得
        \[
            \iint\limits_{D} f(x, y) \, \mathrm{d} x \mathrm{d} y = f(\xi, \eta) \, A
        \]
        \item 连续非负函数的积分性质 \\
        若\(f(x, y)\)在有界区域\(D\)上连续非负，若\(D_0 \subset D\),则
        \[
            \iint\limits_{D_0} f(x, y) \, \mathrm{d} x \mathrm{d} y
            \leq
            \iint\limits_{D} f(x, y) \, \mathrm{d} x \mathrm{d} y
        \]
        设\(f(x,y)\)在\(D\)上连续且非负，且\(
            \iint\limits_{D} f(x, y) \, \mathrm{d} x \mathrm{d} y = 0
        \),则\(
            f(x, y) \equiv 0((x, y)\in D)
        \)
        \item 设\(f(x, y)\)在\(D\)上连续，若在\(D\)内任意子区域\(D_0\),有
        \(
            \iint\limits_{D_0} f(x, y) \, \mathrm{d} x \mathrm{d} y = 0
        \),则\(f(x, y) \equiv 0((x, y)\in D_)\)。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{两类曲线积分的关系}{curve-integral-compare}
    设\(L_{\overset{\frown}{AB}}\)为分段光滑曲线，则
    \[
        \int_{L_{\overset{\frown}{AB}}} P \mathrm{d} x + Q \mathrm{d} y = 
        \int_{L_{\overset{\frown}{AB}}} (P \cos\alpha + Q \cos\beta) \, \mathrm{d} s
    \]
    其中\(\cos \alpha, \cos\beta\)分别是曲线\(
        L_{\overset{\frown}{AB}}
    \)沿从\(A\)到\(B\)的切线的方向的余弦。\\
    \(P, Q\)是\(L_{\overset{\frown}{AB}}\)连续导数，推广到空间情形，可得
    \[
        \int_{L_{\overset{\frown}{AB}}} P \mathrm{d} x + Q \mathrm{d} y + R \mathrm{d} z = 
        \int_{L_{\overset{\frown}{AB}}} (P \cos\alpha + Q \cos\beta + R \cos\gamma) \, \mathrm{d} s
    \]
    写为点积形式，可得函数
    \[
        \int_{L_{\overset{\frown}{AB}}} \mathbf{F} \, \mathrm{d} s = 
        \int_{L_{\overset{\frown}{AB}}} \mathbf{F} \, \mathbf{\tau} \, \mathrm{d} s
    \]
    其中\(
    \mathbf{F} = (P, Q, R) d\mathbf{s} = \{
        dx, dy, dz
    \}, \mathbf{\tau} = \{
        \cos \alpha, \cos \beta, \cos \gamma
    \} 
    \)是单位上任一点的切向量。\\
\end{conclusion}

\begin{conclusion}{两类曲面积分的关系}{surface-integral-compare}
    设\(\Sigma\)为分段光滑曲面，则
    \[
        \iint\limits_{\Sigma} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y = 
        \iint\limits_{\Sigma} (P \cos \alpha + Q \cos \beta + R \cos \gamma) \, \mathrm{d} S
    \]
    点积形式表示为
    \[
        \iint\limits_{\Sigma} \mathbf{F} \, \mathrm{d} \mathbf{S} = 
        \iint\limits_{\Sigma} \mathbf{F} \, \mathbf{n} \, \mathrm{d} \mathbf{S}
    \]
    其中, \(\mathbf{F} = (P, Q, R), \mathrm{d}\mathbf{S} =
    \{
        dydz, dzdx, dx dy
    \}
    , \mathbf{n} = \{
        \cos \alpha, \cos \beta, \cos \gamma
    \}
    \)是曲面上的单位法向量。\\
\end{conclusion}
\begin{conclusion}{第一二类曲线(面)积分比较}{integral-compare}
    \begin{enumerate}
        \item 第一二类曲线积分的一个重要不同点是，第一类曲线积分与曲线的定向无关，而第二类曲线积分与曲线的定向有关。\\
        若\(L^{+}\)与\(L^{-}\)是同一曲线，但方向相反，则
        \[
            \int_{L^{+}} P \mathrm{d} x + Q \mathrm{d} y = 
            - \int_{L^{-}} P \mathrm{d} x + Q \mathrm{d} y
        \]
        \item 第一二类曲面积分的一个重要不同点是，第一类曲面积分与曲面的定向无关，而第二类曲面积分与曲线的定向有关。\\
        若\(\Sigma^{+}\)与\(\Sigma^{-}\)是同一曲面，但方向相反，则
        \[
            \iint\limits_{\Sigma^{+}} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y = 
            - \iint\limits_{\Sigma^{-}} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y 
        \]
    \end{enumerate}
\end{conclusion}
\begin{conclusion}{一些基本结论}{integral-conclusion}
    \begin{enumerate}
        \item 设\(L\)是空间中分段光滑曲线，则
        \begin{enumerate}
            \item \(\int_{L} P \mathrm{d} x = 0\)\quad(
                若\(L\)在垂直于\(x\)轴的平面上
            )
            \item \(\int_{L} Q \mathrm{d} y = 0\)\quad(
                若\(L\)在垂直于\(y\)轴的平面上
            )
            \item \(\int_{L} R \mathrm{d} z = 0\)\quad(
                若\(L\)在垂直于\(z\)轴的平面上
            )
        \end{enumerate}
        \item 设\(S\)是空间中分块光滑曲面，则
        \begin{enumerate}
            \item \(\iint\limits_{S} P \mathrm{d} y \mathrm{d} z = 0\)\quad(
                若\(S\)在垂直于\(Oyz\)平面，即\(S\)的法向量于\(Oyz\)平面平行
            )
            \item \(\iint\limits_{S} Q \mathrm{d} z \mathrm{d} x = 0\)\quad(
                若\(S\)在垂直于\(Oxz\)平面，即\(S\)的法向量于\(Oxz\)平面平行
            )
            \item \(\iint\limits_{S} R \mathrm{d} x \mathrm{d} y = 0\)\quad(
                若\(S\)在垂直于\(Oxy\)平面，即\(S\)的法向量于\(Oxy\)平面平行
            )
        \end{enumerate}
    \end{enumerate}
\end{conclusion}
\subsection{多元函数积分变换为定积分}
\begin{conclusion}{第一类曲线积分化为定积分}{curve-integral-to-definite}
    设\(f(x, y)\)在空间\(L\)上连续.
    若曲线\(L\)的参数方程\(
    \begin{cases}
        x = \phi(t)
        y = \psi(t)
    \end{cases}, \alpha < t \beta\)
    则\[
        \int_{L} f(x, y) \mathrm{d} s = \int_{\alpha}^{\beta} f(\phi(t), \psi(t)) \, \sqrt{(\phi'(t))^2 + (\psi'(t))^2} \, \mathrm{d} t
    \]
    其中\( \phi(t), \psi(t)\)在\(
    [\alpha, \beta]
    \)有连续的导数且\(
        \varphi^{'}(t)^2 + \psi^{'}(t)^2 \neq 0
    \)
    特别地，若曲线\(L\)能表成\(y = y(x), a \leq x \leq b\)的形式，则
    \[
        \int_{L} f(x, y) \mathrm{d} s = \int_{a}^{b} f(x, y(x)) \sqrt{1 + {y'_{x}}^2} \mathrm{d} x
    \]
    其中\(y(x)\)在\([a, b]\)有连续的导数 \\
    若平面光滑曲线\(L\)由极坐标方程给出\(r = r(\theta)\)\(
        \alpha \leq \theta \leq \beta
    \),则
    \[
        \int_{L} f(x, y) \mathrm{d} s = \int_{\alpha}^{\beta} f(r \cos \theta, r \sin \theta) \sqrt{
            r^2(\theta) + r'^2(\theta)
        } \mathrm{d} \theta
    \]
\end{conclusion}

\begin{conclusion}{第二类曲线积分化为定积分}
    设定向曲线\(L\)的参数方程为\(\begin{cases}
        x = \varphi(t) \\
        y = \psi(t)
    \end{cases}, t \in [\alpha, \beta]\)，\(\alpha\)对应于起点\(A\),\(\beta\)是对应于
    终点\(B(\beta\text{不一定大于}\alpha)\)，则
    \[
        \int_{L_{\overset{\frown}{AB}}} P(x, y) \mathrm{d} x + Q(x, y) \mathrm{d} y 
        = \int_{\alpha}^{\beta} [
            P(\varphi(t), \psi(t))\varphi^{'}(t)  + 
            Q(\varphi(t), \psi(t))\psi'(t)] \mathrm{d} t
    \]
    其中\(
        \varphi(t), \psi(t)
    \)在\(
    [a, b]
    \)或\(
    [\alpha, \beta]
    \)有连续的导数。
\end{conclusion}

\begin{conclusion}{对于曲线积分化为定积分的观察}{observation about curve}
    从上述定理看到,曲线积分化为定积分是直接的，将曲线的参数方程代入被积表达式，归结为微分运算
    \[
        \mathrm{d} s = \sqrt{
            x'^2(t) + y'^2(t)
        } \mathrm{d} t,
        \mathrm{d} x = x'(t)\mathrm{d} t,
        \mathrm{d} y = y'(t) \mathrm{d} t.
    \]\\积分限代入
    曲线端点的参数值,对于第一类曲线积分来说下限小于上限；对于第二类曲线积分来说，下限
    分别是起终点的参数(于曲线的定向有关) \\
    对空间曲线积分\(
        \int_{L} f(x, y, z) \mathrm{d} s, \int_{L} P \mathrm{d} x +
        Q \mathrm{d}  y + R \mathrm{d} z
    \)有类似的计算公式，若未给出曲线的参数方程，需求出曲线的参数方程，才可用上述公式
\end{conclusion}

\begin{conclusion}{二重积分化为累次积分}{two-integral-to-definite}
    设\(f(x, y)\)在有界区域\(D\)上连续，
    若\(D = \{
        (x, y) | a \leq x \leq b,
        \varphi_1(x) \leq y \leq \varphi_2(x)
        \}
    \)其中\(
        \varphi_1(x),
        \varphi_2(x)
    \)在\(
    [a, b]
    \)上连续，则
    \[
        \iint_{D} f(x, y) \mathrm{d} \sigma = 
        \int_{a}^{b} \mathrm{d} x \int_{\varphi_1(x)}^{\varphi_2(x)} f(x, y) \mathrm{d} y
    \]
    若\(D = \{
        (x, y) | \alpha \leq  y \leq \beta,
        \psi_1(y) \leq x \leq \psi_2(y)
    \}\)
    其中\(
        \psi_1(y),
        \psi_2(y)
    \)在\(
    [\alpha, \beta]
    \)上连续，则
    \[
        \iint_{D} f(x, y) \mathrm{d} \sigma = 
        \int_{\alpha}^{\beta} \mathrm{d} y \int_{\psi_1(y)}^{\psi_2(y)} f(x, y) \mathrm{d} x
    \]
    计算二重积分的关键是配置积分限，为此要画好积分区域\(D\)的图形或写出\(D\)的不等式表示。将
    二重积分化为累次积分即二次定积分。
    \begin{conclusion}{特殊情形}{special-case}
        设\(f(x, y)\)在矩形区域\(
            D = \{
                (x, y) | a \leq x \leq b,
                \alpha \leq y \leq \beta
            \}
        \)连续，则
        \begin{enumerate}
            \item \(\iint_{D} f(x, y) \mathrm{d} x \mathrm{d} y = 
            \int_{a}^{b} (
                \int_{\alpha}^{\beta} f(x, y) \mathrm{d} y
            ) \mathrm{d} x
            = \int_{\alpha}^{\beta} (
                \int_{a}^{b} f(x, y) \mathrm{d} x
            ) \mathrm{d} y
        \)
            \item \text{若}\(
                f(x, y) = h(x)g(y)
            \),从而\[
                \iint_{D} h(x)g(y) \mathrm{d} \sigma = 
                \int_{a}^{b} h(x) \mathrm{d} x
                \int_{\alpha}^{\beta} g(y) \mathrm{d} y
            \]
        \end{enumerate}
    \end{conclusion}
\end{conclusion}

\begin{conclusion}{三重积分化为累次积分}{three-integral-to-definite}
    设\(\Omega\)是\(Oxyz\)空间中的有界闭区域，\(f(x, y, z)\)在\(\Omega\)上连续，若
    \(\Omega = \{
        (x, y, z) | z_1(x, y) \leq z \leq \zeta_2(x, y), (x,y) \in D
    \}\)连续，则
    \[
        \iiint_{D} f(x, y, z) \mathrm{d} V = 
        \iint_{D} \mathrm{d} x \mathrm{d} y
        \int_{z_1(x, y)}^{\zeta_2(x, y)} f(x, y, z) \mathrm{d} z
    \]
    若\(\Omega = \{(x, y, z) | \psi_1(x, y) \leq z \leq \psi_2(x, y), (x,y) \in D(z)\}\)，
    其中\(D(z)\)为竖坐标为\(z\)的平面上的有界闭函数，
    则\[
        \iiint_{D} f(x, y, z) \mathrm{d} V =
        \int_{\alpha}^{\beta} \mathrm{d} z
        \iint_{D(z)} f(x, y, z) \mathrm{d} x \mathrm{d} y
    \]
    两种方法即先一后二法或先二后一法。针对的分别是曲顶曲底柱长条区域和界面已知的区域。
\end{conclusion}

\begin{conclusion}{第一类曲面积分化为二重积分}{first-curve-integral-to-two}
    如果积分曲面\(Omega\)能够表为\(z = z(x, y)\)的形式，并且在\(xOy\)平面
    上的投影域为\(D\),那么
    \[
        \iint_{\Omega} f(x, y, z) \mathrm{d} S = 
        \iint_{D} f(x, y, z(x, y)) \sqrt{
            1 + {z'_{x}}^2 + {z'_{y}}^2
        } \mathrm{d} \sigma
    \]
    其中\(f(x,y,z)\)在\(\Omega\)连续，\(z = (x, y)\)在\(D\)\\
    对面积的曲面积分化成了二重积分，而二重积分又可化为累次积分,所以曲面积分化为
    定积分。计算时需要注意三点\begin{enumerate}
        \item 被积函数要代入曲面方程(被积函数定义在曲面\(\Omega\)上)
        \item 积分元素\(
            \mathrm{d} S = \sqrt{
                1 + {z'_{x}}^2 + {z'_{y}}^2
            } \mathrm{d} \sigma
        \)
        \item 积分区域\(D\)是\(Omega\)在\(xOy\)面上的投影
    \end{enumerate}
    转化需要注意以下几点
    \begin{enumerate}
        \item 曲面元\(\mathrm{d} S\)在\((x, y ,z)\)处的单位法向量\(
            \mathbf{n} = (\cos \alpha, \cos \beta, \cos \gamma)
            = \pm \dfrac{1}{\sqrt{
                1 + {z'_{x}}^2 + {z'_{y}}^2
            }}(
                -z'_{x}, -z'_{y}, 1
            )
        \)
        \(\mathbf{dS}\)与它在\(Oxy\)平面上的投影的关系为
        \[
            \mathrm{d} S = \frac{1}{|\cos \mathbf{\gamma}|} \mathrm{d} \gamma
            = \sqrt{
                1 + {z'_{x}}^2 + {z'_{y}}^2
            } \mathrm{d} \sigma
        \]
        \item 若曲面\(\Omega\)的方程为\(
            x = x(y, z), (y, z) \in D_{yz}
        \)或\(
            y = y(z, x), (z, x) \in D_{zx}
        \),有类似的化第一类曲面积分为二重积分的方法。
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{第二类曲面积分化为二重积分}{second-curve-integral-to-two}
    \begin{enumerate}
        \item 若定向曲面\(\Sigma: x = x(y, z), (y, z) \in D_{yz}, x(y, z)\)在
        \(D_{yz}\)上有连续的偏导数，\(P(x,y,z)\)在\(\Sigma\)上连续，则
        \[
            \iint_{\Sigma} P \mathrm{d} y \mathrm{d} z = 
            \pm \iint_{D_{yz}} P(x(y, z), y, z) \mathrm{d} y \mathrm{d} z
        \]
        若\(\Sigma\)的法向量\(\mathbf{n}\)与\(x\)轴正向的夹角为\(\alpha\)，则
        \(\alpha\)为锐角时取前侧，即积分取\(+\)号，反之取负号。
        \item 若定向曲面\(\Sigma: y = y(z, x), (z, x) \in D_{zx}, y(z, x)\)在
        \(D_{zx}\)上有连续的偏导数，\(Q(x,y,z)\)在\(\Sigma\)上连续，则
        \[
            \iint_{\Sigma} Q \mathrm{d} z \mathrm{d} x = 
            \pm \iint_{D_{zx}} Q(x, y(z, x), z) \mathrm{d} z \mathrm{d} x
        \]
        若\(\Sigma\)的法向量\(\mathbf{n}\)与\(y\)轴正向的夹角为\(\beta\)，则
        \(\beta\)为锐角时取前侧，即积分取\(+\)号，反之取负号。
        \item 若定向曲面\(\Sigma: z = z(x, y), (x, y) \in D_{xy}, z(x, y)\)在   
        \(D_{xy}\)上有连续的偏导数，\(R(x,y,z)\)在\(\Sigma\)上连续，则
        \[
            \iint_{\Sigma} R \mathrm{d} x \mathrm{d} y = 
            \pm \iint_{D_{xy}} R(x, y, z(x, y)) \mathrm{d} x \mathrm{d} y
        \]
        若\(\Sigma\)的法向量\(\mathbf{n}\)与\(z\)轴正向的夹角为\(\gamma\)，则
        \(\gamma\)为锐角时取前侧，即积分取\(+\)号，反之取负号。
        \item 若定向曲面\(\Sigma\)由方程\(z = z(x, y)\)给出，\(\Sigma\)在\(Oxy\)平面上的
        投影区域为\(D_{xy}\)。\(z(x, y)\)在\(D_{xy}\)上有连续的偏导数，\(P(x, y, z), Q(x, y, z),
        R(x,y,z)\)在\(\Sigma\)上连续，则
        \[
            \begin{aligned}
            &\iiint_{\Sigma} P \mathrm{d} y \mathrm{d} z = \\ 
            &\pm \iint_{D_{xy}} [P(x, y, z(x, y))  (-\frac{\partial z}{\partial x}) \\
            &+ Q(x, y, z(x, y)) (-\frac{\partial z}{\partial y}) \\
            &+ R(x, y, z(x, y))] \mathrm{d} x \mathrm{d} y
            \end{aligned}
        \]
        其中正负号由\(\Sigma\)的定向决定。
        法向量指向上侧时取正号，否则取负。
    \end{enumerate}
    对于第二类曲面积分到二重积分的转化，需要注意以下几点
    \begin{enumerate}
        \item 前三个转化是将\(\Sigma\)分别投影到\(xOz\)，\(yOx\)和\(xOy\)平面，然后分别
        计算\(
        \iint_{D_{xy}} P(x, y, z) \mathrm{d} x \mathrm{d} y
        \),\(
        \iint_{D_{yz}} Q(x, y, z) \mathrm{d} y \mathrm{d} z
        \),\(
        \iint_{D_{zx}} R(x, y, z) \mathrm{d} z \mathrm{d} x
        \)
        最后一项是将曲面\(\Sigma\)投影到\(xOy\)平面，然后计算
        \[
            \iint_{\Sigma} P \mathrm{d} y \mathrm{d} z +
            Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y
        \]
        投影到其他平面，类似可得
        \item 第一类曲面积分与第二类曲面积分的关系
        \[
            \iint_{\Sigma} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y =
            \iint_{\Sigma} (
                P \cos \alpha + Q \cos \beta + R \cos \gamma
            ) \mathrm{d} S
        \]
        曲面\(\Sigma:z = z(x, y)\)的单位法向量公式
        \[
            \mathbf{n} =  (\cos \alpha, \cos \beta, \cos \gamma) = \pm \frac{
                1
            }{
                \sqrt{
                    1 + {z'_{x}}^2 + {z'_{y}}^2
                }
            }(-z'_{x}, -z'_{y}, 1)
            \]
    \end{enumerate}
    第二类曲面积分化为二重积分的过程，即为
    \[
        \text{第二类曲面积分} \Longrightarrow \text{第一类曲面积分} \Longrightarrow
        \text{二重积分}
    \]
\end{conclusion}
\subsection{重积分的变量替换}
\begin{conclusion}{平移变换}{translation}
    设\(D\)为平面有界闭区域
    \[
        \iint_{D} f(x, y) \mathrm{d} x \mathrm{d} y \overset{
            \substack{u = x - a\\
            v = y - b}
        }{\xlongequal{\quad\quad}} \iint_{D'} f(u + a, v + b)\mathrm{d} u \mathrm{d} v
    \]
    其中\(D'\)是在平移变换\(
        \begin{cases}
            u = x - a \\ 
            v = y - b
        \end{cases}
    \)(即\(
        \begin{cases}
            x = u + a \\
            y = v + b
        \end{cases}
    \))下把\(D\)平移到\(D'\)。\\
    设\(\Omega\)为空间中的有界闭区域,\(f(x, y, z)\)在\(\Omega\)上连续，则
    \[
        \iiint_{\Omega} f(x, y, z) \mathrm{d} x \mathrm{d} y \mathrm{d} z \overset{
            \substack{u = x - a\\
            v = y - b \\
            w = z - c}
        }{\xlongequal{\quad\quad}} \iiint_{\Omega'} f(u + a, v + b, w + c) \mathrm{d} u \mathrm{d} v \mathrm{d} w
    \]
    其中\(\Omega'\)是在平移变换\(
        \begin{cases}
            u = x - a \\ 
            v = y - b \\
            w = z - c
        \end{cases}
    \)(即\(
        \begin{cases}
            x = u + a \\
            y = v + b \\
            z = w + c
        \end{cases}
    \))下把\(\Omega\)平移到\(\Omega'\)。\\
    平移变换下保持区域的形状与体积(或面积)不变。
    如\(
        (x - 1)^2 + (y - 1)^2  + (z - 1)^2 \leq 1
    \)可变为\(
        u^2 + v^2 + w^2 \leq 1
        \)便于计算
\end{conclusion}

\begin{conclusion}{二重积分换元法}{two-integral-change}
    设\(D\)为平面有界闭区域，\(f(x, y)\)在\(D\)上连续，则
    \[
        \iint_{D} f(x, y) \mathrm{d} x \mathrm{d} y = \iint_{D'} f(u(x, y), v(x, y)) 
        \left|
        \frac{\partial (u, v)}{\partial (x, y)} \right| \mathrm{d} u \mathrm{d} v
        \]。
        其中\(D'\)是由\(u(x, y)\),\(v(x, y)\)定义的平面有界闭区域，且\(\left|
            \frac{\partial (u, v)}{\partial (x, y)} \right|
            \)为雅可比行列式。\\    
\end{conclusion}
\begin{conclusion}{二重积分的极坐标变换}{two-integral-polar}
    极坐标变换\(
        x = r \cos \theta \\
        y = r \sin \theta
    \)下有\(
        \mathrm{d} \sigma  = r \mathrm{d} r \mathrm{d} \theta
    \)
    \[
        \iint_{D} f(x, y) \mathrm{d} \sigma
        = \iint_{D'} f(r \cos \theta, r \sin \theta) r \mathrm{d} r \mathrm{d} \theta
    \]
    其中\(D\)为直角坐标系\(Oxy\)平面下的有界闭区域，\(D'\)为直角坐标系\(
        Or\theta
    \)平面下的有界闭区域。\\
    在极坐标变换下化二重积分\(
        \iint_{D} f(x, y) \mathrm{d} \sigma
    \)为累次积分的关键为在\(Oxy\)平面上画出\(D\)的图形并写出相对应\(D\)的极坐标的表示式。\\
    常见的情形有
    \begin{enumerate}
        \item \(
            D = \left\{
                (r, \theta) |
                \alpha \leq \theta \leq \beta,
                r_1(\theta) \leq r \leq r_2(\theta)
            \right\}\),则
            \[
            \iint\limits_{D} f(x, y) \mathrm{d} \sigma = \int_{\alpha}^{\beta} \mathrm{d} \theta \int_{r_1(\theta)}^{r_2(\theta)} f(r \cos \theta, r \sin \theta) r \mathrm{d} r 
            \]
        \item \(
            D = \left\{
                (r, \theta) |
                0 \leq \theta \leq 2\pi,
                r_1(\theta) \leq r \leq r_2(\theta)
            \right\}\),则
            \[
            \iint\limits_{D} f(x, y) \mathrm{d} \sigma = \int_{0}^{2\pi} \mathrm{d} \theta \int_{r_1(\theta)}^{r_2(\theta)} f(r \cos \theta, r \sin \theta) r \mathrm{d} r 
            \]特殊情形式\(
            r_1(\theta) = 0\)
        \item\(
            D = \left\{
                (r, \theta) |
                \alpha \leq \theta \leq \beta,
                0 \leq r \leq r(\theta)
            \right\}\),则
            \[
            \iint\limits_{D} f(x, y) \mathrm{d} \sigma = \int_{\alpha}^{\beta} \mathrm{d} \theta \int_{0}^{r(\theta)} f(r \cos \theta, r \sin \theta) r \mathrm{d} r 
        \]
        \item \(
            D = \left\{
                (r, \theta) |
                a \leq r \leq b, \theta_1(r) \leq \theta \leq \theta_2(r)
            \right\}\),则
            \[
            \iint\limits_{D} f(x, y) \mathrm{d} \sigma = \int_{a}^{b} \mathrm{d} r \int_{\theta_1(r)}^{\theta_2(r)} \mathrm{d} \theta f(r \cos \theta, r \sin \theta) r \mathrm{d} \theta
        \]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{三重积分的柱坐标变换}{three-integral-polar}
    柱坐标变换\(x = r \cos \theta, y = r \sin \theta, z = z\)下有\(
        \mathrm{d} V = r \mathrm{d} r \mathrm{d} \theta \mathrm{d} z
    \)
    \[
        \iiint_{\Omega} f(x, y, z) \mathrm{d} V = \iint_{\Omega'} f(r \cos \theta, r \sin \theta, z) r \mathrm{d} r \mathrm{d} \theta \mathrm{d} z
    \]
    其中\(\Omega\)为直角坐标系\(Oxyz\)下的有界闭区域，\(f(x, y, z)\)在\(\Omega\)上连续，\(\Omega'\)是直角坐标系\(
    Or\theta z\)下的有界闭区域。\\
    在柱坐标变换下\(\Omega\)和\(\Omega'\)是一一对应的。\\
    \(\Omega\)的上下曲顶方程\(
    z = z_1(r, \theta), z = z_2(r, \theta)\)下有
    \[
        \iiint_{\Omega} f(x, y, z) \mathrm{d} V = \iint_{D'} r \mathrm{d} r \mathrm{d} \theta \int_{z_1(r, \theta)}^{z_2(r, \theta)} 
        f(r \cos \theta, r \sin \theta, z) \mathrm{d} z
    \]
    其中若\(D'\)是\(Or\theta\)中对应的\(D\)的区域，则
    \[
        \iiint_{\Omega} f(x, y, z) \mathrm{d} V = \int_{\alpha}^{\beta}  \mathrm{d} \theta \int_{r_1(\theta)}^{r_2(\theta)} r\mathrm{d} r \int_{z_1(r, \theta)}^{z_2(r, \theta)} f(r \cos \theta, r \sin \theta, z)  \mathrm{d} z 
    \]
    若\(Oxyz\)介于半平面\(
        \theta  = \alpha, \theta = \beta
    \)(
        \(0 \leq \alpha < \beta \leq 2\pi\)
    )之间且极角为\(
        \theta \in [
            \alpha, \beta
        ]\)的任一半平面与
    \(
        \Omega
    \)相截得平面区域\(D(\theta)\)，则
    \[
        \iiint_{\Omega} f(x, y, z) \mathrm{d} x \mathrm{d} y \mathrm{d} z = \int_{\alpha}^{\beta}  \mathrm{d} \theta \iint_{D(\theta)} f(r \cos \theta, r \sin \theta, z) r \mathrm{d} r \mathrm{d} z
    \]
\end{conclusion}

\begin{conclusion}{三重积分的球坐标变换}{three-integral-polar}
    在球坐标变换\(
    x = \rho \sin \varphi \cos \theta, y = \rho \sin \varphi \sin \theta, z = \rho \cos \varphi
    \)下有\(
        \mathrm{d} V = \rho^2 \sin \phi \mathrm{d} \rho \mathrm{d} \theta \mathrm{d} \phi
    \)
    \[
        \iiint\limits_{\Omega} f(x, y, z) \mathrm{d} V = \iiint\limits_{\Omega'} f(\rho \sin \varphi \cos \theta, \rho \sin \varphi \sin \theta, \rho \cos \varphi) \rho^2 \sin \varphi \mathrm{d} \rho \mathrm{d} \theta \mathrm{d} \varphi
    \]
    其中\(\Omega\)为直角坐标系\(Oxyz\)下的有界闭区域，\(f(x, y, z)\)在\(\Omega\)上连续，\(\Omega'\)是直角坐标系
    \(O\rho \theta \varphi\)相应的有界闭区域。\\
    在球坐标变换下\(\Omega\)和\(\Omega'\)是一一对应的。\\
    记\(
        F(\rho, \theta, \varphi) = f(\rho \sin \varphi \cos \theta, \rho \sin \varphi \sin \theta, \rho \cos \varphi)
    \)
    若\(O\)位于闭曲面\(S\)所围区域\(\Omega\)内，\(S\)的球坐标方程为\(
    \rho = \rho(\theta, \varphi)\)，则
    \[
        \iiint\limits_{\Omega} f(x, y, z) \mathrm{d} V = 
        \int_{0}^{2\pi} \mathrm{d} \theta \int_{0}^{\pi} \mathrm{d} \varphi \int_{0}^{\rho(\varphi, \theta)} F(\rho, \theta, \varphi) \rho^2 \sin \varphi \mathrm{d} \rho 
    \]
    若\(Omega\)由锥面\(
        \varphi = a
    \)及球坐标方程\(
        \rho = \rho(\theta, \varphi)
    \)的曲面所围，则
    \[
        \iiint\limits_{\Omega} f(x, y, z) \mathrm{d} V = 
        \int_{0}^{2\pi} \mathrm{d} \theta \int_{0}^{\pi} \mathrm{d} \varphi \int_{0}^{\rho(\theta, \varphi)} F(\rho, \theta, \varphi) \rho^2 \sin \varphi \mathrm{d} \rho 
    \]
\end{conclusion}
\subsection{多元函数积分计算技巧}
\begin{conclusion}{选择积分顺序}{integral-order}   
    计算二重积分\(
    \iint f(x, y) \mathrm{d} x \mathrm{d} y
    \)时先对 \(x\) 积分还是先对 \(y\) 积分，既取决于积分区域\(D\)的类型，又取决于被积函数的特点。\\
    不同的积分顺序会影响计算的繁简程度，有时甚至是能否计算出积分的区别。\\
\end{conclusion}

\subsubsection{对称性与奇偶性}
\begin{conclusion}{二重积分的对称性与奇偶性}{two-integral-symmetry}
    设\(f(x, y)\)在有界闭区域\(D\)上连续，则
    \begin{enumerate}
        \item 若\(D\)关于\(x\)轴对称，则
        \[
            \iint\limits_{D} f(x, y) \mathrm{d} x \mathrm{d} y = 
            \begin{cases}
                0, &\begin{aligned}
                    &\text{若} f(x, y) 
                    \text{对} x \text{为奇函数},\text{即}\\
                    &f(x, -y) = -f(x, y), \\
                    &\forall (x, y) \in D\\
                \end{aligned} \\
                2\iint\limits_{D_1} f(x, y) \mathrm{d} x \mathrm{d} y, &
                \begin{aligned}
                   & \text{若} f(x, y) \text{对} y \text{为偶函数},\text{即} \\
                &f(x, -y) = f(x, y), \\
                &\forall (x, y) \in D\\
                \end{aligned}
            \end{cases}
        \]
        其中\(D_1 = D \cap \{y \geq 0\}\)。
        \item 若\(D\)关于\(y\)轴对称，则
        \[
            \iint\limits_{D} f(x, y) \mathrm{d} x \mathrm{d} y = 
            \begin{cases}
                0, & f(x,y) \text{对} x \text{为奇函数},\\
                2\iint\limits_{D_1} f(x, y) \mathrm{d} x \mathrm{d} y, &
                f(x,y) \text{对} x \text{为偶函数},\\
            \end{cases}
        \]
        其中\(D_1 = D \cap \{x \geq 0\}\)。
        \item 若\(D\)关于原点对称，即\(
            (x, y) \in D \Leftrightarrow (-x, -y) \in D
        \)
        \[
            \iint\limits_{D} f(x, y) \mathrm{d} x \mathrm{d} y = 
            \begin{cases}
                0, & \begin{aligned}
                    &\text{若} f(x, y) \text{关于} (x, y)\text{为奇函数},\\
                    &\text{即}f(-x, -y) = -f(x, y), \\
                    &\forall (x, y) \in D
                \end{aligned} \\
                2\iint\limits_{D_1} f(x, y) \mathrm{d} x \mathrm{d} y, &
                \begin{aligned}
                   & \text{若} f(x, y) \text{关于} (x, y)\text{为偶函数},\\
                    &\text{即}f(-x, -y) = f(x, y), \\
                    &\forall (x, y) \in D
                \end{aligned}
            \end{cases}
        \]
        将\(D\)分割成关于原点对称的两部分，即\(D = D_1 \cup D_2\)，其中\(D_1\) 与\(D_2\) 关于原点对称。
        \item 变量的轮换对称性，若\(D\)关于直线\(
            y = x 
        \)对称，即\(
            (x, y) \in D \Leftrightarrow (y, x) \in D
        \),则
        \[
            \iint\limits_{D} f(x, y) \mathrm{d} x \mathrm{d} y = 
            \iint\limits_{D} f(y, x) \mathrm{d} x \mathrm{d} y
        \]
        在这种情形下，若将\(D\)分成两部分，即\(D = D_1 \cup D_2\)，其中\(D_1\) 与\(D_2\) 关于直线\(y = x\)对称，则
        \[
            \iint\limits_{D_1} f(x, y) \mathrm{d} x \mathrm{d} y = 
            \iint\limits_{D_2} f(y, x) \mathrm{d} x \mathrm{d} y
        \]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{三重积分的对称性与奇偶性}{three-integral-symmetry}
    设\(f(x, y, z)\)在有界闭区域\(\Omega\)上连续，若\(\Omega\)关于\(xy\)平面对称(
        \((x, y , z) \in \Omega \Leftrightarrow (x, y , -z) \in \Omega\)
    )，则
    \[
        \iint\limits_{\Omega} f(x, y, z) \mathrm{d} V = 
        \begin{cases}
            0, & \begin{aligned}
                &\text{若} f(x, y, z) \text{对} z\text{为奇函数},\text{即} \\
                &f(x, y, -z) = -f(x, y, z), \\
                &\forall (x, y, z) \in \Omega
            \end{aligned} \\
            \\
            2\iint\limits_{\Omega_1} f(x, y, z) \mathrm{d} V, &
            \begin{aligned}
                &\text{若} f(x, y, z) \text{对} z\text{为偶函数},\text{即}\\
                &f(x, y, -z) = f(x, y, z), \\
                &\forall (x, y, z) \in \Omega
            \end{aligned}
        \end{cases}
    \]
    若\(\Omega\)关于\(yz\)平面(\(zx\)平面)对称，\(f(x, y ,z)\)关于\(x\)或\(y\)为奇函数或偶函数，则同理可得。
\end{conclusion}

\begin{conclusion}{第一类曲线,曲面积分的对称性与奇偶性}{first-curve-integral-symmetry}

        \end{enumerate}
\end{conclusion}

\begin{conclusion}{分块积分法}{integral-block}
    以二重积分为例，若\(D = D_1 \cup D_2\)，其中\(D_1\)与\(D_2\)有公共边界，则
    \[
        \iint\limits_{D} f(x, y) \mathrm{d} x \mathrm{d} y = 
        \iint\limits_{D_1} f(x, y) \mathrm{d} x \mathrm{d} y + 
        \iint\limits_{D_2} f(x, y) \mathrm{d} x \mathrm{d} y
    \]
    分块的情形有
    \begin{enumerate}
        \item 区域的边界线段是分段表示的
        若\(D\)的边界线段是分段表示的，即\(D = D_1 \cup D_2\)，则
        \[
            \iint\limits_{D_1} f(x, y) \mathrm{d} x \mathrm{d} y + 
            \iint\limits_{D_2} f(x, y) \mathrm{d} x \mathrm{d} y
            = \iint\limits_{D} f(x, y) \mathrm{d} x \mathrm{d} y
        \]
        \item 被积函数是分段表示的
        若\(D = D_1 \cup D_2\)，且\(
        f(x, y) = 
        \begin{cases}
            f_1(x, y), & (x, y) \in D_1 \\
            f_2(x, y), & (x, y) \in D_2 \\
        \end{cases}
        \),则
        \[
            \iint\limits_{D} f(x, y) \mathrm{d} x \mathrm{d} y = 
            \iint\limits_{D_1} f_1(x, y) \mathrm{d} x \mathrm{d} y + 
            \iint\limits_{D_2} f_2(x, y) \mathrm{d} x \mathrm{d} y
        \]
        \item 需要将区域分块才可应用重积分化为累次积分公式的情形
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{计算第二类曲面积分时选择投影方向}{projection-direction}

    若\(S\)既可表示为\(z = z(x, y), (x, y) \in D_{xy}\),即\(S\)在\(xy\)平面上的投影区域，也可表示为\(

    \]

\subsection{多元函数积分的各种公式}
\begin{mydef}{平面上的单连通区域与区域的正向边界}{single-region}
    设\(D\)为平面区域,如果\(D\)内任一闭曲线所围部分都属于\(D\),则称
    \(D\)为平面单连通区域。否则称为复连通区域。通俗讲法可以这样说单连通区域不含点洞
    复连通区域含有点洞。\\
    区域的正方向是逆时针方向，区域内的洞的正方向是顺时针方向。
\end{mydef}

\begin{formula}{格林公式}{Green-formula}
    设平面上的有界闭区域\(D\)由分段光滑的曲线\(L\)围成，函数\(
        P(x, y), Q(x, y)\)在\(D\)上有连续的一阶偏导数，则
        \[
            \iint\limits_{D} \left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right)
            \mathrm{d} x \mathrm{d} y = 
            \oint\limits_{L} P \mathrm{d} x + Q \mathrm{d} y 
        \]
        或将\(Q\)换成\(P\),\(P\)换成\(-Q\)得到格林公式的另一种形式
        \[
            \iint\limits_{D} \left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y}\right)
            \mathrm{d} x \mathrm{d} y = 
            \oint\limits_{L} P \mathrm{d} y - Q \mathrm{d} x 
        \]
        其中\(L\)为\(D\)取正向的边界曲线
        对于复连通区域\(D\)，边界曲线不止一条闭曲线，格林公式中的右端积分应包括沿区域\(D\)的全部边界的曲线积分
        若\(D\)为复连通区域，则有
        \[
            \iint\limits_{D} \left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right)
            = \int_{C} P \mathrm{d} x + Q \mathrm{d} y 
            + \sum_{i} \int_{C_i} P \mathrm{d} x + Q \mathrm{d} y 
        \]
        其中\(C\)为\(D\)的正向边界曲线，取逆时针方向，
        \(C_i\)为区域内的洞的正向边界曲线，取顺时针方向。
        若\(C_i\)为逆时针方向，则有
        \[
            \iint\limits_{D} \left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right)
            = \int_{C} P \mathrm{d} x + Q \mathrm{d} y 
            - \sum_{i} \int_{C_i} P \mathrm{d} x + Q \mathrm{d} y 
        \]
        平面上分段光滑闭曲线\(L\)围成的区域\(D\)，则
        \begin{formula}{平面上分段光滑闭曲线\(L\)围成的区域\(D\)的面积公式}{region-formula}
            \[
            \frac{1}{2} \oint_{L^{+}} -y \mathrm{d} x + x \mathrm{d} y 
            \]
            其中，\(L^{+}\)取正向
        \end{formula}
\end{formula}
\begin{formula}{高斯公式}{Gauss-formula}
    设\(\Omega\)是空间中的有界闭区域,由分块光滑的曲面\(S\)围成。函数\(
        P(x, y, z), Q(x, y, z), R(x, y, z)\)在\(\Omega\)上有连续的一阶偏导数.则
        \[
            \iiint\limits_{\Omega} \left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}\right)
            \mathrm{d} V =
            \iint\limits_{S} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x  + R \mathrm{d} x \mathrm{d} y
        \]
        或
        \[
            \iiint\limits_{\Omega} \left(\frac{\partial Q}{\partial z} + \frac{\partial R}{\partial y} + \frac{\partial P}{\partial x}\right)
            \mathrm{d} V =
            \iint\limits_{S} \left(
                P \cos \alpha + Q \cos \beta + R \cos \gamma
            \right)
        \]
        这里\(S\)是\(\Omega\)的整个边界的外侧(即取外法向)，\(
            \cos \alpha, \cos \beta, \cos \gamma
        \)是\(S\)上点\((x, y ,z)\)处的外法向量的方向余弦。
        若\(\Omega\)的边界不止一张曲面，高斯公式中的右端积分应包括沿区域\(\Omega\)的全部边界上的曲面积分。
        若\(\Omega\)的外边界为闭曲面\(\Sigma\),内边界为闭曲面\(\Sigma_1, \dots ,\Sigma_n\),则
        \[
            \iiint\limits_{\Omega} \left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}\right)
            = \iint\limits_{\Sigma} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x  + R \mathrm{d} x \mathrm{d} y
            + \sum_{i} \iint\limits_{\Sigma_i} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x  + R \mathrm{d} x \mathrm{d} y
        \]
        或
        \[
            \iiint\limits_{\Omega} \left(\frac{\partial Q}{\partial z} + \frac{\partial R}{\partial y} + \frac{\partial P}{\partial x}\right)
            = \iint\limits_{\Sigma} \left(P \cos \alpha + Q \cos \beta + R \cos \gamma\right) \mathrm{d} S
            + \sum_{i} \iint\limits_{\Sigma_i} \left(P \cos \alpha + Q \cos \beta + R \cos \gamma\right) \mathrm{d} S
        \]
        \begin{formula}{空间分段光滑闭曲面\(S\)围成的区域\(\Omega\)的体积公式}{region-formula-space}
            \[
            \frac{1}{3} \iint\limits_{S^+} x \mathrm{d} y \mathrm{d} z + y \mathrm{d} z \mathrm{d} x  + z \mathrm{d} x \mathrm{d} y
            \]
            其中\(S^{+}\)取外法向
        \end{formula}
\end{formula}
\begin{formula}{斯托克斯公式}{Stokes-formula}
    设\(\Gamma\)是空间中的分段光滑闭曲线，S是以\(\Gamma\)为边界的分块光滑曲面，\(\Gamma\)的正向与\(S\)的侧(
        即法向量的指向
    )符合右手法则，函数\(
        P(x, y , z), Q(x, y , z), R(x, y , z)\)在\(S\)上有连续的一阶偏导数，则有
        \[
            \begin{aligned}
                &\iint\limits_{S} \left(\frac{\partial R}{\partial y}
                - \frac{\partial Q}{\partial z}
            \right) \mathrm{d} y \mathrm{d} z
            + \left(\frac{\partial P}{\partial z}
                - \frac{\partial R}{\partial x}\right) \mathrm{d} z \mathrm{d} x
            + \left(\frac{\partial Q}{\partial x}
                - \frac{\partial P}{\partial y}\right) \mathrm{d} x \mathrm{d} y \\
            & = \iint\limits_{\Gamma} P \mathrm{d} x + Q \mathrm{d} y + R \mathrm{d} z
            \end{aligned}
        \]
    即
    \[
    \iint\limits_{S} 
        \begin{vmatrix}
            \mathrm{d} y \mathrm{d} z & \mathrm{d} z \mathrm{d} x &  \mathrm{d} x \mathrm{d} y \\
            \frac{{\partial}}{\partial x} & \frac{{\partial}}{\partial y} & \frac{{\partial}}{\partial z}\\
            P & Q & R
        \end{vmatrix} = \int_{\Gamma} P \mathrm{d} x + Q \mathrm{d} y + R \mathrm{d} z
    \]
    或\(
        \iint\limits_{S} 
        \left[
            \left(
                \frac{\partial R}{\partial y} - \frac{\partial Q}{\partial z}
            \right) \cos \alpha
            + \left(
                \frac{\partial P}{\partial z} - \frac{\partial R}{\partial x}
            \right) \cos \beta
            + \left(
                \frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}
            \right) \cos \gamma
        \right] \mathrm{d} S =
        \int_{\Gamma} P \mathrm{d} x + Q \mathrm{d} y + R \mathrm{d} z
    \) \\
    即
    \[
    \iint\limits_{S}\begin{vmatrix}
        \cos \alpha & \cos \beta & \cos \gamma \\
        \frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
        P & Q & R
    \end{vmatrix}  \mathrm{d} S
    = \int_{\Gamma} P \mathrm{d} x + Q \mathrm{d} y + R \mathrm{d} z
    \]
    设\(\Gamma\)为分段光滑的空间有向闭曲线，\(S\)是以\(\Gamma\)为边界的分块光滑曲面，当右手
    外的四指依\(\Gamma\)的绕行方向时，拇指所指的方向与\(S\)上的法向量的指向相同时,这时
    称\(\Gamma\)时有向曲面\(S\)的正向边界曲线
\end{formula}


\begin{conclusion}{格林公式、高斯公式、斯托克斯公式的关系}{Green-Gauss-Stokes}
    格林公式建立了平面上曲线积分与二重积分的连续；\\
    高斯公式建立了曲面积分与三重积分的关系；\\
    斯托克斯公式建立了空间中曲线积分与曲面积分的关系。\\
    斯托克斯公式与高斯公式退化成二维情形，分别是两种不同形式的格林公式。
\end{conclusion}

\begin{mydef}{向量场的通量与散度}{flux-divergence}
    \begin{mydef}{通量}{flux}
        设有向量场\(\mathbf{F}(x, y ,z) = P(x, y ,z)\mathbf{i}
        + Q(x, y ,z)\mathbf{j}
        + R(x, y ,z)\mathbf{k}
        \),则\(\mathbf{F}\)沿分块光滑定向曲面\(S\)的通量为
        \[
        \begin{aligned}
        \int_{S} \mathbf{F}\cdot \mathbf{n} \mathrm{d} S &= 
        \iint_{S} (P \cos \alpha + Q \cos \beta + R \cos \gamma) \mathrm{d} S \\
        &= \iint\limits_{S} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x  + R \mathrm{d} x \mathrm{d} y
        = \iint_{S} \mathbf{F} \cdot \mathrm{d} S
        \end{aligned}
        \]
        其中\(
            \mathbf{n}
        \)是\(S\)上任意点\((x, y ,z)\)处的单位法向量，\(
        \mathbf{n} = (
            \cos \alpha , \cos \beta , \cos \gamma
        )\)
    \end{mydef}

    \begin{mydef}{散度}{divergence}
        设\(\mathbf{F}(x, y ,z) = (
            P(x, y ,z), Q(x, y ,z), R(x, y ,z)
        )\)，在\(\Omega\)有连续的一阶偏导数，则
        \(
            \forall M(x, y, z) \in \Omega
        \),\(
            \mathbf{F}
        \)在点\(M\)的散度如下
        \[
        \mathrm{div} \mathbf{F} =
        \frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}
        \overset{\Delta}{=} \nabla \cdot \mathbf{F}
        \]
        其中\(
        \nabla = \left(\frac{\partial}{\partial x}, \frac{\partial}{\partial y}, \frac{\partial}{\partial z}\right)
        \),称作梯度算符。\\
    \end{mydef}
    \begin{conclusion}{通量与散度的关系}{flux-divergence-relation}
        高斯公式表述了向量场的散度与通量的关系:设\(\Omega\)是空间中有界闭区域，
        边界是分块光滑的有向闭曲面\(S\),取外法向，\(
            \mathbf{F}(x, y ,z) = (
                P(x, y ,z), Q(x, y ,z), R(x, y ,z)
            )
        \)在\(\Omega\)内有连续的一阶偏导数，则
        \[
            \iiint\limits_{\Omega}    
        \mathrm{div} \mathbf{F} \mathrm{d} V 
            = \iint\limits_{S} \mathbf{F} \cdot \mathbf{n} \mathrm{d} S
        \]
    \end{conclusion}
\end{mydef}
\begin{mydef}{向量场的环流量与旋度}{flux-divergence}
    \begin{mydef}{环流量}{circle-flux}
        设有向量场\(\mathbf{F}(x, y ,z) = P(x, y ,z)\mathbf{i}
        + Q(x, y ,z)\mathbf{j}
        + R(x, y ,z)\mathbf{k}
        \),则\(\mathbf{F}\)沿分段光滑定向闭曲线\(L\)的环流量(
            或环量
        )为\[
        \begin{aligned}
            &\oint_{L} \mathbf{F}\cdot \mathbf{\tau} \mathrm{d} s = 
            \oint_{L} \mathbf{F}\cdot \mathbf{\tau} \mathrm{d} s = \\
            &\oint_{L} (P \cos \alpha + Q \cos \beta + R \cos \gamma) \mathrm{d} s \\
            &= 
            \oint_{L} P \mathrm{d} x + Q \mathrm{d} y + R \mathrm{d} z
            = \int_{L} \mathbf{F} \cdot \mathrm{d} s
        \end{aligned}
        \]
        其中\(\tau = (
            \cos \alpha , \cos \beta , \cos \gamma
        )\)是\(\Gamma\)上任意点\((x, y ,z)\)处的单位切向量(
            指向曲线方向
        )
    \end{mydef}
    \begin{mydef}{旋度}{circle-divergence}
        设\(\mathbf{F}(x, y ,z) = (
            P(x, y ,z), Q(x, y ,z), R(x, y ,z)
        )\)，在\(\Omega\)有连续的一阶偏导数，则
        \(
            \forall M(x, y, z) \in \Omega
        \),\(
            \mathbf{F}
        \)在点\(M\)的旋度如下
        \[
        \mathbf{rot} ~ \mathbf{F} =
        \left(\frac{\partial R}{\partial y} - \frac{\partial Q}{\partial z},
        \frac{\partial P}{\partial z} - \frac{\partial R}{\partial x},
        \frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right)
        \overset{\Delta}{=}
        \begin{vmatrix}
            \mathbf{i} & \mathbf{j} & \mathbf{k}\\
            \frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z}\\
            P & Q & R
        \end{vmatrix}
        \overset{\Delta}{=} \nabla \times \mathbf{F}
        \]
    \end{mydef}
    \begin{conclusion}{环流量与散度的关系}{circle-flux-divergence-relation}
        斯托克斯公式表述了向量场的旋度与环流量的关系:即斯托克斯公式可改写为
        \[
            \iint_{S} \mathbf{rot} ~ \mathbf{F} \cdot \mathbf{n} \mathrm{d} s = \int_{\Gamma} \mathbf{F} \cdot \mathbf{\tau} \mathrm{d} s
        \]
        其中\(\tau\)为曲线\(\Gamma\)沿它的方向的单位切向量。
    \end{conclusion}
\end{mydef}

\begin{conclusion}{应用格林公式计算曲线积分}{Green-curve}
    由格林公式\(
        \int_{L} P \mathrm{d} x + Q \mathrm{d} y = 
        \iint_{D} \left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right) \mathrm{d} x \mathrm{d} y
    \),若其中之一易算，另一也就求出
    若平面区域\(D\)由分段光滑曲线\(C\)和\(C_0\)围成，其中\(C_0\)与\(C\)均取逆时针方向
    ,由格林公式得
    \[
        \int_{C} P \mathrm{d} x + Q \mathrm{d} y = 
        \int_{C_0} P \mathrm{d} x + Q \mathrm{d} y = 
        + \iint_{D} \left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right) \mathrm{d} x \mathrm{d} y
    \]
    于是将求\(
        \int_{C} P \mathrm{d} x + Q \mathrm{d} y
    \)转化为求\(
        \int_{C_0} P \mathrm{d} x + Q \mathrm{d} y
    \)与\(\iint\limits_{D} \left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right) \mathrm{d} x \mathrm{d} y\)
    若在\(D\)上\(
        \left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right) = 0
    \),则有
    \[
        \int_{C} P \mathrm{d} x + Q \mathrm{d} y = 
        \int_{C_0} P \mathrm{d} x + Q \mathrm{d} y 
    \]
    于是将求\(
        \int_{C} P \mathrm{d} x + Q \mathrm{d} y
        \)转化为求\(\int_{C_0} P \mathrm{d} x + Q \mathrm{d} y\) \\
    应用格林公式计算曲线积分时应注意以下两个条件
    \begin{enumerate}
        \item \(L\)必须是封闭曲线，若不封闭，要添加辅助线使之封闭，并保证添加得线的线积分可得。
        \item \(P(x,y)\)和\(Q(x,y)\)在\(D\)上连续,若存在奇点，则挖去奇点。
    \end{enumerate}
\end{conclusion}
\begin{conclusion}{应用高斯公式计算曲面积分}{Gauss-surface}
    由高斯公式
    \[
        \iint\limits_{S} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y = 
        \iiint\limits_{\Omega} \left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}\right) \mathrm{d} V
    \]
    若其中之一易算，另一也就求出
    若闭曲面\(\Sigma\)与\(\Sigma_0\)围成的区域\(\Omega\),\(\Sigma\)为\(\Omega\)的外边界，\(
        \Sigma_0
    \)为\(\Omega\)的内边界，则
    \[
    \begin{aligned}
    \iint\limits_{\Sigma} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y = 
        \iiint\limits_{\Omega} \left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}\right) \mathrm{d} V \\
        + \iint\limits_{\Sigma_0} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y
    \end{aligned}
    \]
    其中\(\Sigma\)的法向指向\(\Omega\)的外部，\(\Sigma_0\)的法向指向\(\Omega\)的内部。
    于是将求\(\iint\limits_{\Sigma} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y\)转化为求
    \(\iiint\limits_{\Omega} \left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}\right) \mathrm{d} V\)
    与\(
    \iint\limits_{\Sigma_0} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y
    \)
    若在\(\Omega\)上\(
        \left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}\right) \equiv 0
    \),则有
    \[
        \iint\limits_{\Sigma} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y = 
        \iint\limits_{\Sigma_0} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} z \mathrm{d} x + R \mathrm{d} x \mathrm{d} y
    \]
\end{conclusion}

\begin{conclusion}{应用斯托克斯公式计算曲线积分}{Stokes-curve}
    常用斯托克斯公式来简化某些空间曲线积分的计算。
\end{conclusion}

\begin{mydef}{曲线积分与路径无关问题}{path-independent}
    若\(D\)是一个平面区域，若对\(D\)内任意两点\(A, B\)及\(D\)内从
    \(A\)点到\(B\)的任意两条分段光滑曲线\(L_1, L_2\),有
    \[
        \int_{L_1} P \mathrm{d} x + Q \mathrm{d} y = \int_{L_2} P \mathrm{d} x + Q \mathrm{d} y
    \]
    恒成立，则称此\(
    \int_{L} P \mathrm{d} x + Q \mathrm{d} y
    \)与路径无关。
\end{mydef}

\begin{mydef}{微分式的原函数问题}{indefinite-integral}
    若在区域\(D\)内\(\exists\)存在函数\(u(x, y)\)，使得\(
        \mathrm{d} u = P \mathrm{d} x + Q \mathrm{d} y
    \),则称\(
        u(x, y)
    \)是\(
        P \mathrm{d} x + Q \mathrm{d} y
    \)在区域\(D\)的原函数
    关于原函数，有如下结论
    \begin{enumerate}
        \item 设\(u(x, y)\)是\(
            P \mathrm{d} x + Q \mathrm{d} y
        \)的原函数，则\(
            P \mathrm{d} x + Q \mathrm{d} y 
        \)的全体原函数为\(
            u(x, y) + C
        \),其中\(C\)为\(\forall\)常数。
        \item 设\(P(x,y)\)和\(Q(x,y)\)在区域\(D\)内连续，则
        \[
            \mathrm{d} u = P \mathrm{d} x + Q \mathrm{d} y(
                (x,y) \in D
            ) \leftrightarrow \frac{\partial u}{\partial x} = P, \frac{\partial u}{\partial y} = Q(
                (x,y) \in D
            )
        \]
    \end{enumerate}
\end{mydef}

\begin{conclusion}{\(\int_{L}P(x,y)dx+Q(x,y)dy\)与路径无关时的特征}{path-independent-feature}
    设\(P(x, y)\)和\(Q(x, y)\)在区域\(D\)内连续，若曲线积分\(\int_{L}P(x,y)dx+Q(x,y)dy\)与路径无关，则 \\
    \(
        \Leftrightarrow
    \)对\(D\)内任意分段光滑曲线\(C\)，有
    \(
        \oint_{C} P(x,y)dx+Q(x,y)dy = 0 
    \) \\
    \(
    \Leftrightarrow
    \)在\(D\)内存在函数\(
        u = u(x, y),
    \)使得\(
        \mathrm{d} u(x, y) = P \mathrm{d} x + Q \mathrm{d} y
    (
        \frac{\partial u}{\partial x} = P, \frac{\partial u}{\partial y} = Q
        )
    \)\\
    此时\(u(x, y) = \int^{(x, y)}_{(x_0, y_0)} P \mathrm{d} x + Q \mathrm{d} y + C
    \),其中\(C\)为任意常数
    \newline
    \newline
    设\(P(x, y), Q(x, y)\)在区域\(D\)内有连续的一阶偏导数
    \begin{enumerate}
        \item 若\(\int_{L} P \mathrm{d} x + Q \mathrm{d} y\)与路径无关，则\(
            \frac{\partial Q}{\partial x} = \frac{\partial P}{\partial y}(
                (x, y) \in D
            )
        \)
        \item 若\(D\)是单连通区域,又\(
            \frac{\partial Q}{\partial x} = \frac{\partial P}{\partial y}(
                (x, y) \in D
            )
        \),则\(
            \int_{L} P \mathrm{d} x + Q \mathrm{d} y
        \)在\(D\)内与路径无关。
    \end{enumerate}
\end{conclusion}
\begin{conclusion}{如何判断曲线积分\(\int_{L}P(x,y)dx+Q(x,y)dy\)与路径无关}{path-independent-feature-judgement}
    假设\(P(x, y), Q(x, y)\)在区域\(D\)连续或有连续的一阶偏导数
    \begin{enumerate}
        \item 判断曲线积分与路径无关
        \begin{enumerate}
            \item 存在分段光滑曲线\(C \subset D\),\(
                \int_{C} P \mathrm{d} x + Q \mathrm{d} y \neq 0
            \)
            \item \(
                \exists (x, y) \in D, \frac{\partial Q}{\partial x} \neq \frac{\partial P}{\partial y}
            \)
        \end{enumerate}
        \item 判断曲线积分与路径无关
        \begin{enumerate}
            \item 求得\(u(x, y)\)使得\(
                \mathrm{d} u = P(x, y) \mathrm{d} x + Q(x, y) \mathrm{d} y(
                    \forall (x, y) \in D
                )
            \)
            \item 若\(D\)是单连通区域,又\(
                \frac{\partial Q}{\partial x} = \frac{\partial P}{\partial y}(
                    (x, y) \in D
                )
            \),则\(
                \int_{L} P \mathrm{d} x + Q \mathrm{d} y
            \)在\(D\)内与路径无关。
            \item 若\(D = D_0 \ \{M_0\}\),\(D_0\)是单连通区域，\(M_0 \in D\),若\(
                \frac{\partial Q}{\partial x} = \frac{\partial P}{\partial y}(\forall (x, y) \in D)  
            \),又存在一条分段光滑闭曲线\(C_0\),它包围点\(M_0\),则\(
                \oint_{C_0} P \mathrm{d} x + Q \mathrm{d} y = 0
            \)
        \end{enumerate}
    \end{enumerate}
\end{conclusion}
\begin{conclusion}{积分与路径无关时求\(I = \int_{L}P(x,y)dx+Q(x,y)dy\)}{path-independent-integral}
    设\(P(x, y), Q(x, y)\)在区域\(D\)内连续，又设积分\(\int_{L} P \mathrm{d} x + Q \mathrm{d} y\)与路径无关，则
    可按如下方法求曲线积分\(
        I = \int_{\overset{\frown}{AB}} P \mathrm{d} x + Q \mathrm{d} y\quad (
            A, B \in D
        )
    \)
    \begin{enumerate}
        \item 此时一定\(
            \exists u(x, y)
        \)使得\(
            \mathrm{d} u = P \mathrm{d} x + Q \mathrm{d} y(
                \forall (x, y) \in D
            )
        \),若求得这种\(u\),则
        \[
            I = \int_{\overset{\frown}{AB}} P \mathrm{d} x + Q \mathrm{d} y = 
            u(x, y)|_{A}^{B} = 
            u(B) - u(A)
        \]
        \item 选取特殊的路径代替\(
            \overset{\frown}{AB}
        \)
    \end{enumerate}
\end{conclusion}
\begin{conclusion}{积分与路径无关时求原函数}{path-independent-indefinite}
    设\(P(x, y), Q(x, y)\)在区域\(D\)内连续，判断\(
        P \mathrm{d} x + Q \mathrm{d} y
    \)在\(D\)是否存在原函数等同于判断\(
        \int_{L} P \mathrm{d} x + Q \mathrm{d} y
    \)在\(D\)是否与路径无关，可用以下方法求原函数
    \begin{enumerate}
        \item 不定积分法 \\
        由\(
            \frac{\partial u}{\partial x} = P(x, y)
        \),对\(x\)积分得\(
            u = \int P(x, y) \mathrm{d} x + C(y)
        \),由\(
            \frac{\partial u}{\partial y} = Q(x, y)
            = \frac{\partial}{\partial y} (\int P(x, y) \mathrm{d} x) + C'(y) = Q(x, y)
        \)，求出\(C'(y)\),再求\(C(y)\)即可。 \\
        \item 特殊路径积分法，若在单连通区域\(D\)内\(
            \frac{\partial P}{\partial y} = \frac{\partial Q}{\partial x}
        \),则积分与路径无关 \\
        于是\(u(x, y) = \int_{(x_0 , y_0)}^{x, y} P \mathrm{d} x + Q \mathrm{d} y\)
        即为原函数。\((x_0 , y_0)\)是\(D\)中的定点。\(
        (x, y)\)是\(D\)中的动点,适当选取从\(
            (x_0, y_0)
        \)到\((x, y)\)的路径来求这个积分，若在非单连通区域内\(
            \frac{\partial P}{\partial y} = \frac{\partial Q}{\partial x}\)
            则不能保证积分与路径无关，仍可取特殊路径求出\(
            u(x, y) = \int_{(x_0 , y_0)}^{x, y} P \mathrm{d} x + Q \mathrm{d} y
            \)，但仍需进一步验证,若\(
                \frac{\partial u}{\partial x} = P(x, y)\)和\(
                \frac{\partial u}{\partial y} = Q(x, y)\)在\(D\)成立
            则\(u(x, y)\)就是\(
                P \mathrm{d} x + Q \mathrm{d} y
            \)在\(D\)上的一个原函数，否则\(
                P \mathrm{d} x + Q \mathrm{d} y
            \)在\(D\)就不\(\exists\)原函数
        \item 观察法(凑微分法)
    \end{enumerate}
\end{conclusion}
\subsection{多元函数积分的几何应用}
\begin{conclusion}{空间图形的体积}{space-volume}
    平行截面为已知的体积和旋转体的体积可化为定积分来计算
    对于柱形长条区域的体积
    \begin{enumerate}
        \item 设\(z = z_1(x, y), z = z_2(x, y)\)在有界闭区域\(D\)上连续，\(z_1(x, y) \leq z_2(x, y)\)，\(\Omega\)是由
        曲面\(z = z_1(x, y)\)和\(z = z_2(x, y)\)所围成的空间区域及以\(D\)的边界为准线，母线平行于\(z\)轴的柱面所围成的空间区域(可称为柱形长条区域),则
        \(\Omega\)的体积为
        \[
            V = \iiint\limits_{\Omega} \mathrm{d} V  = 
            \iint\limits_{D} (\int_{z_2(x, y)}{z_1(x, y)} \mathrm{d}z) \mathrm{d} x \mathrm{d} y
            = \iint\limits_{D} [z_2(x, y) - z_1(x, y)] \mathrm{d} x \mathrm{d} y
        \]
        \item 特别地，\(z_1(x, y) = 0, z_2(x, y) = f(x, y)\)，得曲顶柱体的体积
        为\[
            V = \iint\limits_{D} f(x, y) \mathrm{d} x \mathrm{d} y
        \]
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{一般曲面的面积}{surface-area}
    设空间某平面图形的面积为\(A\)，该平面图形在\(xy\)平面上的投影为\(\sigma\),则
    \[
        A = \frac{\sigma}{|\cos \gamma|}
    \]
    其中\(\gamma\)法向量\(\mathbf{n}\)与\(z\)轴正向的夹角。
    \begin{enumerate}
        \item 设曲面\(S\)由方程\(z = f(x, y), (x, y) \in D\)给出，则\(D\)为\(S\)在\(xy\)平面上的投影区域，则
        \(f(x, y)\)在有界闭区域\(D\)由连续的偏导数\(f'(x), f'(y)\),\(S\)上任意点处的单位法向量\(\mathbf{n} =
        (\cos \alpha, \cos \beta, \cos \gamma) = \pm 
        \dfrac{1}{\sqrt{1 + f'(x)^2 + f'(y)^2}}(
            -f'(x), -f'(y), 1
        )\)则\(S\)的面积为
        \[
            A = 
            \iint\limits_{D} \frac{\mathrm{d} \sigma}{\cos \gamma} = 
            \iint\limits_{D} \sqrt{1 + f'(x)^2 + f'(y)^2} \mathrm{d} x \mathrm{d} y
        \]
        \item 若曲面\(S\)的方程为\((x = g(y, z), (y, z) \in D_{yz})\)或\(
            (y = h(x, z), (x, z) \in D_{xz})
        \),则可得类似的计算公式
        \item 柱面被曲面部分所得面积：设有\(Oxy\)平面上的光滑曲线\(L\),以\(L\)为准线，母线平行于\(z\)轴作柱面，此柱面
        在\(xy\)平面与连续曲面\(z = f(x, y)(\geq 0)\)之间的部分的面积为\(
        A = \int_{L} f(x, y) \mathrm{d} s
        \)
    \end{enumerate}
\end{conclusion}

\begin{mydef}{质量、质心、形心与转动惯量}{mass-center-moment}
    \begin{enumerate}
        \item 质量：在空间直角坐标系\(Oxyz\)中，\(n\)个质点组成的质点组\(M_i(x_i, y_i, z_i)\)的质量为\(m_i\),则
        \begin{enumerate}
            \item 质心\(
                (\bar{x}, \bar{y}, \bar{z})
                = \dfrac{
                    m_x, m_y, m_z
                }{M}
                =
                \dfrac{
                (
                    \sum_{i}^{n} m_i x_i, \sum_{i}^{n} m_i y_i, \sum_{i}^{n} m_i z_i
                )
                }
                {\sum_{i}^{n} m_i}
            \)
            \item 关于直线\(L\)与点\(P\)的转动惯量分别为\(
                I_L = \sum_{i}^{n} r_i^2 m_i,
                I_P = \sum_{i}^{n} d_i^2 m_i
            \),其中\(
                r_i, d_i
            \)分别是点\(M_i\)到直线\(L\)与点\(P\)的距离
        \end{enumerate}
        \item 在空间直角坐标系\(Oxyz\)中，物体占据空间区域\(\Omega\)，体密度\(
        \rho(x, y ,z)\),则
        \begin{enumerate}
            \item 质量：\(
                M = \iiint\limits_{\Omega} \rho(x, y, z) \mathrm{d} V
            \)
            \item 质心
            \(
            (\bar{x}, \bar{y}, \bar{z}) =
            \dfrac{
            \iiint\limits_{\Omega} x \rho(x, y, z) \mathrm{d} V,
            \iiint\limits_{\Omega} y \rho(x, y, z) \mathrm{d} V,
            \iiint\limits_{\Omega} z \rho(x, y, z) \mathrm{d} V

                \iiint\limits_{\Omega} \rho \mathrm{d} V
            }
            \)
            \(
                \bar{x}, \bar{y}, \bar{z}
            \)也叫作空间区域\(\Omega\)的形心
            \item 关于\(x, y, z\)轴及原点的转动惯量分别为
            \[
                \begin{aligned}
                &I_x = \iiint\limits_{\Omega} (y^2 + z^2) \rho(x, y, z) \mathrm{d} V,
                I_y = \iiint\limits_{\Omega} (x^2 + z^2) \rho(x, y, z) \mathrm{d} V, \\
                &I_z = \iiint\limits_{\Omega} (x^2 + y^2) \rho(x, y, z) \mathrm{d} V,
                I_o = \iiint\limits_{\Omega} (x^2 + y^2 + z^2) \rho(x, y, z) \mathrm{d} V
                \end{aligned}
            \]
        \end{enumerate}
    \end{enumerate}
\end{mydef}

\begin{conclusion}{引力计算}{gravity-integral}
    \begin{enumerate}
        \item 质点\(M_0(x_0, y_0, z_0), M_i(x_i, y_i, z_i)\)的质量分别为\(m_0, m_i\),则质点\(M_i\) 
        对\(M_0\)的引力为
        \[
            F = \frac{km_0m_i}{r_i^3} r_i
        \]
        其中\(k\)为引力常数，\(r_i = \overrightarrow{M_0 M_i}
        = (x_i - x_0, y_i - y_0, z_i - z_0)\),\\
        \[
            r_i = |r_i| = \sqrt{(x_i - x_0)^2 + (y_i - y_0)^2 + (z_i - z_0)^2}
        \]
        \item 空间曲线\(L\)对外点\(M_0(x_0, y_0, z_0)\)处质量为\(m_0\)的质点的引力为
        \[
            F = \int_{L} km_0\rho(x, y, z) \frac{1}{r^3}(
                x - x_0, y - y_0, z - z_0
            ) \mathrm{d} s
        \]
        其中\(\rho(x, y, z)\)为曲线\(L\)的线密度，
        \item 空间曲线或空间区域对外一点有类似的计算公式
    \end{enumerate}
\end{conclusion}

\begin{conclusion}{功的计算}{work-integral}
    \begin{enumerate}
        \item 点\(M\)在常力\(
            F = (P, Q. R)
        \)作用下由\(
            A(x_A, y_A, z_A)
        \)沿直线运动到\(B(x_B, y_B, z_B)\),则力\(F\)做的功为
        \[  
            W = \mathbf{F} \cdot \overrightarrow{\text{AB}} = 
            = P(x_B - x_A) + Q(y_B - y_A) + R(z_B - z_A)
        \]
        \item 点\(M\)在变力\(
            \mathbf{F}(x, y, z) = (P(x, y, z), Q(x, y, z), R(x, y, z))
            \)作用下由A沿直线运动到B,则力\(F\)做的功为
            \[
                W = \int_{\overset{\frown}{AB}} P(x, y, z) \mathrm{d} x + Q(x, y, z) \mathrm{d} y + R(x, y, z) \mathrm{d} z
            \]
        \item 质点\(\mathbf{M}\)在平面力\(
            \mathbf{F}(x, y) = (
                P(x, y), Q(x, y)
            )
        \)的作用下由\(A\)点沿平面曲线\(L\)运动到B,则力\(\mathbf{F}\)做的功为
        \[
            W = \int_{\overset{\frown}{AB}} P(x, y) \mathrm{d} x + Q(x, y) \mathrm{d} y
        \]
    \end{enumerate}
\end{conclusion}
\begin{conclusion}{流量计算}{flow-integral}
    \begin{enumerate}
        \item 流体在空间流动，流速\(\mathbf{v}\)为常向量,则单位时间内通过定向平面上区域\(D\)的流体体积(即体积流量，简称流量)为
        \[
            q = |\mathbf{v}| \cos \theta \cdot S = \mathbf{v} \cdot \mathbf{n}S
        \]
        其中\(\mathbf{n}\)为定向平面的法向量，\(\theta\)是\(
            \mathbf{v}
        \)与\(
            \mathbf{n}
        \)的夹角,\(S\)是\(D\)的面积
        \item 流体在空间流动，流速为
        \[
            \begin{aligned}
                q &= \iint\limits_{S} \mathbf{v} \cdot \mathbf{n} \mathrm{d} S
                = \iint\limits_{S} (P \cos \alpha + Q \cos \beta + R \cos \gamma) \mathrm{d} S \\
                &= \iint\limits_{S} P \mathrm{d} y \mathrm{d} z + Q \mathrm{d} x \mathrm{d} z + R \mathrm{d} x \mathrm{d} y
            \end{aligned}
        \]
        其中\(\mathbf{n}\)是曲面\(S\)的单位向量,且\(
            \mathbf{n} = (
                \cos \alpha, \cos \beta, \cos \gamma
            )
        \)
    \end{enumerate}
\end{conclusion}
